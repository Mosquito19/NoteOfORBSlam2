143 template <typename Traits>
144 bool BlockSolver<Traits>::buildStructure(bool zeroBlocks)
145 {
146  assert(_optimizer);
147 
148  size_t sparseDim = 0;
149  _numPoses=0;
150  _numLandmarks=0;
151  _sizePoses=0;
152  _sizeLandmarks=0;
153  int* blockPoseIndices = new int[_optimizer->indexMapping().size()];
154  int* blockLandmarkIndices = new int[_optimizer->indexMapping().size()];
155 
156  for (size_t i = 0; i < _optimizer->indexMapping().size(); ++i) {
157  OptimizableGraph::Vertex* v = _optimizer->indexMapping()[i];
158  int dim = v->dimension();
159  if (! v->marginalized()){
160  v->setColInHessian(_sizePoses);
161  _sizePoses+=dim;
162  blockPoseIndices[_numPoses]=_sizePoses;
163  ++_numPoses;
164  } else {
165  v->setColInHessian(_sizeLandmarks);
166  _sizeLandmarks+=dim;
167  blockLandmarkIndices[_numLandmarks]=_sizeLandmarks;
168  ++_numLandmarks;
169  }
170  sparseDim += dim;
171  }
172  resize(blockPoseIndices, _numPoses, blockLandmarkIndices, _numLandmarks, sparseDim);
173  delete[] blockLandmarkIndices;
174  delete[] blockPoseIndices;
175 
176  // allocate the diagonal on Hpp and Hll
177  int poseIdx = 0;
178  int landmarkIdx = 0;
179  for (size_t i = 0; i < _optimizer->indexMapping().size(); ++i) {
180  OptimizableGraph::Vertex* v = _optimizer->indexMapping()[i];
181  if (! v->marginalized()){
182  //assert(poseIdx == v->hessianIndex());
183  PoseMatrixType* m = _Hpp->block(poseIdx, poseIdx, true);
184  if (zeroBlocks)
185  m->setZero();
186  v->mapHessianMemory(m->data());
187  ++poseIdx;
188  } else {
189  LandmarkMatrixType* m = _Hll->block(landmarkIdx, landmarkIdx, true);
190  if (zeroBlocks)
191  m->setZero();
192  v->mapHessianMemory(m->data());
193  ++landmarkIdx;
194  }
195  }
196  assert(poseIdx == _numPoses && landmarkIdx == _numLandmarks);
197 
198  // temporary structures for building the pattern of the Schur complement
199  SparseBlockMatrixHashMap<PoseMatrixType>* schurMatrixLookup = 0;
200  if (_doSchur) {
201  schurMatrixLookup = new SparseBlockMatrixHashMap<PoseMatrixType>(_Hschur->rowBlockIndices(), _Hschur->colBlockIndices());
202  schurMatrixLookup->blockCols().resize(_Hschur->blockCols().size());
203  }
204 
205  // here we assume that the landmark indices start after the pose ones
206  // create the structure in Hpp, Hll and in Hpl
207  for (SparseOptimizer::EdgeContainer::const_iterator it=_optimizer->activeEdges().begin(); it!=_optimizer->activeEdges().end(); ++it){
208  OptimizableGraph::Edge* e = *it;
209 
210  for (size_t viIdx = 0; viIdx < e->vertices().size(); ++viIdx) {
211  OptimizableGraph::Vertex* v1 = (OptimizableGraph::Vertex*) e->vertex(viIdx);
212  int ind1 = v1->hessianIndex();
213  if (ind1 == -1)
214  continue;
215  int indexV1Bak = ind1;
216  for (size_t vjIdx = viIdx + 1; vjIdx < e->vertices().size(); ++vjIdx) {
217  OptimizableGraph::Vertex* v2 = (OptimizableGraph::Vertex*) e->vertex(vjIdx);
218  int ind2 = v2->hessianIndex();
219  if (ind2 == -1)
220  continue;
221  ind1 = indexV1Bak;
222  bool transposedBlock = ind1 > ind2;
223  if (transposedBlock){ // make sure, we allocate the upper triangle block
224  swap(ind1, ind2);
225  }
226  if (! v1->marginalized() && !v2->marginalized()){
227  PoseMatrixType* m = _Hpp->block(ind1, ind2, true);
228  if (zeroBlocks)
229  m->setZero();
230  e->mapHessianMemory(m->data(), viIdx, vjIdx, transposedBlock);
231  if (_Hschur) {// assume this is only needed in case we solve with the schur complement
232  schurMatrixLookup->addBlock(ind1, ind2);
233  }
234  } else if (v1->marginalized() && v2->marginalized()){
235  // RAINER hmm.... should we ever reach this here????
236  LandmarkMatrixType* m = _Hll->block(ind1-_numPoses, ind2-_numPoses, true);
237  if (zeroBlocks)
238  m->setZero();
239  e->mapHessianMemory(m->data(), viIdx, vjIdx, false);
240  } else {
241  if (v1->marginalized()){
242  PoseLandmarkMatrixType* m = _Hpl->block(v2->hessianIndex(),v1->hessianIndex()-_numPoses, true);
243  if (zeroBlocks)
244  m->setZero();
245  e->mapHessianMemory(m->data(), viIdx, vjIdx, true); // transpose the block before writing to it
246  } else {
247  PoseLandmarkMatrixType* m = _Hpl->block(v1->hessianIndex(),v2->hessianIndex()-_numPoses, true);
248  if (zeroBlocks)
249  m->setZero();
250  e->mapHessianMemory(m->data(), viIdx, vjIdx, false); // directly the block
251  }
252  }
253  }
254  }
255  }
256 
257  if (! _doSchur)
258  return true;
259 
260  _DInvSchur->diagonal().resize(landmarkIdx);
261  _Hpl->fillSparseBlockMatrixCCS(*_HplCCS);
262 
263  for (size_t i = 0; i < _optimizer->indexMapping().size(); ++i) {
264  OptimizableGraph::Vertex* v = _optimizer->indexMapping()[i];
265  if (v->marginalized()){
266  const HyperGraph::EdgeSet& vedges=v->edges();
267  for (HyperGraph::EdgeSet::const_iterator it1=vedges.begin(); it1!=vedges.end(); ++it1){
268  for (size_t i=0; i<(*it1)->vertices().size(); ++i)
269  {
270  OptimizableGraph::Vertex* v1= (OptimizableGraph::Vertex*) (*it1)->vertex(i);
271  if (v1->hessianIndex()==-1 || v1==v)
272  continue;
273  for (HyperGraph::EdgeSet::const_iterator it2=vedges.begin(); it2!=vedges.end(); ++it2){
274  for (size_t j=0; j<(*it2)->vertices().size(); ++j)
275  {
276  OptimizableGraph::Vertex* v2= (OptimizableGraph::Vertex*) (*it2)->vertex(j);
277  if (v2->hessianIndex()==-1 || v2==v)
278  continue;
279  int i1=v1->hessianIndex();
280  int i2=v2->hessianIndex();
281  if (i1<=i2) {
282  schurMatrixLookup->addBlock(i1, i2);
283  }
284  }
285  }
286  }
287  }
288  }
289  }
290 
291  _Hschur->takePatternFromHash(*schurMatrixLookup);
292  delete schurMatrixLookup;
293  _Hschur->fillSparseBlockMatrixCCSTransposed(*_HschurTransposedCCS);
294 
295  return true;
296 }

298 template <typename Traits>
299 bool BlockSolver<Traits>::updateStructure(const std::vector<HyperGraph::Vertex*>& vset, const HyperGraph::EdgeSet& edges)
300 {
301  for (std::vector<HyperGraph::Vertex*>::const_iterator vit = vset.begin(); vit != vset.end(); ++vit) {
302  OptimizableGraph::Vertex* v = static_cast<OptimizableGraph::Vertex*>(*vit);
303  int dim = v->dimension();
304  if (! v->marginalized()){
305  v->setColInHessian(_sizePoses);
306  _sizePoses+=dim;
307  _Hpp->rowBlockIndices().push_back(_sizePoses);
308  _Hpp->colBlockIndices().push_back(_sizePoses);
309  _Hpp->blockCols().push_back(typename SparseBlockMatrix<PoseMatrixType>::IntBlockMap());
310  ++_numPoses;
311  int ind = v->hessianIndex();
312  PoseMatrixType* m = _Hpp->block(ind, ind, true);
313  v->mapHessianMemory(m->data());
314  } else {
315  std::cerr << "updateStructure(): Schur not supported" << std::endl;
316  abort();
317  }
318  }
319  resizeVector(_sizePoses + _sizeLandmarks);
320 
321  for (HyperGraph::EdgeSet::const_iterator it = edges.begin(); it != edges.end(); ++it) {
322  OptimizableGraph::Edge* e = static_cast<OptimizableGraph::Edge*>(*it);
323 
324  for (size_t viIdx = 0; viIdx < e->vertices().size(); ++viIdx) {
325  OptimizableGraph::Vertex* v1 = (OptimizableGraph::Vertex*) e->vertex(viIdx);
326  int ind1 = v1->hessianIndex();
327  int indexV1Bak = ind1;
328  if (ind1 == -1)
329  continue;
330  for (size_t vjIdx = viIdx + 1; vjIdx < e->vertices().size(); ++vjIdx) {
331  OptimizableGraph::Vertex* v2 = (OptimizableGraph::Vertex*) e->vertex(vjIdx);
332  int ind2 = v2->hessianIndex();
333  if (ind2 == -1)
334  continue;
335  ind1 = indexV1Bak;
336  bool transposedBlock = ind1 > ind2;
337  if (transposedBlock) // make sure, we allocate the upper triangular block
338  swap(ind1, ind2);
339 
340  if (! v1->marginalized() && !v2->marginalized()) {
341  PoseMatrixType* m = _Hpp->block(ind1, ind2, true);
342  e->mapHessianMemory(m->data(), viIdx, vjIdx, transposedBlock);
343  } else {
344  std::cerr << __PRETTY_FUNCTION__ << ": not supported" << std::endl;
345  }
346  }
347  }
348 
349  }
350 
351  return true;
352 }

354 template <typename Traits>
355 bool BlockSolver<Traits>::solve(){
356  //cerr << __PRETTY_FUNCTION__ << endl;
357  if (! _doSchur){
358  double t=get_monotonic_time();
359  bool ok = _linearSolver->solve(*_Hpp, _x, _b);
360  G2OBatchStatistics* globalStats = G2OBatchStatistics::globalStats();
361  if (globalStats) {
362  globalStats->timeLinearSolver = get_monotonic_time() - t;
363  globalStats->hessianDimension = globalStats->hessianPoseDimension = _Hpp->cols();
364  }
365  return ok;
366  }
367 
368  // schur thing
369 
370  // backup the coefficient matrix
371  double t=get_monotonic_time();
372 
373  // _Hschur = _Hpp, but keeping the pattern of _Hschur
374  _Hschur->clear();
375  _Hpp->add(_Hschur);
376 
377  //_DInvSchur->clear();
378  memset (_coefficients, 0, _sizePoses*sizeof(double));
379 # ifdef G2O_OPENMP
380 # pragma omp parallel for default (shared) schedule(dynamic, 10)
381 # endif
382  for (int landmarkIndex = 0; landmarkIndex < static_cast<int>(_Hll->blockCols().size()); ++landmarkIndex) {
383  const typename SparseBlockMatrix<LandmarkMatrixType>::IntBlockMap& marginalizeColumn = _Hll->blockCols()[landmarkIndex];
384  assert(marginalizeColumn.size() == 1 && "more than one block in _Hll column");
385 
386  // calculate inverse block for the landmark
387  const LandmarkMatrixType * D = marginalizeColumn.begin()->second;
388  assert (D && D->rows()==D->cols() && "Error in landmark matrix");
389  LandmarkMatrixType& Dinv = _DInvSchur->diagonal()[landmarkIndex];
390  Dinv = D->inverse();
391 
392  LandmarkVectorType db(D->rows());
393  for (int j=0; j<D->rows(); ++j) {
394  db[j]=_b[_Hll->rowBaseOfBlock(landmarkIndex) + _sizePoses + j];
395  }
396  db=Dinv*db;
397 
398  assert((size_t)landmarkIndex < _HplCCS->blockCols().size() && "Index out of bounds");
399  const typename SparseBlockMatrixCCS<PoseLandmarkMatrixType>::SparseColumn& landmarkColumn = _HplCCS->blockCols()[landmarkIndex];
400 
401  for (typename SparseBlockMatrixCCS<PoseLandmarkMatrixType>::SparseColumn::const_iterator it_outer = landmarkColumn.begin();
402  it_outer != landmarkColumn.end(); ++it_outer) {
403  int i1 = it_outer->row;
404 
405  const PoseLandmarkMatrixType* Bi = it_outer->block;
406  assert(Bi);
407 
408  PoseLandmarkMatrixType BDinv = (*Bi)*(Dinv);
409  assert(_HplCCS->rowBaseOfBlock(i1) < _sizePoses && "Index out of bounds");
410  typename PoseVectorType::MapType Bb(&_coefficients[_HplCCS->rowBaseOfBlock(i1)], Bi->rows());
411 # ifdef G2O_OPENMP
412  ScopedOpenMPMutex mutexLock(&_coefficientsMutex[i1]);
413 # endif
414  Bb.noalias() += (*Bi)*db;
415 
416  assert(i1 >= 0 && i1 < static_cast<int>(_HschurTransposedCCS->blockCols().size()) && "Index out of bounds");
417  typename SparseBlockMatrixCCS<PoseMatrixType>::SparseColumn::iterator targetColumnIt = _HschurTransposedCCS->blockCols()[i1].begin();
418 
419  typename SparseBlockMatrixCCS<PoseLandmarkMatrixType>::RowBlock aux(i1, 0);
420  typename SparseBlockMatrixCCS<PoseLandmarkMatrixType>::SparseColumn::const_iterator it_inner = lower_bound(landmarkColumn.begin(), landmarkColumn.end(), aux);
421  for (; it_inner != landmarkColumn.end(); ++it_inner) {
422  int i2 = it_inner->row;
423  const PoseLandmarkMatrixType* Bj = it_inner->block;
424  assert(Bj);
425  while (targetColumnIt->row < i2 /*&& targetColumnIt != _HschurTransposedCCS->blockCols()[i1].end()*/)
426  ++targetColumnIt;
427  assert(targetColumnIt != _HschurTransposedCCS->blockCols()[i1].end() && targetColumnIt->row == i2 && "invalid iterator, something wrong with the matrix structure");
428  PoseMatrixType* Hi1i2 = targetColumnIt->block;//_Hschur->block(i1,i2);
429  assert(Hi1i2);
430  (*Hi1i2).noalias() -= BDinv*Bj->transpose();
431  }
432  }
433  }
434  //cerr << "Solve [marginalize] = " << get_monotonic_time()-t << endl;
435 
436  // _bschur = _b for calling solver, and not touching _b
437  memcpy(_bschur, _b, _sizePoses * sizeof(double));
438  for (int i=0; i<_sizePoses; ++i){
439  _bschur[i]-=_coefficients[i];
440  }
441 
442  G2OBatchStatistics* globalStats = G2OBatchStatistics::globalStats();
443  if (globalStats){
444  globalStats->timeSchurComplement = get_monotonic_time() - t;
445  }
446 
447  t=get_monotonic_time();
448  bool solvedPoses = _linearSolver->solve(*_Hschur, _x, _bschur);
449  if (globalStats) {
450  globalStats->timeLinearSolver = get_monotonic_time() - t;
451  globalStats->hessianPoseDimension = _Hpp->cols();
452  globalStats->hessianLandmarkDimension = _Hll->cols();
453  globalStats->hessianDimension = globalStats->hessianPoseDimension + globalStats->hessianLandmarkDimension;
454  }
455  //cerr << "Solve [decompose and solve] = " << get_monotonic_time()-t << endl;
456 
457  if (! solvedPoses)
458  return false;
459 
460  // _x contains the solution for the poses, now applying it to the landmarks to get the new part of the
461  // solution;
462  double* xp = _x;
463  double* cp = _coefficients;
464 
465  double* xl=_x+_sizePoses;
466  double* cl=_coefficients + _sizePoses;
467  double* bl=_b+_sizePoses;
468 
469  // cp = -xp
470  for (int i=0; i<_sizePoses; ++i)
471  cp[i]=-xp[i];
472 
473  // cl = bl
474  memcpy(cl,bl,_sizeLandmarks*sizeof(double));
475 
476  // cl = bl - Bt * xp
477  //Bt->multiply(cl, cp);
478  _HplCCS->rightMultiply(cl, cp);
479 
480  // xl = Dinv * cl
481  memset(xl,0, _sizeLandmarks*sizeof(double));
482  _DInvSchur->multiply(xl,cl);
483  //_DInvSchur->rightMultiply(xl,cl);
484  //cerr << "Solve [landmark delta] = " << get_monotonic_time()-t << endl;
485 
486  return true;
487 }

502 template <typename Traits>
503 bool BlockSolver<Traits>::buildSystem()
504 {
505  // clear b vector
506 # ifdef G2O_OPENMP
507 # pragma omp parallel for default (shared) if (_optimizer->indexMapping().size() > 1000)
508 # endif
509  for (int i = 0; i < static_cast<int>(_optimizer->indexMapping().size()); ++i) {
510  OptimizableGraph::Vertex* v=_optimizer->indexMapping()[i];
511  assert(v);
512  v->clearQuadraticForm();
513  }
514  _Hpp->clear();
515  if (_doSchur) {
516  _Hll->clear();
517  _Hpl->clear();
518  }
519 
520  // resetting the terms for the pairwise constraints
521  // built up the current system by storing the Hessian blocks in the edges and vertices
522 # ifndef G2O_OPENMP
523  // no threading, we do not need to copy the workspace
524  JacobianWorkspace& jacobianWorkspace = _optimizer->jacobianWorkspace();
525 # else
526  // if running with threads need to produce copies of the workspace for each thread
527  JacobianWorkspace jacobianWorkspace = _optimizer->jacobianWorkspace();
528 # pragma omp parallel for default (shared) firstprivate(jacobianWorkspace) if (_optimizer->activeEdges().size() > 100)
529 # endif
530  for (int k = 0; k < static_cast<int>(_optimizer->activeEdges().size()); ++k) {
531  OptimizableGraph::Edge* e = _optimizer->activeEdges()[k];
532  e->linearizeOplus(jacobianWorkspace); // jacobian of the nodes' oplus (manifold)
533  e->constructQuadraticForm();
534 # ifndef NDEBUG
535  for (size_t i = 0; i < e->vertices().size(); ++i) {
536  const OptimizableGraph::Vertex* v = static_cast<const OptimizableGraph::Vertex*>(e->vertex(i));
537  if (! v->fixed()) {
538  bool hasANan = arrayHasNaN(jacobianWorkspace.workspaceForVertex(i), e->dimension() * v->dimension());
539  if (hasANan) {
540  cerr << "buildSystem(): NaN within Jacobian for edge " << e << " for vertex " << i << endl;
541  break;
542  }
543  }
544  }
545 # endif
546  }
547 
548  // flush the current system in a sparse block matrix
549 # ifdef G2O_OPENMP
550 # pragma omp parallel for default (shared) if (_optimizer->indexMapping().size() > 1000)
551 # endif
552  for (int i = 0; i < static_cast<int>(_optimizer->indexMapping().size()); ++i) {
553  OptimizableGraph::Vertex* v=_optimizer->indexMapping()[i];
554  int iBase = v->colInHessian();
555  if (v->marginalized())
556  iBase+=_sizePoses;
557  v->copyB(_b+iBase);
558  }
559 
560  return 0;
561 }

564 template <typename Traits>
