292 {
293  // 该优化函数主要用于Tracking线程中：运动跟踪、参考帧跟踪、地图跟踪、重定位
294

310  g2o::VertexSE3Expmap * vSE3 = new g2o::VertexSE3Expmap();
311  vSE3->setEstimate(Converter::toSE3Quat(pFrame->mTcw));

327  vector<size_t> vnIndexEdgeStereo;
328  vpEdgesStereo.reserve(N);
329  vnIndexEdgeStereo.reserve(N);
330 
331  const float deltaMono = sqrt(5.991);
332  const float deltaStereo = sqrt(7.815);
333 
334  // 步骤3：添加一元边：相机投影模型
335  {
336  unique_lock<mutex> lock(MapPoint::mGlobalMutex);
337 
338  for(int i=0; i<N; i++)
339  {
340  MapPoint* pMP = pFrame->mvpMapPoints[i];
341  if(pMP)
342  {
343  // Monocular observation
344  // 单目情况, 也有可能在双目下, 当前帧的左兴趣点找不到匹配的右兴趣点
345  if(pFrame->mvuRight[i]<0)
346  {
347  nInitialCorrespondences++;
348  pFrame->mvbOutlier[i] = false;
349 
350  Eigen::Matrix<double,2,1> obs;
351  const cv::KeyPoint &kpUn = pFrame->mvKeysUn[i];
352  obs << kpUn.pt.x, kpUn.pt.y;
353 
354  g2o::EdgeSE3ProjectXYZOnlyPose* e = new g2o::EdgeSE3ProjectXYZOnlyPose();
355 
356  e->setVertex(0, dynamic_cast<g2o::OptimizableGraph::Vertex*>(optimizer.vertex(0)));
357  e->setMeasurement(obs);
358  const float invSigma2 = pFrame->mvInvLevelSigma2[kpUn.octave];
359  e->setInformation(Eigen::Matrix2d::Identity()*invSigma2);
360 
361  g2o::RobustKernelHuber* rk = new g2o::RobustKernelHuber;
362  e->setRobustKernel(rk);
363  rk->setDelta(deltaMono);
364 
365  e->fx = pFrame->fx;
366  e->fy = pFrame->fy;
367  e->cx = pFrame->cx;
368  e->cy = pFrame->cy;

372  e->Xw[2] = Xw.at<float>(2);

378  }

381  nInitialCorrespondences++;

399  e->setRobustKernel(rk);
400  rk->setDelta(deltaStereo);
401 
402  e->fx = pFrame->fx;
403  e->fy = pFrame->fy;
404  e->cx = pFrame->cx;
405  e->cy = pFrame->cy;
406  e->bf = pFrame->mbf;

408  e->Xw[0] = Xw.at<float>(0);
409  e->Xw[1] = Xw.at<float>(1);

423  if(nInitialCorrespondences<3)
424  return 0;

426  // We perform 4 optimizations, after each optimization we classify observation as inlier/outlier
427  // At the next optimization, outliers are not included, but at the end they can be classified as inliers again.
428  // 步骤4：开始优化，总共优化四次，每次优化后，将观测分为outlier和inlier，outlier不参与下次优化
429  // 由于每次优化后是对所有的观测进行outlier和inlier判别，因此之前被判别为outlier有可能变成inlier，反之亦然
430  // 基于卡方检验计算出的阈值（假设测量有一个像素的偏差）
431  const float chi2Mono[4]={5.991,5.991,5.991,5.991};
432  const float chi2Stereo[4]={7.815,7.815,7.815, 7.815};
433  const int its[4]={10,10,10,10};// 四次迭代，每次迭代的次数
434 
435  int nBad=0;
436  for(size_t it=0; it<4; it++)
437  {
438 
439  vSE3->setEstimate(Converter::toSE3Quat(pFrame->mTcw));
440  optimizer.initializeOptimization(0);// 对level为0的边进行优化
441  optimizer.optimize(its[it]);
442 
443  nBad=0;
444  for(size_t i=0, iend=vpEdgesMono.size(); i<iend; i++)
445  {
446  g2o::EdgeSE3ProjectXYZOnlyPose* e = vpEdgesMono[i];
447 
448  const size_t idx = vnIndexEdgeMono[i];
449 
450  if(pFrame->mvbOutlier[idx])
451  {
452  e->computeError(); // NOTE g2o只会计算active edge的误差
453  }
454 
455  const float chi2 = e->chi2();
456 
457  if(chi2>chi2Mono[it])
458  {
459  pFrame->mvbOutlier[idx]=true;
460  e->setLevel(1); // 设置为outlier
461  nBad++;
462  }
463  else
464  {
465  pFrame->mvbOutlier[idx]=false;
466  e->setLevel(0); // 设置为inlier
467  }
468 
469  if(it==2)
470  e->setRobustKernel(0); // 除了前两次优化需要RobustKernel以外, 其余的优化都不需要
471  }
472 
473  for(size_t i=0, iend=vpEdgesStereo.size(); i<iend; i++)
474  {
475  g2o::EdgeStereoSE3ProjectXYZOnlyPose* e = vpEdgesStereo[i];
476 
477  const size_t idx = vnIndexEdgeStereo[i];
478 
479  if(pFrame->mvbOutlier[idx])
480  {
481  e->computeError();
482  }
483 
484  const float chi2 = e->chi2();
485 
486  if(chi2>chi2Stereo[it])
487  {
488  pFrame->mvbOutlier[idx]=true;
489  e->setLevel(1);
490  nBad++;
491  }
492  else
493  {
494  e->setLevel(0);
495  pFrame->mvbOutlier[idx]=false;
496  }
497 
498  if(it==2)
499  e->setRobustKernel(0);
500  }
501 
502  if(optimizer.edges().size()<10)
503  break;
504  }
505 
506  // Recover optimized pose and return number of inliers
507  g2o::VertexSE3Expmap* vSE3_recov = static_cast<g2o::VertexSE3Expmap*>(optimizer.vertex(0));
508  g2o::SE3Quat SE3quat_recov = vSE3_recov->estimate();
509  cv::Mat pose = Converter::toCvMat(SE3quat_recov);
510  pFrame->SetPose(pose);
511 
512  return nInitialCorrespondences-nBad;
513 }
