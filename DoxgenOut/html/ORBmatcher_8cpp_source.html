166 bool ORBmatcher::CheckDistEpipolarLine(const cv::KeyPoint &kp1,const cv::KeyPoint &kp2,const cv::Mat &F12,const KeyFrame* pKF2)
167 {
168  // Epipolar line in second image l = x1'F12 = [a b c]
169  // 求出kp1在pKF2上对应的极线
170  const float a = kp1.pt.x*F12.at<float>(0,0)+kp1.pt.y*F12.at<float>(1,0)+F12.at<float>(2,0);
171  const float b = kp1.pt.x*F12.at<float>(0,1)+kp1.pt.y*F12.at<float>(1,1)+F12.at<float>(2,1);
172  const float c = kp1.pt.x*F12.at<float>(0,2)+kp1.pt.y*F12.at<float>(1,2)+F12.at<float>(2,2);
173 
174  // 计算kp2特征点到极线的距离：
175  // 极线l：ax + by + c = 0
176  // (u,v)到l的距离为： |au+bv+c| / sqrt(a^2+b^2)
177 
178  const float num = a*kp2.pt.x+b*kp2.pt.y+c;
179 
180  const float den = a*a+b*b;
181 
182  if(den==0)
183  return false;
184 
185  const float dsqr = num*num/den;
186 
187  // 尺度越大，范围应该越大。
188  // 金字塔最底层一个像素就占一个像素，在倒数第二层，一个像素等于最底层1.2个像素（假设金字塔尺度为1.2）
189  return dsqr<3.84*pKF2->mvLevelSigma2[kp2.octave];

352 int ORBmatcher::SearchByProjection(KeyFrame* pKF, cv::Mat Scw, const vector<MapPoint*> &vpPoints, vector<MapPoint*> &vpMatched, int th)
353 {
354  // Get Calibration Parameters for later projection
355  const float &fx = pKF->fx;
356  const float &fy = pKF->fy;
357  const float &cx = pKF->cx;
358  const float &cy = pKF->cy;
359 
360  // Decompose Scw
361  cv::Mat sRcw = Scw.rowRange(0,3).colRange(0,3);
362  const float scw = sqrt(sRcw.row(0).dot(sRcw.row(0)));// 计算得到尺度s
363  cv::Mat Rcw = sRcw/scw;
364  cv::Mat tcw = Scw.rowRange(0,3).col(3)/scw;// pKF坐标系下，世界坐标系到pKF的位移，方向由世界坐标系指向pKF
365  cv::Mat Ow = -Rcw.t()*tcw;// 世界坐标系下，pKF到世界坐标系的位移（世界坐标系原点相对pKF的位置），方向由pKF指向世界坐标系
366 
367  // Set of MapPoints already found in the KeyFrame
368  // 使用set类型，并去除没有匹配的点，用于快速检索某个MapPoint是否有匹配
369  set<MapPoint*> spAlreadyFound(vpMatched.begin(), vpMatched.end());
370  spAlreadyFound.erase(static_cast<MapPoint*>(NULL));
371 
372  int nmatches=0;
373 
374  // For each Candidate MapPoint Project and Match
375  // 遍历所有的MapPoints
376  for(int iMP=0, iendMP=vpPoints.size(); iMP<iendMP; iMP++)
377  {
378  MapPoint* pMP = vpPoints[iMP];
379 
380  // Discard Bad MapPoints and already found
381  // 丢弃坏的MapPoints和已经匹配上的MapPoints
382  if(pMP->isBad() || spAlreadyFound.count(pMP))
383  continue;
384 
385  // Get 3D Coords.
386  cv::Mat p3Dw = pMP->GetWorldPos();
387 
388  // Transform into Camera Coords.
389  cv::Mat p3Dc = Rcw*p3Dw+tcw;
390 
391  // Depth must be positive
392  if(p3Dc.at<float>(2)<0.0)
393  continue;
394 
395  // Project into Image
396  const float invz = 1/p3Dc.at<float>(2);
397  const float x = p3Dc.at<float>(0)*invz;
398  const float y = p3Dc.at<float>(1)*invz;
399 
400  const float u = fx*x+cx;
401  const float v = fy*y+cy;
402 
403  // Point must be inside the image
404  if(!pKF->IsInImage(u,v))
405  continue;
406 
407  // Depth must be inside the scale invariance region of the point
408  // 判断距离是否在尺度协方差范围内
409  const float maxDistance = pMP->GetMaxDistanceInvariance();
410  const float minDistance = pMP->GetMinDistanceInvariance();
411  cv::Mat PO = p3Dw-Ow;
412  const float dist = cv::norm(PO);
413 
414  if(dist<minDistance || dist>maxDistance)
415  continue;
416 
417  // Viewing angle must be less than 60 deg
418  cv::Mat Pn = pMP->GetNormal();
419 
420  if(PO.dot(Pn)<0.5*dist)
421  continue;
422 
423  int nPredictedLevel = pMP->PredictScale(dist,pKF);
424 
425  // Search in a radius
426  // 根据尺度确定搜索半径
427  const float radius = th*pKF->mvScaleFactors[nPredictedLevel];
428 
429  const vector<size_t> vIndices = pKF->GetFeaturesInArea(u,v,radius);
430 
431  if(vIndices.empty())
432  continue;
433 
434  // Match to the most similar keypoint in the radius
435  const cv::Mat dMP = pMP->GetDescriptor();
436 
437  int bestDist = 256;
438  int bestIdx = -1;
439  // 遍历搜索区域内所有特征点，与该MapPoint的描述子进行匹配
440  for(vector<size_t>::const_iterator vit=vIndices.begin(), vend=vIndices.end(); vit!=vend; vit++)
441  {
442  const size_t idx = *vit;
443  if(vpMatched[idx])
444  continue;
445 
446  const int &kpLevel= pKF->mvKeysUn[idx].octave;
447 
448  if(kpLevel<nPredictedLevel-1 || kpLevel>nPredictedLevel)
449  continue;
450 
451  const cv::Mat &dKF = pKF->mDescriptors.row(idx);
452 
453  const int dist = DescriptorDistance(dMP,dKF);
454 
455  if(dist<bestDist)
456  {
457  bestDist = dist;
458  bestIdx = idx;
459  }
460  }
461 
462  // 该MapPoint与bestIdx对应的特征点匹配成功
463  if(bestDist<=TH_LOW)
464  {
465  vpMatched[bestIdx]=pMP;
466  nmatches++;
467  }
468 
469  }
470 
471  return nmatches;
