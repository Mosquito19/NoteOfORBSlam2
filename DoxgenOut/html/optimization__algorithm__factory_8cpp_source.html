103  void OptimizationAlgorithmFactory::listSolvers(std::ostream& os) const
104  {
105  size_t solverNameColumnLength = 0;
106  for (CreatorList::const_iterator it = _creator.begin(); it != _creator.end(); ++it)
107  solverNameColumnLength = std::max(solverNameColumnLength, (*it)->property().name.size());
108  solverNameColumnLength += 4;
109 
110  for (CreatorList::const_iterator it = _creator.begin(); it != _creator.end(); ++it) {
111  const OptimizationAlgorithmProperty& sp = (*it)->property();
112  os << sp.name;
113  for (size_t i = sp.name.size(); i < solverNameColumnLength; ++i)
114  os << ' ';
115  os << sp.desc << endl;
116  }
117  }
