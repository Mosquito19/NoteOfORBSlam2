47  //取出所有的地图点
48  const vector<MapPoint*> &vpMPs = mpMap->GetAllMapPoints();

50  const vector<MapPoint*> &vpRefMPs = mpMap->GetReferenceMapPoints();
51 
52  //将vpRefMPs从vector容器类型转化为set容器类型，便于使用set::count快速统计
53  set<MapPoint*> spRefMPs(vpRefMPs.begin(), vpRefMPs.end());
54 
55  if(vpMPs.empty())
56  return;
57 
58  // for AllMapPoints
59  //显示所有的地图点（不包括局部地图点），大小为2个像素，黑色
60  glPointSize(mPointSize);
61  glBegin(GL_POINTS);
62  glColor3f(0.0,0.0,0.0);
63 
64  for(size_t i=0, iend=vpMPs.size(); i<iend;i++)
65  {
66  // 不包括ReferenceMapPoints（局部地图点）
67  if(vpMPs[i]->isBad() || spRefMPs.count(vpMPs[i]))
68  continue;
69  cv::Mat pos = vpMPs[i]->GetWorldPos();
70  glVertex3f(pos.at<float>(0),pos.at<float>(1),pos.at<float>(2));
71  }
72  glEnd();
73 
74  // for ReferenceMapPoints
75  //显示局部地图点，大小为2个像素，红色
76  glPointSize(mPointSize);
77  glBegin(GL_POINTS);
78  glColor3f(1.0,0.0,0.0);
79 
80  for(set<MapPoint*>::iterator sit=spRefMPs.begin(), send=spRefMPs.end(); sit!=send; sit++)
81  {
82  if((*sit)->isBad())
83  continue;
84  cv::Mat pos = (*sit)->GetWorldPos();
85  glVertex3f(pos.at<float>(0),pos.at<float>(1),pos.at<float>(2));
86 
87  }
88  glEnd();
89 }

92 void MapDrawer::DrawKeyFrames(const bool bDrawKF, const bool bDrawGraph)
93 {
94  //历史关键帧图标：宽度占总宽度比例为0.05
95  const float &w = mKeyFrameSize;
96  const float h = w*0.75;
97  const float z = w*0.6;
98 
99  //步骤1：取出所有的关键帧
100  const vector<KeyFrame*> vpKFs = mpMap->GetAllKeyFrames();
101 
102  //步骤2：显示所有关键帧图标
103  //通过显示界面选择是否显示历史关键帧图标
104  if(bDrawKF)
105  {
106  for(size_t i=0; i<vpKFs.size(); i++)
107  {
108  KeyFrame* pKF = vpKFs[i];
109  //转置, OpenGL中的矩阵为列优先存储
110  cv::Mat Twc = pKF->GetPoseInverse().t();
111 
112  glPushMatrix();
113 
114  //（由于使用了glPushMatrix函数，因此当前帧矩阵为世界坐标系下的单位矩阵）
115  //因为OpenGL中的矩阵为列优先存储，因此实际为Tcw，即相机在世界坐标下的位姿
116  glMultMatrixf(Twc.ptr<GLfloat>(0));
117 
118  //设置绘制图形时线的宽度
119  glLineWidth(mKeyFrameLineWidth);
120  //设置当前颜色为蓝色(关键帧图标显示为蓝色)
121  glColor3f(0.0f,0.0f,1.0f);
122  //用线将下面的顶点两两相连
123  glBegin(GL_LINES);
124  glVertex3f(0,0,0);
125  glVertex3f(w,h,z);
126  glVertex3f(0,0,0);
127  glVertex3f(w,-h,z);
128  glVertex3f(0,0,0);
129  glVertex3f(-w,-h,z);
130  glVertex3f(0,0,0);
131  glVertex3f(-w,h,z);
132 
133  glVertex3f(w,h,z);
134  glVertex3f(w,-h,z);
135 
136  glVertex3f(-w,h,z);
137  glVertex3f(-w,-h,z);
138 
139  glVertex3f(-w,h,z);
140  glVertex3f(w,h,z);
141 
142  glVertex3f(-w,-h,z);
143  glVertex3f(w,-h,z);
144  glEnd();
145 
146  glPopMatrix();
147  }
148  }
149 
150  //步骤3：显示所有关键帧的Essential Graph
151  //通过显示界面选择是否显示关键帧连接关系
152  if(bDrawGraph)
153  {
154  //设置绘制图形时线的宽度
155  glLineWidth(mGraphLineWidth);
156  //设置共视图连接线为绿色，透明度为0.6f
157  glColor4f(0.0f,1.0f,0.0f,0.6f);
158  glBegin(GL_LINES);
159 
160  for(size_t i=0; i<vpKFs.size(); i++)
161  {
162  // Covisibility Graph
163  //步骤3.1 共视程度比较高的共视关键帧用线连接
164  //遍历每一个关键帧，得到它们共视程度比较高的关键帧
165  const vector<KeyFrame*> vCovKFs = vpKFs[i]->GetCovisiblesByWeight(100);
166  //遍历每一个关键帧，得到它在世界坐标系下的相机坐标
167  cv::Mat Ow = vpKFs[i]->GetCameraCenter();
168  if(!vCovKFs.empty())
169  {
170  for(vector<KeyFrame*>::const_iterator vit=vCovKFs.begin(), vend=vCovKFs.end(); vit!=vend; vit++)
171  {
172  if((*vit)->mnId<vpKFs[i]->mnId)
173  continue;
174  cv::Mat Ow2 = (*vit)->GetCameraCenter();
175  glVertex3f(Ow.at<float>(0),Ow.at<float>(1),Ow.at<float>(2));
176  glVertex3f(Ow2.at<float>(0),Ow2.at<float>(1),Ow2.at<float>(2));
177  }
178  }
179 
180  // Spanning tree
181  //步骤3.2 连接最小生成树
182  KeyFrame* pParent = vpKFs[i]->GetParent();
183  if(pParent)
184  {
185  cv::Mat Owp = pParent->GetCameraCenter();
186  glVertex3f(Ow.at<float>(0),Ow.at<float>(1),Ow.at<float>(2));
187  glVertex3f(Owp.at<float>(0),Owp.at<float>(1),Owp.at<float>(2));
188  }
189 
190  // Loops
191  //步骤3.3 连接闭环时形成的连接关系
192  set<KeyFrame*> sLoopKFs = vpKFs[i]->GetLoopEdges();
193  for(set<KeyFrame*>::iterator sit=sLoopKFs.begin(), send=sLoopKFs.end(); sit!=send; sit++)
194  {
195  if((*sit)->mnId<vpKFs[i]->mnId)
196  continue;
197  cv::Mat Owl = (*sit)->GetCameraCenter();
198  glVertex3f(Ow.at<float>(0),Ow.at<float>(1),Ow.at<float>(2));
199  glVertex3f(Owl.at<float>(0),Owl.at<float>(1),Owl.at<float>(2));
200  }
201  }
202 
203  glEnd();
204  }

266 void MapDrawer::GetCurrentOpenGLCameraMatrix(pangolin::OpenGlMatrix &M)
267 {
268  if(!mCameraPose.empty())
269  {
270  cv::Mat Rwc(3,3,CV_32F);
271  cv::Mat twc(3,1,CV_32F);
272  {
273  unique_lock<mutex> lock(mMutexCamera);
274  Rwc = mCameraPose.rowRange(0,3).colRange(0,3).t();
275  twc = -Rwc*mCameraPose.rowRange(0,3).col(3);
276  }
277 
278  M.m[0] = Rwc.at<float>(0,0);
279  M.m[1] = Rwc.at<float>(1,0);
280  M.m[2] = Rwc.at<float>(2,0);
281  M.m[3] = 0.0;
282 
283  M.m[4] = Rwc.at<float>(0,1);
284  M.m[5] = Rwc.at<float>(1,1);
285  M.m[6] = Rwc.at<float>(2,1);
286  M.m[7] = 0.0;
287 
288  M.m[8] = Rwc.at<float>(0,2);
289  M.m[9] = Rwc.at<float>(1,2);
290  M.m[10] = Rwc.at<float>(2,2);
291  M.m[11] = 0.0;
292 
293  M.m[12] = twc.at<float>(0);
294  M.m[13] = twc.at<float>(1);
295  M.m[14] = twc.at<float>(2);
296  M.m[15] = 1.0;
297  }
298  else
299  M.SetIdentity();
