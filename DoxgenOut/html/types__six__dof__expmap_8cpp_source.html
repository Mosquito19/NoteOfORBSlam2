112 void EdgeSE3ProjectXYZ::linearizeOplus() {
113  VertexSE3Expmap * vj = static_cast<VertexSE3Expmap *>(_vertices[1]);
114  SE3Quat T(vj->estimate());
115  VertexSBAPointXYZ* vi = static_cast<VertexSBAPointXYZ*>(_vertices[0]);
116  Vector3d xyz = vi->estimate();
117  Vector3d xyz_trans = T.map(xyz);
118 
119  double x = xyz_trans[0];
120  double y = xyz_trans[1];
121  double z = xyz_trans[2];
122  double z_2 = z*z;
123 
124  Matrix<double,2,3> tmp;
125  tmp(0,0) = fx;
126  tmp(0,1) = 0;
127  tmp(0,2) = -x/z*fx;
128 
129  tmp(1,0) = 0;
130  tmp(1,1) = fy;
131  tmp(1,2) = -y/z*fy;
132 
133  // 2*3
134  _jacobianOplusXi = -1./z * tmp * T.rotation().toRotationMatrix();
135 
136  // 2*6 the same as EdgeSE3ProjectXYZOnlyPose::linearizeOplus()
137  _jacobianOplusXj(0,0) = x*y/z_2 *fx;
138  _jacobianOplusXj(0,1) = -(1+(x*x/z_2)) *fx;
139  _jacobianOplusXj(0,2) = y/z *fx;
140  _jacobianOplusXj(0,3) = -1./z *fx;
141  _jacobianOplusXj(0,4) = 0;
142  _jacobianOplusXj(0,5) = x/z_2 *fx;
143 
144  _jacobianOplusXj(1,0) = (1+y*y/z_2) *fy;
145  _jacobianOplusXj(1,1) = -x*y/z_2 *fy;
146  _jacobianOplusXj(1,2) = -x/z *fy;
147  _jacobianOplusXj(1,3) = 0;
148  _jacobianOplusXj(1,4) = -1./z *fy;
149  _jacobianOplusXj(1,5) = y/z_2 *fy;
150 }

200  VertexSE3Expmap * vj = static_cast<VertexSE3Expmap *>(_vertices[1]);
201  SE3Quat T(vj->estimate());
202  VertexSBAPointXYZ* vi = static_cast<VertexSBAPointXYZ*>(_vertices[0]);
203  Vector3d xyz = vi->estimate();
204  Vector3d xyz_trans = T.map(xyz);
205 
206  const Matrix3d R = T.rotation().toRotationMatrix();
207 
208  double x = xyz_trans[0];
209  double y = xyz_trans[1];
210  double z = xyz_trans[2];
211  double z_2 = z*z;
212 
213  _jacobianOplusXi(0,0) = -fx*R(0,0)/z+fx*x*R(2,0)/z_2;
214  _jacobianOplusXi(0,1) = -fx*R(0,1)/z+fx*x*R(2,1)/z_2;
215  _jacobianOplusXi(0,2) = -fx*R(0,2)/z+fx*x*R(2,2)/z_2;
216 
217  _jacobianOplusXi(1,0) = -fy*R(1,0)/z+fy*y*R(2,0)/z_2;
218  _jacobianOplusXi(1,1) = -fy*R(1,1)/z+fy*y*R(2,1)/z_2;
219  _jacobianOplusXi(1,2) = -fy*R(1,2)/z+fy*y*R(2,2)/z_2;
220 
221  _jacobianOplusXi(2,0) = _jacobianOplusXi(0,0)-bf*R(2,0)/z_2;
222  _jacobianOplusXi(2,1) = _jacobianOplusXi(0,1)-bf*R(2,1)/z_2;
223  _jacobianOplusXi(2,2) = _jacobianOplusXi(0,2)-bf*R(2,2)/z_2;
224 
225  _jacobianOplusXj(0,0) = x*y/z_2 *fx;
226  _jacobianOplusXj(0,1) = -(1+(x*x/z_2)) *fx;
227  _jacobianOplusXj(0,2) = y/z *fx;
228  _jacobianOplusXj(0,3) = -1./z *fx;
229  _jacobianOplusXj(0,4) = 0;
230  _jacobianOplusXj(0,5) = x/z_2 *fx;
231 
232  _jacobianOplusXj(1,0) = (1+y*y/z_2) *fy;
233  _jacobianOplusXj(1,1) = -x*y/z_2 *fy;
234  _jacobianOplusXj(1,2) = -x/z *fy;
235  _jacobianOplusXj(1,3) = 0;
236  _jacobianOplusXj(1,4) = -1./z *fy;
237  _jacobianOplusXj(1,5) = y/z_2 *fy;
238 
239  _jacobianOplusXj(2,0) = _jacobianOplusXj(0,0)-bf*y/z_2;
240  _jacobianOplusXj(2,1) = _jacobianOplusXj(0,1)+bf*x/z_2;
241  _jacobianOplusXj(2,2) = _jacobianOplusXj(0,2);
