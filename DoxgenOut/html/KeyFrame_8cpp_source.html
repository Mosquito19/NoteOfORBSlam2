76 void KeyFrame::SetPose(const cv::Mat &Tcw_)
77 {
78  unique_lock<mutex> lock(mMutexPose);
79  Tcw_.copyTo(Tcw);
80  cv::Mat Rcw = Tcw.rowRange(0,3).colRange(0,3);
81  cv::Mat tcw = Tcw.rowRange(0,3).col(3);
82  cv::Mat Rwc = Rcw.t();
83  Ow = -Rwc*tcw;
84 
85  Twc = cv::Mat::eye(4,4,Tcw.type());
86  Rwc.copyTo(Twc.rowRange(0,3).colRange(0,3));
87  Ow.copyTo(Twc.rowRange(0,3).col(3));
88  // center为相机坐标系（左目）下，立体相机中心的坐标
89  // 立体相机中心点坐标与左目相机坐标之间只是在x轴上相差mHalfBaseline,
90  // 因此可以看出，立体相机中两个摄像头的连线为x轴，正方向为左目相机指向右目相机
91  cv::Mat center = (cv::Mat_<float>(4,1) << mHalfBaseline, 0 , 0, 1);
92  // 世界坐标系下，左目相机中心到立体相机中心的向量，方向由左目相机指向立体相机中心
93  Cw = Twc*center;
94 }

161 void KeyFrame::UpdateBestCovisibles()
162 {
163  unique_lock<mutex> lock(mMutexConnections);
164  // http://stackoverflow.com/questions/3389648/difference-between-stdliststdpair-and-stdmap-in-c-stl
165  vector<pair<int,KeyFrame*> > vPairs;
166  vPairs.reserve(mConnectedKeyFrameWeights.size());
167  // 取出所有连接的关键帧，mConnectedKeyFrameWeights的类型为std::map<KeyFrame*,int>，而vPairs变量将共视的3D点数放在前面，利于排序
168  for(map<KeyFrame*,int>::iterator mit=mConnectedKeyFrameWeights.begin(), mend=mConnectedKeyFrameWeights.end(); mit!=mend; mit++)
169  vPairs.push_back(make_pair(mit->second,mit->first));
170 
171  // 按照权重进行排序
172  sort(vPairs.begin(),vPairs.end());
173  list<KeyFrame*> lKFs; // keyframe
174  list<int> lWs; // weight
175  for(size_t i=0, iend=vPairs.size(); i<iend;i++)
176  {
177  lKFs.push_front(vPairs[i].second);
178  lWs.push_front(vPairs[i].first);
179  }
180 
181  // 权重从大到小
182  mvpOrderedConnectedKeyFrames = vector<KeyFrame*>(lKFs.begin(),lKFs.end());
183  mvOrderedWeights = vector<int>(lWs.begin(), lWs.end());
184 }

230 vector<KeyFrame*> KeyFrame::GetCovisiblesByWeight(const int &w)
231 {
232  unique_lock<mutex> lock(mMutexConnections);
233 
234  if(mvpOrderedConnectedKeyFrames.empty())
235  return vector<KeyFrame*>();
236

238  // 从mvOrderedWeights找出第一个大于w的那个迭代器

369 void KeyFrame::UpdateConnections()
370 {
371  // 在没有执行这个函数前，关键帧只和MapPoints之间有连接关系，这个函数可以更新关键帧之间的连接关系
372 
373  //===============1==================================
374  map<KeyFrame*,int> KFcounter; // 关键帧-权重，权重为其它关键帧与当前关键帧共视3d点的个数
375 
376  vector<MapPoint*> vpMP;
377 
378  {
379  // 获得该关键帧的所有3D点
380  unique_lock<mutex> lockMPs(mMutexFeatures);
381  vpMP = mvpMapPoints;
382  }
383 
384  //For all map points in keyframe check in which other keyframes are they seen
385  //Increase counter for those keyframes
386  // 通过3D点间接统计可以观测到这些3D点的所有关键帧之间的共视程度
387  // 即统计每一个关键帧都有多少关键帧与它存在共视关系，统计结果放在KFcounter
388  for(vector<MapPoint*>::iterator vit=vpMP.begin(), vend=vpMP.end(); vit!=vend; vit++)
389  {
390  MapPoint* pMP = *vit;
391 
392  if(!pMP)
393  continue;
394 
395  if(pMP->isBad())
396  continue;
397 
398  // 对于每一个MapPoint点，observations记录了可以观测到该MapPoint的所有关键帧
399  map<KeyFrame*,size_t> observations = pMP->GetObservations();
400 
401  for(map<KeyFrame*,size_t>::iterator mit=observations.begin(), mend=observations.end(); mit!=mend; mit++)
402  {
403  // 除去自身，自己与自己不算共视
404  if(mit->first->mnId==mnId)//pkfini帧
405  continue;
406  KFcounter[mit->first]++;
407  }
409  }
410 
411  // This should not happen
412  if(KFcounter.empty())
413  return;
414 
415  //===============2==================================
416  // If the counter is greater than threshold add connection
417  // In case no keyframe counter is over threshold add the one with maximum counter
418  int nmax=0;
419  KeyFrame* pKFmax=NULL;
420  int th = 15;
421 
422  // vPairs记录与其它关键帧共视帧数大于th的关键帧
423  // pair<int,KeyFrame*>将关键帧的权重写在前面，关键帧写在后面方便后面排序
424  vector<pair<int,KeyFrame*> > vPairs;
425  vPairs.reserve(KFcounter.size());
426  for(map<KeyFrame*,int>::iterator mit=KFcounter.begin(), mend=KFcounter.end(); mit!=mend; mit++)
427  {
428  if(mit->second>nmax)
429  {
430  nmax=mit->second;
431  // 找到对应权重最大的关键帧（共视程度最高的关键帧）
432  pKFmax=mit->first;
433  }
434  if(mit->second>=th)
435  {
436  // 对应权重需要大于阈值，对这些关键帧建立连接
437  vPairs.push_back(make_pair(mit->second,mit->first));
438  // 更新KFcounter中该关键帧的mConnectedKeyFrameWeights
439  // 更新其它KeyFrame的mConnectedKeyFrameWeights，更新其它关键帧与当前帧的连接权重
440  (mit->first)->AddConnection(this,mit->second);
441  }
442  }
443 
444  // 如果没有超过阈值的权重，则对权重最大的关键帧建立连接
445  if(vPairs.empty())
446  {
447  // 如果每个关键帧与它共视的关键帧的个数都少于th，
448  // 那就只更新与其它关键帧共视程度最高的关键帧的mConnectedKeyFrameWeights
449  // 这是对之前th这个阈值可能过高的一个补丁
450  vPairs.push_back(make_pair(nmax,pKFmax));
451  pKFmax->AddConnection(this,nmax);
452  }
453 
454  // vPairs里存的都是相互共视程度比较高的关键帧和共视权重，由大到小
455  sort(vPairs.begin(),vPairs.end());
456  list<KeyFrame*> lKFs;
457  list<int> lWs;
458  for(size_t i=0; i<vPairs.size();i++)
459  {
460  lKFs.push_front(vPairs[i].second);
461  lWs.push_front(vPairs[i].first);
462  }
463 
464  //===============3==================================
465  {
466  unique_lock<mutex> lockCon(mMutexConnections);
467 
468  // mspConnectedKeyFrames = spConnectedKeyFrames;
469  // 更新图的连接(权重)
470  mConnectedKeyFrameWeights = KFcounter;//更新该KeyFrame的mConnectedKeyFrameWeights，更新当前帧与其它关键帧的连接权重
471  mvpOrderedConnectedKeyFrames = vector<KeyFrame*>(lKFs.begin(),lKFs.end());
472  mvOrderedWeights = vector<int>(lWs.begin(), lWs.end());
473 
474  // 更新生成树的连接
475  if(mbFirstConnection && mnId!=0)
476  {
477  // 初始化该关键帧的父关键帧为共视程度最高的那个关键帧
478  mpParent = mvpOrderedConnectedKeyFrames.front();
479  // 建立双向连接关系
480  mpParent->AddChild(this);
481  mbFirstConnection = false;
482  }
483  }
484 }

560 void KeyFrame::SetBadFlag()
561 {
562  {
563  unique_lock<mutex> lock(mMutexConnections);
564  if(mnId==0)
565  return;
566  else if(mbNotErase)// mbNotErase表示不应该擦除该KeyFrame，于是把mbToBeErased置为true，表示已经擦除了，其实没有擦除
567  {
568  mbToBeErased = true;
569  return;
570  }
571  }
572 
573  for(map<KeyFrame*,int>::iterator mit = mConnectedKeyFrameWeights.begin(), mend=mConnectedKeyFrameWeights.end(); mit!=mend; mit++)
574  mit->first->EraseConnection(this);// 让其它的KeyFrame删除与自己的联系
575 
576  for(size_t i=0; i<mvpMapPoints.size(); i++)
577  if(mvpMapPoints[i])
578  mvpMapPoints[i]->EraseObservation(this);// 让与自己有联系的MapPoint删除与自己的联系
579 
580  {
581  unique_lock<mutex> lock(mMutexConnections);
582  unique_lock<mutex> lock1(mMutexFeatures);
583 
584  //清空自己与其它关键帧之间的联系
585  mConnectedKeyFrameWeights.clear();
586  mvpOrderedConnectedKeyFrames.clear();
587 
588  // Update Spanning Tree
589  set<KeyFrame*> sParentCandidates;
590  sParentCandidates.insert(mpParent);
591 
592  // Assign at each iteration one children with a parent (the pair with highest covisibility weight)
593  // Include that children as new parent candidate for the rest
594  // 如果这个关键帧有自己的孩子关键帧，告诉这些子关键帧，它们的父关键帧不行了，赶紧找新的父关键帧
595  while(!mspChildrens.empty())
596  {
597  bool bContinue = false;
598 
599  int max = -1;
600  KeyFrame* pC;
601  KeyFrame* pP;
602 
603  // 遍历每一个子关键帧，让它们更新它们指向的父关键帧
604  for(set<KeyFrame*>::iterator sit=mspChildrens.begin(), send=mspChildrens.end(); sit!=send; sit++)
605  {
606  KeyFrame* pKF = *sit;
607  if(pKF->isBad())
608  continue;
609 
610  // Check if a parent candidate is connected to the keyframe
611  // 子关键帧遍历每一个与它相连的关键帧（共视关键帧）
612  vector<KeyFrame*> vpConnected = pKF->GetVectorCovisibleKeyFrames();
613  for(size_t i=0, iend=vpConnected.size(); i<iend; i++)
614  {
615  for(set<KeyFrame*>::iterator spcit=sParentCandidates.begin(), spcend=sParentCandidates.end(); spcit!=spcend; spcit++)
616  {
617  // 如果该帧的子节点和父节点（祖孙节点）之间存在连接关系（共视）
618  // 举例：B-->A（B的父节点是A） C-->B（C的父节点是B） D--C（D与C相连） E--C（E与C相连） F--C（F与C相连） D-->A（D的父节点是A） E-->A（E的父节点是A）
619  // 现在B挂了，于是C在与自己相连的D、E、F节点中找到父节点指向A的D
620  // 此过程就是为了找到可以替换B的那个节点。
621  // 上面例子中，B为当前要设置为SetBadFlag的关键帧
622  // A为spcit，也即sParentCandidates
623  // C为pKF,pC，也即mspChildrens中的一个
624  // D、E、F为vpConnected中的变量，由于C与D间的权重 比 C与E间的权重大，因此D为pP
625  if(vpConnected[i]->mnId == (*spcit)->mnId)
626  {
627  int w = pKF->GetWeight(vpConnected[i]);
628  if(w>max)
629  {
630  pC = pKF;
631  pP = vpConnected[i];
632  max = w;
633  bContinue = true;
634  }
635  }

757  // 在ComputeStereoMatches函数中应该对校正后的特征点求深度？？ (wubo???)

759  const float v = mvKeys[i].pt.y;
760  const float x = (u-cx)*z*invfx;
761  const float y = (v-cy)*z*invfy;
762  cv::Mat x3Dc = (cv::Mat_<float>(3,1) << x, y, z);
763

766  // Twc为相机坐标系到世界坐标系的变换矩阵
767  // Twc.rosRange(0,3).colRange(0,3)取Twc矩阵的前3行与前3列

779 float KeyFrame::ComputeSceneMedianDepth(const int q)
780 {
781  vector<MapPoint*> vpMapPoints;
782  cv::Mat Tcw_;
783  {
784  unique_lock<mutex> lock(mMutexFeatures);
785  unique_lock<mutex> lock2(mMutexPose);
786  vpMapPoints = mvpMapPoints;
787  Tcw_ = Tcw.clone();
788  }
789 
790  vector<float> vDepths;
791  vDepths.reserve(N);
792  cv::Mat Rcw2 = Tcw_.row(2).colRange(0,3);
793  Rcw2 = Rcw2.t();
794  float zcw = Tcw_.at<float>(2,3);
795  for(int i=0; i<N; i++)
796  {
797  if(mvpMapPoints[i])
798  {
799  MapPoint* pMP = mvpMapPoints[i];
800  cv::Mat x3Dw = pMP->GetWorldPos();
801  float z = Rcw2.dot(x3Dw)+zcw; // (R*x3Dw+t)的第三行，即z
802  vDepths.push_back(z);
803  }
804  }
805 
806  sort(vDepths.begin(),vDepths.end());
807 
808  return vDepths[(vDepths.size()-1)/q];
