54 // pangolin库的文档：http://docs.ros.org/fuerte/api/pangolin_wrapper/html/namespacepangolin.html
55 void Viewer::Run()
56 {
57  //这个变量配合SetFinish函数用于指示该函数是否执行完毕
58  mbFinished = false;
59 
60  pangolin::CreateWindowAndBind("ORB-SLAM2: Map Viewer",1024,768);
61 
62  // 3D Mouse handler requires depth testing to be enabled
63  // 启动深度测试，OpenGL只绘制最前面的一层，绘制时检查当前像素前面是否有别的像素，如果别的像素挡住了它，那它就不会绘制
64  glEnable(GL_DEPTH_TEST);
65 
66  // Issue specific OpenGl we might need
67  // 在OpenGL中使用颜色混合
68  glEnable(GL_BLEND);
69  // 选择混合选项
70  glBlendFunc(GL_SRC_ALPHA, GL_ONE_MINUS_SRC_ALPHA);
71 
72  // 新建按钮和选择框，第一个参数为按钮的名字，第二个为默认状态，第三个为是否有选择框
73  pangolin::CreatePanel("menu").SetBounds(0.0,1.0,0.0,pangolin::Attach::Pix(175));
74  pangolin::Var<bool> menuFollowCamera("menu.Follow Camera",true,true);
75  pangolin::Var<bool> menuShowPoints("menu.Show Points",true,true);
76  pangolin::Var<bool> menuShowKeyFrames("menu.Show KeyFrames",true,true);
77  pangolin::Var<bool> menuShowGraph("menu.Show Graph",true,true);
78  pangolin::Var<bool> menuLocalizationMode("menu.Localization Mode",false,true);
79  pangolin::Var<bool> menuReset("menu.Reset",false,false);
80 
81  // Define Camera Render Object (for view / scene browsing)
82  // 定义相机投影模型：ProjectionMatrix(w, h, fu, fv, u0, v0, zNear, zFar)
83  // 定义观测方位向量：观测点位置：(mViewpointX mViewpointY mViewpointZ)
84  // 观测目标位置：(0, 0, 0)
85  // 观测的方位向量：(0.0,-1.0, 0.0)
86  pangolin::OpenGlRenderState s_cam(
87  pangolin::ProjectionMatrix(1024,768,mViewpointF,mViewpointF,512,389,0.1,1000),
88  pangolin::ModelViewLookAt(mViewpointX,mViewpointY,mViewpointZ, 0,0,0,0.0,-1.0, 0.0)
89  );
90 
91  // Add named OpenGL viewport to window and provide 3D Handler
92  // 定义显示面板大小，orbslam中有左右两个面板，昨天显示一些按钮，右边显示图形
93  // 前两个参数（0.0, 1.0）表明宽度和面板纵向宽度和窗口大小相同
94  // 中间两个参数（pangolin::Attach::Pix(175), 1.0）表明右边所有部分用于显示图形
95  // 最后一个参数（-1024.0f/768.0f）为显示长宽比
96  pangolin::View& d_cam = pangolin::CreateDisplay()
97  .SetBounds(0.0, 1.0, pangolin::Attach::Pix(175), 1.0, -1024.0f/768.0f)
98  .SetHandler(new pangolin::Handler3D(s_cam));
99 
100  pangolin::OpenGlMatrix Twc;
101  Twc.SetIdentity();
102 
103  cv::namedWindow("ORB-SLAM2: Current Frame");
104 
105  bool bFollow = true;
106  bool bLocalizationMode = false;
107 
108  while(1)
109  {
110  // 清除缓冲区中的当前可写的颜色缓冲 和 深度缓冲
111  glClear(GL_COLOR_BUFFER_BIT | GL_DEPTH_BUFFER_BIT);
112 
113  // 步骤1：得到最新的相机位姿
114  mpMapDrawer->GetCurrentOpenGLCameraMatrix(Twc);
115 
116  // 步骤2：根据相机的位姿调整视角
117  // menuFollowCamera为按钮的状态，bFollow为真实的状态
118  if(menuFollowCamera && bFollow)
119  {
120  s_cam.Follow(Twc);
121  }
122  else if(menuFollowCamera && !bFollow)
123  {
124  s_cam.SetModelViewMatrix(pangolin::ModelViewLookAt(mViewpointX,mViewpointY,mViewpointZ, 0,0,0,0.0,-1.0, 0.0));
125  s_cam.Follow(Twc);
126  bFollow = true;
127  }
128  else if(!menuFollowCamera && bFollow)
129  {
130  bFollow = false;
131  }
132 
133  if(menuLocalizationMode && !bLocalizationMode)
134  {
135  mpSystem->ActivateLocalizationMode();
136  bLocalizationMode = true;
137  }
138  else if(!menuLocalizationMode && bLocalizationMode)
139  {
140  mpSystem->DeactivateLocalizationMode();
141  bLocalizationMode = false;
142  }
143 
144  d_cam.Activate(s_cam);
145  // 步骤3：绘制地图和图像
146  // 设置为白色，glClearColor(red, green, blue, alpha），数值范围(0, 1)
147  glClearColor(1.0f,1.0f,1.0f,1.0f);
148  mpMapDrawer->DrawCurrentCamera(Twc);
149  if(menuShowKeyFrames || menuShowGraph)
150  mpMapDrawer->DrawKeyFrames(menuShowKeyFrames,menuShowGraph);
151  if(menuShowPoints)
152  mpMapDrawer->DrawMapPoints();
153 
154  pangolin::FinishFrame();
155 
156  cv::Mat im = mpFrameDrawer->DrawFrame();
157  cv::imshow("ORB-SLAM2: Current Frame",im);
158  cv::waitKey(mT);
159 
160  if(menuReset)
161  {
162  menuShowGraph = true;
163  menuShowKeyFrames = true;
164  menuShowPoints = true;
165  menuLocalizationMode = false;
166  if(bLocalizationMode)
167  mpSystem->DeactivateLocalizationMode();
168  bLocalizationMode = false;
169  bFollow = true;
170  menuFollowCamera = true;
171  mpSystem->Reset();
172  menuReset = false;
173  }
174 
175  if(Stop())
176  {
177  while(isStopped())
178  {
179  //usleep(3000);
180  std::this_thread::sleep_for(std::chrono::milliseconds(3));
181 
182  }
183  }
184 
185  if(CheckFinish())
186  break;
187  }
188 
189  SetFinish();
190 }
