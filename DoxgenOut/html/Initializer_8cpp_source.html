162 {
163  // Number of putative matches
164  const int N = mvMatches12.size();
165 
166  // Normalize coordinates

184  vector<bool> vbCurrentInliers(N,false);

186 
187  // Perform all RANSAC iterations and save the solution with highest score
188  //迭代200次，保存一个最好的值...觉得有点奇怪呢.....
189  for(int it=0; it<mMaxIterations; it++)
190  {
191  // Select a minimum set

295 {
296  const int N = vP1.size();
297 
298  cv::Mat A(2*N,9,CV_32F); // 2N*9 16*9;
299 
300  for(int i=0; i<N; i++)
301  {
302  const float u1 = vP1[i].x;
303  const float v1 = vP1[i].y;
304  const float u2 = vP2[i].x;
305  const float v2 = vP2[i].y;
306 
307  A.at<float>(2*i,0) = 0.0;
308  A.at<float>(2*i,1) = 0.0;
309  A.at<float>(2*i,2) = 0.0;
310  A.at<float>(2*i,3) = -u1;
311  A.at<float>(2*i,4) = -v1;
312  A.at<float>(2*i,5) = -1;
313  A.at<float>(2*i,6) = v2*u1;
314  A.at<float>(2*i,7) = v2*v1;
315  A.at<float>(2*i,8) = v2;
316 
317  A.at<float>(2*i+1,0) = u1;
318  A.at<float>(2*i+1,1) = v1;
319  A.at<float>(2*i+1,2) = 1;
320  A.at<float>(2*i+1,3) = 0.0;
321  A.at<float>(2*i+1,4) = 0.0;
322  A.at<float>(2*i+1,5) = 0.0;
323  A.at<float>(2*i+1,6) = -u2*u1;
324  A.at<float>(2*i+1,7) = -u2*v1;
325  A.at<float>(2*i+1,8) = -u2;
326 
327  }
328 
329  cv::Mat u,w,vt;
330 
331  cv::SVDecomp(A,w,u,vt,cv::SVD::MODIFY_A | cv::SVD::FULL_UV);
332 
333  return vt.row(8).reshape(0, 3); // v的最后一列 reshape 成了3*3的矩阵 --->H得到

347 cv::Mat Initializer::ComputeF21(const vector<cv::Point2f> &vP1,const vector<cv::Point2f> &vP2)
348 {
349  const int N = vP1.size();
350 
351  cv::Mat A(N,9,CV_32F); // N*9
352 
353  for(int i=0; i<N; i++)
354  {
355  const float u1 = vP1[i].x;
356  const float v1 = vP1[i].y;
357  const float u2 = vP2[i].x;
358  const float v2 = vP2[i].y;
359 
360  A.at<float>(i,0) = u2*u1;
361  A.at<float>(i,1) = u2*v1;
362  A.at<float>(i,2) = u2;
363  A.at<float>(i,3) = v2*u1;
364  A.at<float>(i,4) = v2*v1;
365  A.at<float>(i,5) = v2;
366  A.at<float>(i,6) = u1;
367  A.at<float>(i,7) = v1;
368  A.at<float>(i,8) = 1;
369  }
370 
371  cv::Mat u,w,vt;
372 
373  cv::SVDecomp(A,w,u,vt,cv::SVD::MODIFY_A | cv::SVD::FULL_UV);
374 
375  cv::Mat Fpre = vt.row(8).reshape(0, 3); // v的最后一行,1*9reshape成了3*3的矩阵
376 
377  cv::SVDecomp(Fpre,w,u,vt,cv::SVD::MODIFY_A | cv::SVD::FULL_UV);
378 
379  w.at<float>(2)=0; // 秩2约束，将第3个奇异值设为0
380 
381  return u*cv::Mat::diag(w)*vt;
382 }

392 float Initializer::CheckHomography(const cv::Mat &H21, const cv::Mat &H12, vector<bool> &vbMatchesInliers, float sigma)
393 {
394  const int N = mvMatches12.size();
395 
396  // |h11 h12 h13|
397  // |h21 h22 h23|
398  // |h31 h32 h33|
399  const float h11 = H21.at<float>(0,0);
400  const float h12 = H21.at<float>(0,1);
401  const float h13 = H21.at<float>(0,2);
402  const float h21 = H21.at<float>(1,0);
403  const float h22 = H21.at<float>(1,1);
404  const float h23 = H21.at<float>(1,2);
405  const float h31 = H21.at<float>(2,0);
406  const float h32 = H21.at<float>(2,1);
407  const float h33 = H21.at<float>(2,2);
408 
409  // |h11inv h12inv h13inv|
410  // |h21inv h22inv h23inv|
411  // |h31inv h32inv h33inv|
412  const float h11inv = H12.at<float>(0,0);
413  const float h12inv = H12.at<float>(0,1);
414  const float h13inv = H12.at<float>(0,2);
415  const float h21inv = H12.at<float>(1,0);
416  const float h22inv = H12.at<float>(1,1);
417  const float h23inv = H12.at<float>(1,2);
418  const float h31inv = H12.at<float>(2,0);
419  const float h32inv = H12.at<float>(2,1);
420  const float h33inv = H12.at<float>(2,2);
421 
422  vbMatchesInliers.resize(N);
423 
424  float score = 0;
425 
426  // 基于卡方检验计算出的阈值（假设测量有一个像素的偏差）
427  const float th = 5.991;
428 
429  //信息矩阵，方差平方的倒数 msigma 初始化值为1 mpInitialzer中初始化的
430  const float invSigmaSquare = 1.0/(sigma*sigma);
431 
432  // N对特征匹配点
433  for(int i=0; i<N; i++)
434  {
435  bool bIn = true;
436 
437 
438  const cv::KeyPoint &kp1 = mvKeys1[mvMatches12[i].first];
439  const cv::KeyPoint &kp2 = mvKeys2[mvMatches12[i].second];
440 
441  const float u1 = kp1.pt.x;
442  const float v1 = kp1.pt.y;
443  const float u2 = kp2.pt.x;
444  const float v2 = kp2.pt.y;
445 
446  // Reprojection error in first image
447  // x2in1 = H12*x2 ///////// H12 是图2到图1的变换矩阵
448  // 将图像2中的特征点单应到图像1中
449  // |u1| |h11inv h12inv h13inv||u2|
450  // |v1| = |h21inv h22inv h23inv||v2|
451  // |1 | |h31inv h32inv h33inv||1 |
452  const float w2in1inv = 1.0/(h31inv*u2+h32inv*v2+h33inv);
453  const float u2in1 = (h11inv*u2+h12inv*v2+h13inv)*w2in1inv;
454  const float v2in1 = (h21inv*u2+h22inv*v2+h23inv)*w2in1inv;
455 
456  // 计算重投影误差
457  const float squareDist1 = (u1-u2in1)*(u1-u2in1)+(v1-v2in1)*(v1-v2in1);
458 
459  // 根据方差归一化误差
460  const float chiSquare1 = squareDist1*invSigmaSquare;
461 
462  if(chiSquare1>th)
463  bIn = false;
464  else
465  score += th - chiSquare1;
466 
467  // Reprojection error in second image
468  // x1in2 = H21*x1
469  // 将图像1中的特征点单应到图像2中
470  const float w1in2inv = 1.0/(h31*u1+h32*v1+h33);
471  const float u1in2 = (h11*u1+h12*v1+h13)*w1in2inv;
472  const float v1in2 = (h21*u1+h22*v1+h23)*w1in2inv;
473 
474  const float squareDist2 = (u2-u1in2)*(u2-u1in2)+(v2-v1in2)*(v2-v1in2);
475 
476  const float chiSquare2 = squareDist2*invSigmaSquare;
477 
478  if(chiSquare2>th)
479  bIn = false;
480  else
481  score += th - chiSquare2;
482 
483  if(bIn)
484  vbMatchesInliers[i]=true;
485  else
486  vbMatchesInliers[i]=false;
487  }
488 
489  return score;
490 }

519  const float th = 3.841;
520  const float thScore = 5.991;

522  const float invSigmaSquare = 1.0/(sigma*sigma);
523

706 // H矩阵分解常见有两种方法：Faugeras SVD-based decomposition 和 Zhang SVD-based decomposition
707 // 参考文献：Motion and structure from motion in a piecewise plannar environment
708 // 这篇参考文献和下面的代码使用了Faugeras SVD-based decomposition算法
709 
717 bool Initializer::ReconstructH(vector<bool> &vbMatchesInliers, cv::Mat &H21, cv::Mat &K,
718  cv::Mat &R21, cv::Mat &t21, vector<cv::Point3f> &vP3D, vector<bool> &vbTriangulated, float minParallax, int minTriangulated)
719 {
720  int N=0;
721  for(size_t i=0, iend = vbMatchesInliers.size() ; i<iend; i++)
722  if(vbMatchesInliers[i])
723  N++;
724 
725  // We recover 8 motion hypotheses using the method of Faugeras et al.
726  // Motion and structure from motion in a piecewise planar environment.
727  // International Journal of Pattern Recognition and Artificial Intelligence, 1988
728 
729  // 因为特征点是图像坐标系，所以讲H矩阵由相机坐标系换算到图像坐标系
730  cv::Mat invK = K.inv();
731  cv::Mat A = invK*H21*K;
732 
733  cv::Mat U,w,Vt,V;
734  cv::SVD::compute(A,w,U,Vt,cv::SVD::FULL_UV);
735  V=Vt.t();
736 
737  float s = cv::determinant(U)*cv::determinant(Vt);
738 
739  float d1 = w.at<float>(0);
740  float d2 = w.at<float>(1);
741  float d3 = w.at<float>(2);
742 
743  // SVD分解的正常情况是特征值降序排列
744  if(d1/d2<1.00001 || d2/d3<1.00001)
745  {
746  return false;
747  }
748 
749  vector<cv::Mat> vR, vt, vn;
750  vR.reserve(8);
751  vt.reserve(8);
752  vn.reserve(8);
753 
754  //n'=[x1 0 x3] 4 posibilities e1=e3=1, e1=1 e3=-1, e1=-1 e3=1, e1=e3=-1
755  // 法向量n'= [x1 0 x3] 对应ppt的公式17
756  float aux1 = sqrt((d1*d1-d2*d2)/(d1*d1-d3*d3));
757  float aux3 = sqrt((d2*d2-d3*d3)/(d1*d1-d3*d3));
758  float x1[] = {aux1,aux1,-aux1,-aux1};
759  float x3[] = {aux3,-aux3,aux3,-aux3};
760 
761  //case d'=d2
762  // 计算ppt中公式19
763  float aux_stheta = sqrt((d1*d1-d2*d2)*(d2*d2-d3*d3))/((d1+d3)*d2);
764 
765  float ctheta = (d2*d2+d1*d3)/((d1+d3)*d2);
766  float stheta[] = {aux_stheta, -aux_stheta, -aux_stheta, aux_stheta};
767 
768  // 计算旋转矩阵 R‘，计算ppt中公式18
769  // | ctheta 0 -aux_stheta| | aux1|
770  // Rp = | 0 1 0 | tp = | 0 |
771  // | aux_stheta 0 ctheta | |-aux3|
772 
773  // | ctheta 0 aux_stheta| | aux1|
774  // Rp = | 0 1 0 | tp = | 0 |
775  // |-aux_stheta 0 ctheta | | aux3|
776 
777  // | ctheta 0 aux_stheta| |-aux1|
778  // Rp = | 0 1 0 | tp = | 0 |
779  // |-aux_stheta 0 ctheta | |-aux3|
780 
781  // | ctheta 0 -aux_stheta| |-aux1|
782  // Rp = | 0 1 0 | tp = | 0 |
783  // | aux_stheta 0 ctheta | | aux3|
784  for(int i=0; i<4; i++)
785  {
786  cv::Mat Rp=cv::Mat::eye(3,3,CV_32F);
787  Rp.at<float>(0,0)=ctheta;
788  Rp.at<float>(0,2)=-stheta[i];
789  Rp.at<float>(2,0)=stheta[i];
790  Rp.at<float>(2,2)=ctheta;
791 
792  cv::Mat R = s*U*Rp*Vt;
793  vR.push_back(R);
794 
795  cv::Mat tp(3,1,CV_32F);
796  tp.at<float>(0)=x1[i];
797  tp.at<float>(1)=0;
798  tp.at<float>(2)=-x3[i];
799  tp*=d1-d3;
800 
801  // 这里虽然对t有归一化，并没有决定单目整个SLAM过程的尺度
802  // 因为CreateInitialMapMonocular函数对3D点深度会缩放，然后反过来对 t 有改变
803  cv::Mat t = U*tp;
804  vt.push_back(t/cv::norm(t));
805 
806  cv::Mat np(3,1,CV_32F);
807  np.at<float>(0)=x1[i];
808  np.at<float>(1)=0;
809  np.at<float>(2)=x3[i];
810 
811  cv::Mat n = V*np;
812  if(n.at<float>(2)<0)
813  n=-n;
814  vn.push_back(n);
815  }
816 
817  //case d'=-d2
818  // 计算ppt中公式22
819  float aux_sphi = sqrt((d1*d1-d2*d2)*(d2*d2-d3*d3))/((d1-d3)*d2);
820 
821  float cphi = (d1*d3-d2*d2)/((d1-d3)*d2);
822  float sphi[] = {aux_sphi, -aux_sphi, -aux_sphi, aux_sphi};
823 
824  // 计算旋转矩阵 R‘，计算ppt中公式21
825  for(int i=0; i<4; i++)
826  {
827  cv::Mat Rp=cv::Mat::eye(3,3,CV_32F);
828  Rp.at<float>(0,0)=cphi;
829  Rp.at<float>(0,2)=sphi[i];
830  Rp.at<float>(1,1)=-1;
831  Rp.at<float>(2,0)=sphi[i];
832  Rp.at<float>(2,2)=-cphi;
833 
834  cv::Mat R = s*U*Rp*Vt;
835  vR.push_back(R);
836 
837  cv::Mat tp(3,1,CV_32F);
838  tp.at<float>(0)=x1[i];
839  tp.at<float>(1)=0;
840  tp.at<float>(2)=x3[i];
841  tp*=d1+d3;
842 
843  cv::Mat t = U*tp;
844  vt.push_back(t/cv::norm(t));
845 
846  cv::Mat np(3,1,CV_32F);
847  np.at<float>(0)=x1[i];
848  np.at<float>(1)=0;
849  np.at<float>(2)=x3[i];
850 
851  cv::Mat n = V*np;
852  if(n.at<float>(2)<0)
853  n=-n;
854  vn.push_back(n);
855  }
856 
857 
858  int bestGood = 0;
859  int secondBestGood = 0;
860  int bestSolutionIdx = -1;
861  float bestParallax = -1;
862  vector<cv::Point3f> bestP3D;
863  vector<bool> bestTriangulated;
864 
865  // Instead of applying the visibility constraints proposed in the Faugeras' paper (which could fail for points seen with low parallax)
866  // We reconstruct all hypotheses and check in terms of triangulated points and parallax
867  // d'=d2和d'=-d2分别对应8组(R t)
868  for(size_t i=0; i<8; i++)
869  {
870  float parallaxi;
871  vector<cv::Point3f> vP3Di;
872  vector<bool> vbTriangulatedi;
873  int nGood = CheckRT(vR[i],vt[i],mvKeys1,mvKeys2,mvMatches12,vbMatchesInliers,K,vP3Di, 4.0*mSigma2, vbTriangulatedi, parallaxi);
874 
875  // 保留最优的和次优的
876  if(nGood>bestGood)
877  {
878  secondBestGood = bestGood;
879  bestGood = nGood;
880  bestSolutionIdx = i;
881  bestParallax = parallaxi;
882  bestP3D = vP3Di;
883  bestTriangulated = vbTriangulatedi;
884  }
885  else if(nGood>secondBestGood)
886  {
887  secondBestGood = nGood;
888  }
889  }
890 
891 
892  if(secondBestGood<0.75*bestGood && bestParallax>=minParallax && bestGood>minTriangulated && bestGood>0.9*N)
893  {
894  vR[bestSolutionIdx].copyTo(R21);
895  vt[bestSolutionIdx].copyTo(t21);
896  vP3D = bestP3D;
897  vbTriangulated = bestTriangulated;
898 
899  return true;
900  }
901 
902  return false;
903 }

906 // Trianularization: 已知匹配特征点对{x x'} 和 各自相机矩阵{P P'}, 估计三维点 X
907 // x' = P'X x = PX
908 // 它们都属于 x = aPX模型
909 // |X|
910 // |x| |p1 p2 p3 p4 ||Y| |x| |--p0--||.|
911 // |y| = a |p5 p6 p7 p8 ||Z| ===>|y| = a|--p1--||X|
912 // |z| |p9 p10 p11 p12||1| |z| |--p2--||.|
913 // 采用DLT的方法：x叉乘PX = 0
914 // |yp2 - p1| |0|
915 // |p0 - xp2| X = |0|
916 // |xp1 - yp0| |0|
917 // 两个点:
918 // |yp2 - p1 | |0|
919 // |p0 - xp2 | X = |0| ===> AX = 0
920 // |y'p2' - p1' | |0|
921 // |p0' - x'p2'| |0|
922 // 变成程序中的形式：
923 // |xp2 - p0 | |0|
924 // |yp2 - p1 | X = |0| ===> AX = 0
925 // |x'p2'- p0'| |0|
926 // |y'p2'- p1'| |0|

938 void Initializer::Triangulate(const cv::KeyPoint &kp1, const cv::KeyPoint &kp2, const cv::Mat &P1, const cv::Mat &P2, cv::Mat &x3D)
939 {
940  // 在DecomposeE函数和ReconstructH函数中对t有归一化
941  // 这里三角化过程中恢复的3D点深度取决于 t 的尺度，
942  // 但是这里恢复的3D点并没有决定单目整个SLAM过程的尺度
943  // 因为CreateInitialMapMonocular函数对3D点深度会缩放，然后反过来对 t 有改变
944 
945  cv::Mat A(4,4,CV_32F);
946 
947  A.row(0) = kp1.pt.x*P1.row(2)-P1.row(0);
948  A.row(1) = kp1.pt.y*P1.row(2)-P1.row(1);
949  A.row(2) = kp2.pt.x*P2.row(2)-P2.row(0);
950  A.row(3) = kp2.pt.y*P2.row(2)-P2.row(1);
951 
952  cv::Mat u,w,vt;
953  cv::SVD::compute(A,w,u,vt,cv::SVD::MODIFY_A| cv::SVD::FULL_UV);
954  x3D = vt.row(3).t();
955  x3D = x3D.rowRange(0,3)/x3D.at<float>(3);
956 }

968 void Initializer::Normalize(const vector<cv::KeyPoint> &vKeys, vector<cv::Point2f> &vNormalizedPoints, cv::Mat &T)
969 {
970  float meanX = 0;
971  float meanY = 0;
972  const int N = vKeys.size();
973 
974  vNormalizedPoints.resize(N);
975 
976  for(int i=0; i<N; i++)
977  {
978  meanX += vKeys[i].pt.x;
979  meanY += vKeys[i].pt.y;
980  }
981 
982  meanX = meanX/N;
983  meanY = meanY/N;
984 
985  float meanDevX = 0;
986  float meanDevY = 0;
987 
988  // 将所有vKeys点减去中心坐标，使x坐标和y坐标均值分别为0
989  for(int i=0; i<N; i++)
990  {
991  vNormalizedPoints[i].x = vKeys[i].pt.x - meanX;
992  vNormalizedPoints[i].y = vKeys[i].pt.y - meanY;
993 
994  meanDevX += fabs(vNormalizedPoints[i].x);
995  meanDevY += fabs(vNormalizedPoints[i].y);
996  }
997 
998  meanDevX = meanDevX/N;
999  meanDevY = meanDevY/N;
1000 
1001  float sX = 1.0/meanDevX;
1002  float sY = 1.0/meanDevY;
1003 
1004  // 将x坐标和y坐标分别进行尺度缩放，使得x坐标和y坐标的一阶绝对矩分别为1->就是说到原点的距离为根号2。
1005  for(int i=0; i<N; i++)
1006  {
1007  vNormalizedPoints[i].x = vNormalizedPoints[i].x * sX;
1008  vNormalizedPoints[i].y = vNormalizedPoints[i].y * sY;
1009  }
1010 
1011  // |sX 0 -meanx*sX|
1012  // |0 sY -meany*sY|
1013  // |0 0 1 |
1014  T = cv::Mat::eye(3,3,CV_32F);
1015  T.at<float>(0,0) = sX;
1016  T.at<float>(1,1) = sY;
1017  T.at<float>(0,2) = -meanX*sX;
1018  T.at<float>(1,2) = -meanY*sY;
1019 }

1024 int Initializer::CheckRT(const cv::Mat &R, const cv::Mat &t, const vector<cv::KeyPoint> &vKeys1, const vector<cv::KeyPoint> &vKeys2,
1025  const vector<Match> &vMatches12, vector<bool> &vbMatchesInliers,
1026  const cv::Mat &K, vector<cv::Point3f> &vP3D, float th2, vector<bool> &vbGood, float &parallax)
1027 {
1028  // Calibration parameters
1029  const float fx = K.at<float>(0,0);
1030  const float fy = K.at<float>(1,1);
1031  const float cx = K.at<float>(0,2);
1032  const float cy = K.at<float>(1,2);
1033 
1034  vbGood = vector<bool>(vKeys1.size(),false);
1035  vP3D.resize(vKeys1.size());
1036 
1037  vector<float> vCosParallax;
1038  vCosParallax.reserve(vKeys1.size());
1039 
1040  // Camera 1 Projection Matrix K[I|0]
1041  // 步骤1：得到一个相机的投影矩阵
1042  // 以第一个相机的光心作为世界坐标系
1043  cv::Mat P1(3,4,CV_32F,cv::Scalar(0));
1044  K.copyTo(P1.rowRange(0,3).colRange(0,3));
1045  // 第一个相机的光心在世界坐标系下的坐标
1046  cv::Mat O1 = cv::Mat::zeros(3,1,CV_32F);
1047 
1048  // Camera 2 Projection Matrix K[R|t]
1049  // 步骤2：得到第二个相机的投影矩阵
1050  cv::Mat P2(3,4,CV_32F);
1051  R.copyTo(P2.rowRange(0,3).colRange(0,3));
1052  t.copyTo(P2.rowRange(0,3).col(3));
1053  P2 = K*P2;
1054  // 第二个相机的光心在世界坐标系下的坐标
1055  cv::Mat O2 = -R.t()*t;
1056 
1057  int nGood=0;
1058 
1059  for(size_t i=0, iend=vMatches12.size();i<iend;i++)
1060  {
1061  if(!vbMatchesInliers[i])
1062  continue;
1063 
1064  // kp1和kp2是匹配特征点
1065  const cv::KeyPoint &kp1 = vKeys1[vMatches12[i].first];
1066  const cv::KeyPoint &kp2 = vKeys2[vMatches12[i].second];
1067  cv::Mat p3dC1;
1068 
1069  // 步骤3：利用三角法恢复三维点p3dC1
1070  Triangulate(kp1,kp2,P1,P2,p3dC1);
1071 
1072  if(!isfinite(p3dC1.at<float>(0)) || !isfinite(p3dC1.at<float>(1)) || !isfinite(p3dC1.at<float>(2)))
1073  {
1074  vbGood[vMatches12[i].first]=false;
1075  continue;
1076  }
1077 
1078  // Check parallax
1079  // 步骤4：计算视差角余弦值
1080  cv::Mat normal1 = p3dC1 - O1;
1081  float dist1 = cv::norm(normal1);
1082 
1083  cv::Mat normal2 = p3dC1 - O2;
1084  float dist2 = cv::norm(normal2);
1085 
1086  float cosParallax = normal1.dot(normal2)/(dist1*dist2);
1087 
1088  // 步骤5：判断3D点是否在两个摄像头前方
1089 
1090  // Check depth in front of first camera (only if enough parallax, as "infinite" points can easily go to negative depth)
1091  // 步骤5.1：3D点深度为负，在第一个摄像头后方，淘汰
1092  if(p3dC1.at<float>(2)<=0 && cosParallax<0.99998)
1093  continue;
1094 
1095  // Check depth in front of second camera (only if enough parallax, as "infinite" points can easily go to negative depth)
1096  // 步骤5.2：3D点深度为负，在第二个摄像头后方，淘汰
1097  cv::Mat p3dC2 = R*p3dC1+t;
1098 
1099  if(p3dC2.at<float>(2)<=0 && cosParallax<0.99998)
1100  continue;
1101 
1102  // 步骤6：计算重投影误差
1103 
1104  // Check reprojection error in first image
1105  // 计算3D点在第一个图像上的投影误差
1106  float im1x, im1y;
1107  float invZ1 = 1.0/p3dC1.at<float>(2);
1108  im1x = fx*p3dC1.at<float>(0)*invZ1+cx;
1109  im1y = fy*p3dC1.at<float>(1)*invZ1+cy;
1110 
1111  float squareError1 = (im1x-kp1.pt.x)*(im1x-kp1.pt.x)+(im1y-kp1.pt.y)*(im1y-kp1.pt.y);
1112 
1113  // 步骤6.1：重投影误差太大，跳过淘汰
1114  // 一般视差角比较小时重投影误差比较大
1115  if(squareError1>th2)
1116  continue;
1117 
1118  // Check reprojection error in second image
1119  // 计算3D点在第二个图像上的投影误差
1120  float im2x, im2y;
1121  float invZ2 = 1.0/p3dC2.at<float>(2);
1122  im2x = fx*p3dC2.at<float>(0)*invZ2+cx;
1123  im2y = fy*p3dC2.at<float>(1)*invZ2+cy;
1124 
1125  float squareError2 = (im2x-kp2.pt.x)*(im2x-kp2.pt.x)+(im2y-kp2.pt.y)*(im2y-kp2.pt.y);
1126 
1127  // 步骤6.2：重投影误差太大，跳过淘汰
1128  // 一般视差角比较小时重投影误差比较大
1129  if(squareError2>th2)
1130  continue;
1131
