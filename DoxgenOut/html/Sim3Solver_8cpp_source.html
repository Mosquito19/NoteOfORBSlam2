120 void Sim3Solver::SetRansacParameters(double probability, int minInliers, int maxIterations)
121 {
122  mRansacProb = probability;
123  mRansacMinInliers = minInliers;
124  mRansacMaxIts = maxIterations;
125 
126  N = mvpMapPoints1.size(); // number of correspondences
127 
128  mvbInliersi.resize(N);
129 
130  // Adjust Parameters according to number of correspondences
131  float epsilon = (float)mRansacMinInliers/N;
132 
133  // Set RANSAC iterations according to probability, epsilon, and max iterations
134  int nIterations;
135 
136  if(mRansacMinInliers==N)
137  nIterations=1;
138  else
139  nIterations = ceil(log(1-mRansacProb)/log(1-pow(epsilon,3)));
140 
141  mRansacMaxIts = max(1,min(nIterations,mRansacMaxIts));
142 
143  mnIterations = 0;

147 cv::Mat Sim3Solver::iterate(int nIterations, bool &bNoMore, vector<bool> &vbInliers, int &nInliers)

151  nInliers=0;
152 
153  if(N<mRansacMinInliers)
154  {
155  bNoMore = true;
156  return cv::Mat();
157  }
158 
159  vector<size_t> vAvailableIndices;
160 
161  cv::Mat P3Dc1i(3,3,CV_32F);
162  cv::Mat P3Dc2i(3,3,CV_32F);
163 
164  int nCurrentIterations = 0;
165  while(mnIterations<mRansacMaxIts && nCurrentIterations<nIterations)
166  {
167  nCurrentIterations++;// 这个函数中迭代的次数
168  mnIterations++;// 总的迭代次数，默认为最大为300
169 
170  vAvailableIndices = mvAllIndices;
171 
172  // Get min set of points
173  // 步骤1：任意取三组点算Sim矩阵
174  for(short i = 0; i < 3; ++i)
175  {
176  int randi = DUtils::Random::RandomInt(0, vAvailableIndices.size()-1);
177 
178  int idx = vAvailableIndices[randi];
179 
180  // P3Dc1i和P3Dc2i中点的排列顺序：
181  // x1 x2 x3 ...
182  // y1 y2 y3 ...
183  // z1 z2 z3 ...
184  mvX3Dc1[idx].copyTo(P3Dc1i.col(i));
185  mvX3Dc2[idx].copyTo(P3Dc2i.col(i));
186 
187  vAvailableIndices[randi] = vAvailableIndices.back();
188  vAvailableIndices.pop_back();
189  }
190 
191  // 步骤2：根据两组匹配的3D点，计算之间的Sim3变换
192  ComputeSim3(P3Dc1i,P3Dc2i);
193 
194  // 步骤3：通过投影误差进行inlier检测
195  CheckInliers();
196 
197  if(mnInliersi>=mnBestInliers)
198  {
199  mvbBestInliers = mvbInliersi;
200  mnBestInliers = mnInliersi;
201  mBestT12 = mT12i.clone();
202  mBestRotation = mR12i.clone();
203  mBestTranslation = mt12i.clone();
204  mBestScale = ms12i;
205 
206  if(mnInliersi>mRansacMinInliers)// 只要计算得到一次合格的Sim变换，就直接返回
207  {
208  nInliers = mnInliersi;
209  for(int i=0; i<N; i++)

211  vbInliers[mvnIndices1[i]] = true;
212  return mBestT12;
213  }

242 void Sim3Solver::ComputeSim3(cv::Mat &P1, cv::Mat &P2)
243 {
244  // ！！！！！！！这段代码一定要看这篇论文！！！！！！！！！！！
245  // Custom implementation of:
246  // Horn 1987, Closed-form solution of absolute orientataion using unit quaternions
247 
248  // Step 1: Centroid and relative coordinates
249 
250  cv::Mat Pr1(P1.size(),P1.type()); // Relative coordinates to centroid (set 1)
251  cv::Mat Pr2(P2.size(),P2.type()); // Relative coordinates to centroid (set 2)
252  cv::Mat O1(3,1,Pr1.type()); // Centroid of P1
253  cv::Mat O2(3,1,Pr2.type()); // Centroid of P2
254 
255  // O1和O2分别为P1和P2矩阵中3D点的质心
256  // Pr1和Pr2为减去质心后的3D点
257  ComputeCentroid(P1,Pr1,O1);
258  ComputeCentroid(P2,Pr2,O2);
259 
260  // Step 2: Compute M matrix
261 
262  cv::Mat M = Pr2*Pr1.t();
263 
264  // Step 3: Compute N matrix
265 
266  double N11, N12, N13, N14, N22, N23, N24, N33, N34, N44;
267 
268  cv::Mat N(4,4,P1.type());
269 
270  N11 = M.at<float>(0,0)+M.at<float>(1,1)+M.at<float>(2,2);
271  N12 = M.at<float>(1,2)-M.at<float>(2,1);
272  N13 = M.at<float>(2,0)-M.at<float>(0,2);
273  N14 = M.at<float>(0,1)-M.at<float>(1,0);
274  N22 = M.at<float>(0,0)-M.at<float>(1,1)-M.at<float>(2,2);
275  N23 = M.at<float>(0,1)+M.at<float>(1,0);
276  N24 = M.at<float>(2,0)+M.at<float>(0,2);
277  N33 = -M.at<float>(0,0)+M.at<float>(1,1)-M.at<float>(2,2);
278  N34 = M.at<float>(1,2)+M.at<float>(2,1);
279  N44 = -M.at<float>(0,0)-M.at<float>(1,1)+M.at<float>(2,2);
280 
281  N = (cv::Mat_<float>(4,4) << N11, N12, N13, N14,
282  N12, N22, N23, N24,
283  N13, N23, N33, N34,
284  N14, N24, N34, N44);
285 
286 
287  // Step 4: Eigenvector of the highest eigenvalue
288 
289  cv::Mat eval, evec;
290 
291  cv::eigen(N,eval,evec); //evec[0] is the quaternion of the desired rotation
292 
293  // N矩阵最大特征值（第一个特征值）对应特征向量就是要求的四元数死（q0 q1 q2 q3）
294  // 将(q1 q2 q3)放入vec行向量，vec就是四元数旋转轴乘以sin(ang/2)
295  cv::Mat vec(1,3,evec.type());
296  (evec.row(0).colRange(1,4)).copyTo(vec); //extract imaginary part of the quaternion (sin*axis)
297 
298  // Rotation angle. sin is the norm of the imaginary part, cos is the real part
299  double ang=atan2(norm(vec),evec.at<float>(0,0));
300 
301  vec = 2*ang*vec/norm(vec); //Angle-axis representation. quaternion angle is the half
302 
303  mR12i.create(3,3,P1.type());
304 
305  cv::Rodrigues(vec,mR12i); // computes the rotation matrix from angle-axis
306 
307  // Step 5: Rotate set 2
308 
309  cv::Mat P3 = mR12i*Pr2;
310 
311  // Step 6: Scale
312 
313  if(!mbFixScale)
314  {
315  // 论文中还有一个求尺度的公式，p632右中的位置，那个公式不用考虑旋转
316  double nom = Pr1.dot(P3);
317  cv::Mat aux_P3(P3.size(),P3.type());
318  aux_P3=P3;
319  cv::pow(P3,2,aux_P3);
320  double den = 0;
321 
322  for(int i=0; i<aux_P3.rows; i++)
323  {
324  for(int j=0; j<aux_P3.cols; j++)
325  {
326  den+=aux_P3.at<float>(i,j);
327  }
328  }
329 
330  ms12i = nom/den;
331  }
332  else
333  ms12i = 1.0f;
334 
335  // Step 7: Translation
336 
337  mt12i.create(1,3,P1.type());
338  mt12i = O1 - ms12i*mR12i*O2;
339 
340  // Step 8: Transformation
341 
342  // Step 8.1 T12
343  mT12i = cv::Mat::eye(4,4,P1.type());
344 
345  cv::Mat sR = ms12i*mR12i;
346 
347  // |sR t|
348  // mT12i = | 0 1|
349  sR.copyTo(mT12i.rowRange(0,3).colRange(0,3));
350  mt12i.copyTo(mT12i.rowRange(0,3).col(3));
351 
352  // Step 8.2 T21
353 
354  mT21i = cv::Mat::eye(4,4,P1.type());
355 
356  cv::Mat sRinv = (1.0/ms12i)*mR12i.t();
357 
358  sRinv.copyTo(mT21i.rowRange(0,3).colRange(0,3));
359  cv::Mat tinv = -sRinv*mt12i;
360  tinv.copyTo(mT21i.rowRange(0,3).col(3));
361 }

364 void Sim3Solver::CheckInliers()
365 {
366  vector<cv::Mat> vP1im2, vP2im1;
367  Project(mvX3Dc2,vP2im1,mT12i,mK1);// 把2系中的3D经过Sim3变换(mT12i)到1系中计算重投影坐标
368  Project(mvX3Dc1,vP1im2,mT21i,mK2);// 把1系中的3D经过Sim3变换(mT21i)到2系中计算重投影坐标
369 
370  mnInliersi=0;
371 
372  for(size_t i=0; i<mvP1im1.size(); i++)
373  {
374  cv::Mat dist1 = mvP1im1[i]-vP2im1[i];
375  cv::Mat dist2 = vP1im2[i]-mvP2im2[i];
376 
377  const float err1 = dist1.dot(dist1);
378  const float err2 = dist2.dot(dist2);
379 
380  if(err1<mvnMaxError1[i] && err2<mvnMaxError2[i])
381  {
382  mvbInliersi[i]=true;
383  mnInliersi++;
384  }
385  else
386  mvbInliersi[i]=false;
387  }

407 {

412  const float &cx = K.at<float>(0,2);

414
