211 void MapPoint::Replace(MapPoint* pMP)
212 {
213  if(pMP->mnId==this->mnId)
214  return;
215 
216  int nvisible, nfound;
217  map<KeyFrame*,size_t> obs;// 这一段和SetBadFlag函数相同
218  {
219  unique_lock<mutex> lock1(mMutexFeatures);
220  unique_lock<mutex> lock2(mMutexPos);
221  obs=mObservations;
222  mObservations.clear();
223  mbBad=true;
224  nvisible = mnVisible;
225  nfound = mnFound;
226  mpReplaced = pMP;
227  }
228 
229  // 所有能观测到该MapPoint的keyframe都要替换
230  for(map<KeyFrame*,size_t>::iterator mit=obs.begin(), mend=obs.end(); mit!=mend; mit++)
231  {
232  // Replace measurement in keyframe
233  KeyFrame* pKF = mit->first;
234 
235  if(!pMP->IsInKeyFrame(pKF))
236  {
237  pKF->ReplaceMapPointMatch(mit->second, pMP);// 让KeyFrame用pMP替换掉原来的MapPoint
238  pMP->AddObservation(pKF,mit->second);// 让MapPoint替换掉对应的KeyFrame
239  }
240  else
241  {
242  // 产生冲突，即pKF中有两个特征点a,b（这两个特征点的描述子是近似相同的），这两个特征点对应两个MapPoint为this,pMP
243  // 然而在fuse的过程中pMP的观测更多，需要替换this，因此保留b与pMP的联系，去掉a与this的联系
244  pKF->EraseMapPointMatch(mit->second);
245  }
246  }
247  pMP->IncreaseFound(nfound);
248  pMP->IncreaseVisible(nvisible);
249  pMP->ComputeDistinctiveDescriptors();
250 
251  mpMap->EraseMapPoint(this);
252 }

302 void MapPoint::ComputeDistinctiveDescriptors()
303 {
304  // Retrieve all observed descriptors
305  vector<cv::Mat> vDescriptors;
306 
307  map<KeyFrame*,size_t> observations;
308 
309  {
310  unique_lock<mutex> lock1(mMutexFeatures);
311  if(mbBad)
312  return;
313  observations=mObservations;
314  }
315 
316  if(observations.empty())
317  return;
318 
319  vDescriptors.reserve(observations.size());
320 
321  // 遍历观测到3d点的所有关键帧，获得orb描述子，并插入到vDescriptors中
322  for(map<KeyFrame*,size_t>::iterator mit=observations.begin(), mend=observations.end(); mit!=mend; mit++)
323  {
324  KeyFrame* pKF = mit->first;
325 
326  if(!pKF->isBad())
327  vDescriptors.push_back(pKF->mDescriptors.row(mit->second));
328  }
329 
330  if(vDescriptors.empty())
331  return;
332 
333  // Compute distances between them
334  // 获得这些描述子两两之间的距离
335  const size_t N = vDescriptors.size();
336 
337  //float Distances[N][N];
338  std::vector<std::vector<float> > Distances;
339  Distances.resize(N, vector<float>(N, 0));
340  for (size_t i = 0; i<N; i++)
341  {
342  Distances[i][i]=0;
343  for(size_t j=i+1;j<N;j++)
344  {
345  int distij = ORBmatcher::DescriptorDistance(vDescriptors[i],vDescriptors[j]);
346  Distances[i][j]=distij;
347  Distances[j][i]=distij;
348  }
349  }
350 
351  // Take the descriptor with least median distance to the rest
352  int BestMedian = INT_MAX;
353  int BestIdx = 0;
354  for(size_t i=0;i<N;i++)
355  {
356  // 第i个描述子到其它所有所有描述子之间的距离
357  //vector<int> vDists(Distances[i],Distances[i]+N);
358  vector<int> vDists(Distances[i].begin(), Distances[i].end());
359  sort(vDists.begin(), vDists.end());
360 
361  // 获得中值
362  int median = vDists[0.5*(N-1)];
363 
364  // 寻找最小的中值
365  if(median<BestMedian)
366  {
367  BestMedian = median;
368  BestIdx = i;
369  }
370  }
371 
372  {
373  unique_lock<mutex> lock(mMutexFeatures);
374 
375  // 最好的描述子，该描述子相对于其他描述子有最小的距离中值
376  // 简化来讲，中值代表了这个描述子到其它描述子的平均距离
377  // 最好的描述子就是和其它描述子的平均距离最小
378  mDescriptor = vDescriptors[BestIdx].clone();
379  }
380 }

414 void MapPoint::UpdateNormalAndDepth()
415 {
416  map<KeyFrame*,size_t> observations;
417  KeyFrame* pRefKF;
418  cv::Mat Pos;
419  {
420  unique_lock<mutex> lock1(mMutexFeatures);
421  unique_lock<mutex> lock2(mMutexPos);
422  if(mbBad)
423  return;
424 
425  observations=mObservations; // 获得观测到该3d点的所有关键帧
426  pRefKF=mpRefKF; // 观测到该点的参考关键帧
427  Pos = mWorldPos.clone(); // 3d点在世界坐标系中的位置
428  }
429 
430  if(observations.empty())
431  return;
432 
433  cv::Mat normal = cv::Mat::zeros(3,1,CV_32F);
434  int n=0;
435  for(map<KeyFrame*,size_t>::iterator mit=observations.begin(), mend=observations.end(); mit!=mend; mit++)
436  {
437  KeyFrame* pKF = mit->first;
438  cv::Mat Owi = pKF->GetCameraCenter();
439  cv::Mat normali = mWorldPos - Owi;
440  normal = normal + normali/cv::norm(normali); // 对所有关键帧对该点的观测方向归一化为单位向量进行求和
441  n++;
442  }
443 
444  cv::Mat PC = Pos - pRefKF->GetCameraCenter(); // 参考关键帧相机指向3D点的向量（在世界坐标系下的表示）
445  const float dist = cv::norm(PC); // 该点到参考关键帧相机的距离
446  const int level = pRefKF->mvKeysUn[observations[pRefKF]].octave;
447  const float levelScaleFactor = pRefKF->mvScaleFactors[level];
448  const int nLevels = pRefKF->mnScaleLevels; // 金字塔层数
449 
450  {
451  unique_lock<mutex> lock3(mMutexPos);
452  // 另见PredictScale函数前的注释
453  mfMaxDistance = dist*levelScaleFactor; // 观测到该点的距离下限
454  mfMinDistance = mfMaxDistance/pRefKF->mvScaleFactors[nLevels-1]; // 观测到该点的距离上限
455  mNormalVector = normal/n; // 获得平均的观测方向
456  }
457 }

481 int MapPoint::PredictScale(const float &currentDist, KeyFrame* pKF)
482 {
483  float ratio;
484  {
485  unique_lock<mutex> lock(mMutexPos);
486  // mfMaxDistance = ref_dist*levelScaleFactor为参考帧考虑上尺度后的距离
487  // ratio = mfMaxDistance/currentDist = ref_dist/cur_dist
488  ratio = mfMaxDistance/currentDist;
489  }
490 
491  // 同时取log线性化
492  int nScale = ceil(log(ratio)/pKF->mfLogScaleFactor);
493  if(nScale<0)
494  nScale = 0;
495  else if(nScale>=pKF->mnScaleLevels)
496  nScale = pKF->mnScaleLevels-1;
497 
498  return nScale;
499 }
