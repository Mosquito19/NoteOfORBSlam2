656 template<class TDescriptor, class F>
657 void TemplatedVocabulary<TDescriptor,F>::HKmeansStep(NodeId parent_id,
658  const vector<pDescriptor> &descriptors, int current_level)
659 {
660  if(descriptors.empty()) return;
661 
662  // features associated to each cluster
663  vector<TDescriptor> clusters;
664  vector<vector<unsigned int> > groups; // groups[i] = [j1, j2, ...]
665  // j1, j2, ... indices of descriptors associated to cluster i
666 
667  clusters.reserve(m_k);
668  groups.reserve(m_k);
669 
670  //const int msizes[] = { m_k, descriptors.size() };
671  //cv::SparseMat assoc(2, msizes, CV_8U);
672  //cv::SparseMat last_assoc(2, msizes, CV_8U);
674 
675  if((int)descriptors.size() <= m_k)
676  {
677  // trivial case: one cluster per feature
678  groups.resize(descriptors.size());
679 
680  for(unsigned int i = 0; i < descriptors.size(); i++)
681  {
682  groups[i].push_back(i);
683  clusters.push_back(*descriptors[i]);
684  }
685  }
686  else
687  {
688  // select clusters and groups with kmeans
689 
690  bool first_time = true;
691  bool goon = true;
692 
693  // to check if clusters move after iterations
694  vector<int> last_association, current_association;
695 
696  while(goon)
697  {
698  // 1. Calculate clusters
699 
700  if(first_time)
701  {
702  // random sample
703  initiateClusters(descriptors, clusters);
704  }
705  else
706  {
707  // calculate cluster centres 计算聚类中心
708 
709  for(unsigned int c = 0; c < clusters.size(); ++c)
710  {
711  vector<pDescriptor> cluster_descriptors;
712  cluster_descriptors.reserve(groups[c].size());
713 
714  /*
715  for(unsigned int d = 0; d < descriptors.size(); ++d)
716  {
717  if( assoc.find<unsigned char>(c, d) )
718  {
719  cluster_descriptors.push_back(descriptors[d]);
720  }
721  }
722  */
723 
724  vector<unsigned int>::const_iterator vit;
725  for(vit = groups[c].begin(); vit != groups[c].end(); ++vit)
726  {
727  cluster_descriptors.push_back(descriptors[*vit]);
728  }
729 
730 
731  F::meanValue(cluster_descriptors, clusters[c]);
732  }
733 
734  } // if(!first_time)
735 
736  // 2. Associate features with clusters
737 
738  // calculate distances to cluster centers
739  groups.clear();
740  groups.resize(clusters.size(), vector<unsigned int>());
741  current_association.resize(descriptors.size());
742 
743  //assoc.clear();
744 
745  typename vector<pDescriptor>::const_iterator fit;
746  //unsigned int d = 0;
747  for(fit = descriptors.begin(); fit != descriptors.end(); ++fit)//, ++d)
748  {
749  double best_dist = F::distance(*(*fit), clusters[0]);
750  unsigned int icluster = 0;
751 
752  for(unsigned int c = 1; c < clusters.size(); ++c)
753  {
754  double dist = F::distance(*(*fit), clusters[c]);
755  if(dist < best_dist)
756  {
757  best_dist = dist;
758  icluster = c;
759  }
760  }
761 
762  //assoc.ref<unsigned char>(icluster, d) = 1;
763 
764  groups[icluster].push_back(fit - descriptors.begin());
765  current_association[ fit - descriptors.begin() ] = icluster;
766  }
767 
768  // kmeans++ ensures all the clusters has any feature associated with them
769 
770  // 3. check convergence
771  if(first_time)
772  {
773  first_time = false;
774  }
775  else
776  {
777  //goon = !eqUChar(last_assoc, assoc);
778 
779  goon = false;
780  for(unsigned int i = 0; i < current_association.size(); i++)
781  {
782  // 比较上一次和这一次聚类的结果，如果不同则再迭代
783  if(current_association[i] != last_association[i]){
784  goon = true;
785  break;
786  }
787  }
788  }
789 
790  if(goon)
791  {
792  // copy last feature-cluster association
793  last_association = current_association;
794  //last_assoc = assoc.clone();
795  }
796 
797  } // while(goon)
798 
799  } // if must run kmeans
800 
801  // create nodes
802  for(unsigned int i = 0; i < clusters.size(); ++i)
803  {
804  NodeId id = m_nodes.size();
805  m_nodes.push_back(Node(id));
806  m_nodes.back().descriptor = clusters[i];
807  m_nodes.back().parent = parent_id;
808  m_nodes[parent_id].children.push_back(id);
809  }
810 
811  // go on with the next level
812  if(current_level < m_L)
813  {
814  // iterate again with the resulting clusters
815  const vector<NodeId> &children_ids = m_nodes[parent_id].children;
816  for(unsigned int i = 0; i < clusters.size(); ++i)
817  {
818  NodeId id = children_ids[i];
819 
820  vector<pDescriptor> child_features;
821  child_features.reserve(groups[i].size());
822 
823  vector<unsigned int>::const_iterator vit;
824  for(vit = groups[i].begin(); vit != groups[i].end(); ++vit)
825  {
826  child_features.push_back(descriptors[*vit]);
827  }
828 
829  if(child_features.size() > 1)
830  {
831  HKmeansStep(id, child_features, current_level + 1);
832  }
833  }
834  }
835 }

848 template<class TDescriptor, class F>
849 void TemplatedVocabulary<TDescriptor,F>::initiateClustersKMpp(
850  const vector<pDescriptor> &pfeatures, vector<TDescriptor> &clusters) const
851 {
852  // Implements kmeans++ seeding algorithm
853  // Algorithm:
854  // 1. Choose one center uniformly at random from among the data points.
855  // 2. For each data point x, compute D(x), the distance between x and the nearest
856  // center that has already been chosen.
857  // 3. Add one new data point as a center. Each point x is chosen with probability
858  // proportional to D(x)^2.
859  // 4. Repeat Steps 2 and 3 until k centers have been chosen.
860  // 5. Now that the initial centers have been chosen, proceed using standard k-means
861  // clustering.
862 
863  // 1. 从输入的数据点集合中随机选择一个点作为第一个聚类中心
864  // 2. 对于数据集中的每一个点x，计算它与最近聚类中心(指已选择的聚类中心)的距离D(x)并保存在一个数组里，
865  // 然后把这些距离加起来得到Sum(D(x))。
866  // 3. 选择一个新的数据点作为新的聚类中心，选择的原则是：D(x)较大的点，被选取作为聚类中心的概率较大
867  // 实际做法：取一个0～Sum(D(x))之间的随机值Random，计算Sum(D(0)，D(1)...D(j))>=Random，第j个点为种子点
868  // 4. 重复2和3直到k个聚类中心被选出来
869  // 5. 利用这k个初始的聚类中心来运行标准的k-means算法
870 
871  DUtils::Random::SeedRandOnce();
872 
873  clusters.resize(0);
874  clusters.reserve(m_k);
875  vector<double> min_dists(pfeatures.size(), std::numeric_limits<double>::max());
876 
877  // 1.
878 
879  int ifeature = DUtils::Random::RandomInt(0, pfeatures.size()-1);
880 
881  // create first cluster
882  clusters.push_back(*pfeatures[ifeature]);
883 
884  // compute the initial distances
885  typename vector<pDescriptor>::const_iterator fit;
886  vector<double>::iterator dit;
887  dit = min_dists.begin();
888  for(fit = pfeatures.begin(); fit != pfeatures.end(); ++fit, ++dit)
889  {
890  *dit = F::distance(*(*fit), clusters.back());
891  }
892 
893  while((int)clusters.size() < m_k)
894  {
895  // 2.
896  dit = min_dists.begin();
897  for(fit = pfeatures.begin(); fit != pfeatures.end(); ++fit, ++dit)
898  {
899  if(*dit > 0)
900  {
901  double dist = F::distance(*(*fit), clusters.back());
902  if(dist < *dit) *dit = dist; // 仅保存最近的距离，即与最近聚类中心的距离
903  }
904  }
905 
906  // 3.
907  double dist_sum = std::accumulate(min_dists.begin(), min_dists.end(), 0.0);
908 
909  if(dist_sum > 0)
910  {
911  double cut_d;
912  do
913  {
914  cut_d = DUtils::Random::RandomValue<double>(0, dist_sum);
915  } while(cut_d == 0.0);
916 
917  double d_up_now = 0;
918  for(dit = min_dists.begin(); dit != min_dists.end(); ++dit)
919  {
920  d_up_now += *dit;
921  if(d_up_now >= cut_d) break;
922  }
923 
924  if(dit == min_dists.end())
925  ifeature = pfeatures.size()-1;
926  else
927  ifeature = dit - min_dists.begin();
928 
929  clusters.push_back(*pfeatures[ifeature]);
930 
931  } // if dist_sum > 0
932  else
933  break;
934 
935  } // while(used_clusters < m_k)
936 
937 }

966 template<class TDescriptor, class F>
967 void TemplatedVocabulary<TDescriptor,F>::setNodeWeights
968  (const vector<vector<TDescriptor> > &training_features)
969 {
970  const unsigned int NWords = m_words.size();
971  const unsigned int NDocs = training_features.size();
972 
973  if(m_weighting == TF || m_weighting == BINARY)
974  {
975  // idf part must be 1 always
976  for(unsigned int i = 0; i < NWords; i++)
977  m_words[i]->weight = 1;
978  }
979  else if(m_weighting == IDF || m_weighting == TF_IDF)
980  {
981  // IDF and TF-IDF: we calculte the idf path now
982 
983  // Note: this actually calculates the idf part of the tf-idf score.
984  // The complete tf-idf score is calculated in ::transform
985 
986  vector<unsigned int> Ni(NWords, 0); // 统计词频，范围为0～training_features.size()
987  vector<bool> counted(NWords, false);
988 
989  typename vector<vector<TDescriptor> >::const_iterator mit;
990  typename vector<TDescriptor>::const_iterator fit;
991 
992  for(mit = training_features.begin(); mit != training_features.end(); ++mit)
993  {
994  fill(counted.begin(), counted.end(), false);
995 
996  for(fit = mit->begin(); fit < mit->end(); ++fit)
997  {
998  WordId word_id;
999  transform(*fit, word_id);
1000 
1001  if(!counted[word_id])
1002  {
1003  Ni[word_id]++;
1004  counted[word_id] = true;
1005  }
1006  }
1007  }

1242 void TemplatedVocabulary<TDescriptor,F>::transform(const TDescriptor &feature,
1243  WordId &word_id, WordValue &weight, NodeId *nid, int levelsup) const
1244 {

1246  vector<NodeId> nodes;
1247  typename vector<NodeId>::const_iterator nit;
1248 
1249  // level at which the node must be stored in nid, if given
1250  const int nid_level = m_L - levelsup;
1251  if(nid_level <= 0 && nid != NULL) *nid = 0; // root
1252

1254  int current_level = 0;
1255 
1256  do
1257  {
1258  ++current_level;
1259  nodes = m_nodes[final_id].children;
1260  final_id = nodes[0];
1261 
1262  double best_d = F::distance(feature, m_nodes[final_id].descriptor);
1263 
1264  for(nit = nodes.begin() + 1; nit != nodes.end(); ++nit)
1265  {
1266  NodeId id = *nit;

1268  if(d < best_d)
1269  {
1270  best_d = d;
1271  final_id = id;
1272  }
1273  }
1274 
1275  if(nid != NULL && current_level == nid_level)
1276  *nid = final_id;
1277 
1278  } while( !m_nodes[final_id].isLeaf() );
1279 
1280  // turn node id into word id
1281  word_id = m_nodes[final_id].word_id;
1282  weight = m_nodes[final_id].weight;
