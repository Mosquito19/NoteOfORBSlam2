317 void Tracking::Track()
318 {
319  // track包含两部分：估计运动、跟踪局部地图
320 
321  // mState为tracking的状态机
322  // SYSTME_NOT_READY, NO_IMAGE_YET, NOT_INITIALIZED, OK, LOST
323  // 如果图像复位过、或者第一次运行，则为NO_IMAGE_YET状态
324  if(mState==NO_IMAGES_YET)
325  {
326  mState = NOT_INITIALIZED;
327  }
328 
329  // mLastProcessedState存储了Tracking最新的状态，用于FrameDrawer中的绘制
330  mLastProcessedState=mState;
331 
332  // Get Map Mutex -> Map cannot be changed
333  unique_lock<mutex> lock(mpMap->mMutexMapUpdate);
334 
335  // 步骤1：初始化
336  if(mState==NOT_INITIALIZED)
337  {
338  if(mSensor==System::STEREO || mSensor==System::RGBD)
339  StereoInitialization();
340  else
341  MonocularInitialization();
342 
343  mpFrameDrawer->Update(this);
344 
345  if(mState!=OK)
346  return;
347  }
348  else// 步骤2：跟踪
349  {
350  // System is initialized. Track Frame.
351 
352  // bOK为临时变量，用于表示每个函数是否执行成功
353  bool bOK;
354 
355  // Initial camera pose estimation using motion model or relocalization (if tracking is lost)
356  // 在viewer中有个开关menuLocalizationMode，有它控制是否ActivateLocalizationMode，并最终管控mbOnlyTracking
357  // mbOnlyTracking等于false表示正常VO模式（有地图更新），mbOnlyTracking等于true表示用户手动选择定位模式
358  if(!mbOnlyTracking)
359  {
360  // Local Mapping is activated. This is the normal behaviour, unless
361  // you explicitly activate the "only tracking" mode.
362 
363  // 正常初始化成功
364  if(mState==OK)
365  {
366  // Local Mapping might have changed some MapPoints tracked in last frame
367  // 检查并更新上一帧被替换的MapPoints
368  // 更新Fuse函数和SearchAndFuse函数替换的MapPoints
369  CheckReplacedInLastFrame();
370 
371  // 步骤2.1：跟踪上一帧或者参考帧或者重定位
372 
373  // 运动模型是空的或刚完成重定位
374  // mCurrentFrame.mnId<mnLastRelocFrameId+2这个判断不应该有
375  // 应该只要mVelocity不为空，就优先选择TrackWithMotionModel
376  // mnLastRelocFrameId上一次重定位的那一帧
377  if(mVelocity.empty() || mCurrentFrame.mnId<mnLastRelocFrameId+2)
378  {
379  // 将上一帧的位姿作为当前帧的初始位姿
380  // 通过BoW的方式在参考帧中找当前帧特征点的匹配点
381  // 优化每个特征点都对应3D点重投影误差即可得到位姿
382  bOK = TrackReferenceKeyFrame();
383  }
384  else
385  {
386  // 根据恒速模型设定当前帧的初始位姿
387  // 通过投影的方式在参考帧中找当前帧特征点的匹配点
388  // 优化每个特征点所对应3D点的投影误差即可得到位姿
389  bOK = TrackWithMotionModel();
390  if(!bOK)
391  // TrackReferenceKeyFrame是跟踪参考帧，不能根据固定运动速度模型预测当前帧的位姿态，通过bow加速匹配（SearchByBow）
392  // 最后通过优化得到优化后的位姿
393  bOK = TrackReferenceKeyFrame();
394  }
395  }
396  else
397  {
398  // BOW搜索，PnP求解位姿
399  bOK = Relocalization();
400  }
401  }
402  else
403  {
404  // Localization Mode: Local Mapping is deactivated
405  // 只进行跟踪tracking，局部地图不工作
406 
407  // 步骤2.1：跟踪上一帧或者参考帧或者重定位
408 
409  // tracking跟丢了
410  if(mState==LOST)
411  {
412  bOK = Relocalization();
413  }
414  else
415  {
416  // mbVO是mbOnlyTracking为true时的才有的一个变量
417  // mbVO为false表示此帧匹配了很多的MapPoints，跟踪很正常，
418  // mbVO为true表明此帧匹配了很少的MapPoints，少于10个，要跪的节奏
419  if(!mbVO)
420  {
421  // In last frame we tracked enough MapPoints in the map
422  // mbVO为0则表明此帧匹配了很多的3D map点，非常好
423 
424  if(!mVelocity.empty())
425  {
426  bOK = TrackWithMotionModel();
427  // 这个地方是不是应该加上：
428  // if(!bOK)
429  // bOK = TrackReferenceKeyFrame();
430  }
431  else
432  {
433  bOK = TrackReferenceKeyFrame();
434  }
435  }
436  else
437  {
438  // In last frame we tracked mainly "visual odometry" points.
439 
440  // We compute two camera poses, one from motion model and one doing relocalization.
441  // If relocalization is sucessfull we choose that solution, otherwise we retain
442  // the "visual odometry" solution.
443 
444  // mbVO为1，则表明此帧匹配了很少的3D map点，少于10个，要跪的节奏，既做跟踪又做定位

447  bool bOKReloc = false;

449  vector<bool> vbOutMM;

463  if(bOKMM && !bOKReloc)
464  {
465  // 这三行没啥用？
466  mCurrentFrame.SetPose(TcwMM);
467  mCurrentFrame.mvpMapPoints = vpMPsMM;
468  mCurrentFrame.mvbOutlier = vbOutMM;

472  // 这段代码是不是有点多余？应该放到TrackLocalMap函数中统一做

486  }
487

489  }
490  }
491  }

494  mCurrentFrame.mpReferenceKF = mpReferenceKF;
495

498  // 步骤2.2：在帧间匹配得到初始的姿态后，现在对local map进行跟踪得到更多的匹配，并优化当前位姿
499  // local map:当前帧、当前帧的MapPoints、当前关键帧与其它关键帧共视关系
500  // 在步骤2.1中主要是两两跟踪（恒速模型跟踪上一帧、跟踪参考帧），这里搜索局部关键帧后搜集所有局部MapPoints，
501  // 然后将局部MapPoints和当前帧进行投影匹配，得到更多匹配的MapPoints后进行Pose优化
502  if(!mbOnlyTracking)
503  {
504  if(bOK)
505  bOK = TrackLocalMap();
506  }
507  else
508  {
509  // mbVO true means that there are few matches to MapPoints in the map. We cannot retrieve
510  // a local map and therefore we do not perform TrackLocalMap(). Once the system relocalizes
511  // the camera we will use the local map again.
512 
513  // 重定位成功
514  if(bOK && !mbVO)
515  bOK = TrackLocalMap();
516  }

519  mState = OK;
520  else
521  mState=LOST;
522

524  mpFrameDrawer->Update(this);
525 
526  // If tracking were good, check if we insert a keyframe
527  if(bOK)
528  {
529  // Update motion model
530  if(!mLastFrame.mTcw.empty())
531  {
533  // 步骤2.3：更新恒速运动模型TrackWithMotionModel中的mVelocity
534  cv::Mat LastTwc = cv::Mat::eye(4,4,CV_32F);
535  mLastFrame.GetRotationInverse().copyTo(LastTwc.rowRange(0,3).colRange(0,3));

537  mVelocity = mCurrentFrame.mTcw*LastTwc; // Tcl
538  }
539  else
540  mVelocity = cv::Mat();
541

543

547  for(int i=0; i<mCurrentFrame.N; i++)
548  {

1049  return;

1054

1056  // We sort points according to their measured depth by the stereo/RGB-D sensor
1057  // 步骤2.1：得到上一帧有深度值的特征点
1058  vector<pair<float,int> > vDepthIdx;

1068  }

1071  return;

1074  sort(vDepthIdx.begin(),vDepthIdx.end());

1077  // If less than 100 close points, we insert the 100 closest ones.
1078  // 步骤2.3：将距离比较近的点包装成MapPoints
1079  int nPoints = 0;
1080  for(size_t j=0; j<vDepthIdx.size();j++)
1081  {
1082  int i = vDepthIdx[j].second;
1083

1085

1090  {
1091  bCreateNew = true;
1092  }

1095  {
1096  // 这些生成MapPoints后并没有通过：
1097  // a.AddMapPoint、
1098  // b.AddObservation、
1099  // c.ComputeDistinctiveDescriptors、
1100  // d.UpdateNormalAndDepth添加属性，
1101  // 这些MapPoint仅仅为了提高双目和RGBD的跟踪成功率

1106

1108  mlpTemporalPoints.push_back(pNewMP);
1109  nPoints++;
1110  }
1111  else
1112  {
1113  nPoints++;
1114  }
1115 
1116  if(vDepthIdx[j].first>mThDepth && nPoints>100)
1117  break;
1118  }
