79 static float IC_Angle(const Mat& image, Point2f pt, const vector<int> & u_max)
80 {
81  int m_01 = 0, m_10 = 0;
82 
83  const uchar* center = &image.at<uchar> (cvRound(pt.y), cvRound(pt.x));
84 
85  // Treat the center line differently, v=0
86  for (int u = -HALF_PATCH_SIZE; u <= HALF_PATCH_SIZE; ++u)
87  m_10 += u * center[u];
88 
89  // Go line by line in the circuI853lar patch
90  int step = (int)image.step1();//一行的数据元素个数。。
91  for (int v = 1; v <= HALF_PATCH_SIZE; ++v)
92  {
93  // Proceed over the two lines
94  int v_sum = 0;
95  int d = u_max[v];
96  for (int u = -d; u <= d; ++u)
97  {
98  int val_plus = center[u + v*step], val_minus = center[u - v*step];
100  v_sum += (val_plus - val_minus);
101  m_10 += u * (val_plus + val_minus);//u相当于x的偏移，v相当于y的偏移
102  }
103  m_01 += v * v_sum;
104  }
105 
106  return fastAtan2((float)m_01, (float)m_10);
107 }

435 ORBextractor::ORBextractor(int _nfeatures, float _scaleFactor, int _nlevels,
436  int _iniThFAST, int _minThFAST):
437  nfeatures(_nfeatures), scaleFactor(_scaleFactor), nlevels(_nlevels),
438  iniThFAST(_iniThFAST), minThFAST(_minThFAST)
439 {
440  mvScaleFactor.resize(nlevels);
441  mvLevelSigma2.resize(nlevels);
442  mvScaleFactor[0]=1.0f;
443  mvLevelSigma2[0]=1.0f;
444  for(int i=1; i<nlevels; i++)
445  {
446  mvScaleFactor[i]=mvScaleFactor[i-1]*scaleFactor;
447  mvLevelSigma2[i]=mvScaleFactor[i]*mvScaleFactor[i];
448  }
449 
450  mvInvScaleFactor.resize(nlevels);
451  mvInvLevelSigma2.resize(nlevels);
452  for(int i=0; i<nlevels; i++)
453  {
454  mvInvScaleFactor[i]=1.0f/mvScaleFactor[i];
455  mvInvLevelSigma2[i]=1.0f/mvLevelSigma2[i];
456  }

460  mnFeaturesPerLevel.resize(nlevels);
461  float factor = 1.0f / scaleFactor;
462  float nDesiredFeaturesPerScale = nfeatures*(1 - factor)/(1 - (float)pow((double)factor, (double)nlevels));
464  int sumFeatures = 0;
465  for( int level = 0; level < nlevels-1; level++ )
466  {
467  mnFeaturesPerLevel[level] = cvRound(nDesiredFeaturesPerScale);//
468  sumFeatures += mnFeaturesPerLevel[level];
469  nDesiredFeaturesPerScale *= factor;
470  }
471  mnFeaturesPerLevel[nlevels-1] = std::max(nfeatures - sumFeatures, 0);

473  const int npoints = 512;//256对
474  const Point* pattern0 = (const Point*)bit_pattern_31_;
475  std::copy(pattern0, pattern0 + npoints, std::back_inserter(pattern));

477  //This is for orientation
478  // pre-compute the end of a row in a circular patch
480  umax.resize(HALF_PATCH_SIZE + 1);
481 
482  int v, v0, vmax = cvFloor(HALF_PATCH_SIZE * sqrt(2.f) / 2 + 1);
483  int vmin = cvCeil(HALF_PATCH_SIZE * sqrt(2.f) / 2);

485  for (v = 0; v <= vmax; ++v)
486  umax[v] = cvRound(sqrt(hp2 - v * v));

488  // Make sure we are symmetric
489  for (v = HALF_PATCH_SIZE, v0 = 0; v >= vmin; --v)
490  {
491  while (umax[v0] == umax[v0 + 1])
492  ++v0;
493  umax[v] = v0;
494  ++v0;

507 void ExtractorNode::DivideNode(ExtractorNode &n1, ExtractorNode &n2, ExtractorNode &n3, ExtractorNode &n4)
508 {
509  const int halfX = ceil(static_cast<float>(UR.x-UL.x)/2);
510  const int halfY = ceil(static_cast<float>(BR.y-UL.y)/2);
511 
512  //Define boundaries of childs
513  n1.UL = UL;
514  n1.UR = cv::Point2i(UL.x+halfX,UL.y);
515  n1.BL = cv::Point2i(UL.x,UL.y+halfY);
516  n1.BR = cv::Point2i(UL.x+halfX,UL.y+halfY);
517  n1.vKeys.reserve(vKeys.size());
518 
519  n2.UL = n1.UR;
520  n2.UR = UR;
521  n2.BL = n1.BR;
522  n2.BR = cv::Point2i(UR.x,UL.y+halfY);
523  n2.vKeys.reserve(vKeys.size());
524 
525  n3.UL = n1.BL;
526  n3.UR = n1.BR;
527  n3.BL = BL;
528  n3.BR = cv::Point2i(n1.BR.x,BL.y);
529  n3.vKeys.reserve(vKeys.size());
530 
531  n4.UL = n3.UR;
532  n4.UR = n2.BR;
533  n4.BL = n3.BR;
534  n4.BR = BR;
535  n4.vKeys.reserve(vKeys.size());
536 
537  //Associate points to childs
538  for(size_t i=0;i<vKeys.size();i++)
539  {
540  const cv::KeyPoint &kp = vKeys[i];
541  if(kp.pt.x<n1.UR.x)
542  {
543  if(kp.pt.y<n1.BR.y)

546  n3.vKeys.push_back(kp);
547  }

550  else
551  n4.vKeys.push_back(kp);
552  }
553

555  n1.bNoMore = true;
556  if(n2.vKeys.size()==1)
557  n2.bNoMore = true;

559  n3.bNoMore = true;

565 vector<cv::KeyPoint> ORBextractor::DistributeOctTree(const vector<cv::KeyPoint>& vToDistributeKeys, const int &minX,
566  const int &maxX, const int &minY, const int &maxY, const int &N, const int &level)
567 {
568  // Compute how many initial nodes
569  const int nIni = round(static_cast<float>(maxX-minX)/(maxY-minY));

571  const float hX = static_cast<float>(maxX-minX)/nIni;//600

576  vpIniNodes.resize(nIni);
577 
578  for(int i=0; i<nIni; i++)

593  {
594  const cv::KeyPoint &kp = vToDistributeKeys[i];
595  vpIniNodes[kp.pt.x/hX]->vKeys.push_back(kp);
596  }

599 
600  while(lit!=lNodes.end())
601  {
602  if(lit->vKeys.size()==1)
603  {
604  lit->bNoMore=true;
605  lit++;
606  }
607  else if(lit->vKeys.empty())
608  lit = lNodes.erase(lit);

619

635  {
636  if(lit->bNoMore)
637  {
638  // If node only contains one point do not subdivide and continue
639  lit++;

641  }
642  else
643  {
644  // If more than one point, subdivide
645  ExtractorNode n1,n2,n3,n4;
646  lit->DivideNode(n1,n2,n3,n4); // 再细分成四个子区域

648  // Add childs if they contain points
649  if(n1.vKeys.size()>0)

651  lNodes.push_front(n1);
652  if(n1.vKeys.size()>1)
653  {
654  nToExpand++;//注意条件，大于1时，才会加
655  vSizeAndPointerToNode.push_back(make_pair(n1.vKeys.size(),&lNodes.front()));
656  lNodes.front().lit = lNodes.begin();
657  }
658  }
659  if(n2.vKeys.size()>0)
660  {
661  lNodes.push_front(n2);
662  if(n2.vKeys.size()>1)
663  {
664  nToExpand++;
665  vSizeAndPointerToNode.push_back(make_pair(n2.vKeys.size(),&lNodes.front()));
666  lNodes.front().lit = lNodes.begin();
667  }
668  }
669  if(n3.vKeys.size()>0)
670  {
671  lNodes.push_front(n3);
672  if(n3.vKeys.size()>1)
673  {
674  nToExpand++;
675  vSizeAndPointerToNode.push_back(make_pair(n3.vKeys.size(),&lNodes.front()));
676  lNodes.front().lit = lNodes.begin();
677  }
678  }
679  if(n4.vKeys.size()>0)

681  lNodes.push_front(n4);

683  {
684  nToExpand++;
685  vSizeAndPointerToNode.push_back(make_pair(n4.vKeys.size(),&lNodes.front()));
686  lNodes.front().lit = lNodes.begin();
687  }
688  }
689 
690  lit=lNodes.erase(lit);
691  continue;
692  }
693  }
694 
695  // Finish if there are more nodes than required features
696  // or all nodes contain just one point
697  //N：当层应有的特征点数。。
698  if((int)lNodes.size()>=N || (int)lNodes.size()==prevSize)
699  {
700  bFinish = true;
701  }
702  // 当再划分之后所有的Node数大于要求数目时
703  else if(((int)lNodes.size()+nToExpand*3)>N)
704  {
705 
706  while(!bFinish)
707  {
708 
709  prevSize = lNodes.size();
710 
711  vector<pair<int,ExtractorNode*> > vPrevSizeAndPointerToNode = vSizeAndPointerToNode;
712  vSizeAndPointerToNode.clear();
713 
714  // 对需要划分的部分进行排序, 即对兴趣点数较多的区域进行划分
715  sort(vPrevSizeAndPointerToNode.begin(),vPrevSizeAndPointerToNode.end());
716  //sort 升序排列
717  for(int j=vPrevSizeAndPointerToNode.size()-1;j>=0;j--)
718  {
719  ExtractorNode n1,n2,n3,n4;
720  vPrevSizeAndPointerToNode[j].second->DivideNode(n1,n2,n3,n4);
721 
722  // Add childs if they contain points
723  if(n1.vKeys.size()>0)
724  {
725  lNodes.push_front(n1);
726  if(n1.vKeys.size()>1)
727  {
728  vSizeAndPointerToNode.push_back(make_pair(n1.vKeys.size(),&lNodes.front()));
729  lNodes.front().lit = lNodes.begin();
730  }
731  }
732  if(n2.vKeys.size()>0)
733  {
734  lNodes.push_front(n2);
735  if(n2.vKeys.size()>1)
736  {
737  vSizeAndPointerToNode.push_back(make_pair(n2.vKeys.size(),&lNodes.front()));
738  lNodes.front().lit = lNodes.begin();
739  }
740  }
741  if(n3.vKeys.size()>0)
742  {
743  lNodes.push_front(n3);
744  if(n3.vKeys.size()>1)
745  {
746  vSizeAndPointerToNode.push_back(make_pair(n3.vKeys.size(),&lNodes.front()));
747  lNodes.front().lit = lNodes.begin();
748  }
749  }
750  if(n4.vKeys.size()>0)
751  {
752  lNodes.push_front(n4);
753  if(n4.vKeys.size()>1)
754  {
755  vSizeAndPointerToNode.push_back(make_pair(n4.vKeys.size(),&lNodes.front()));
756  lNodes.front().lit = lNodes.begin();
757  }
758  }
759 
760  lNodes.erase(vPrevSizeAndPointerToNode[j].second->lit);
761 
762  if((int)lNodes.size()>=N)
763  break;
764  }
765 
766  if((int)lNodes.size()>=N || (int)lNodes.size()==prevSize)
767  bFinish = true;
768 
769  }
770  }
771  }
772 
773  // Retain the best point in each node
774  // 保留每个区域响应值最大的一个兴趣点
775  vector<cv::KeyPoint> vResultKeys;
776  vResultKeys.reserve(nfeatures);
777  for(list<ExtractorNode>::iterator lit=lNodes.begin(); lit!=lNodes.end(); lit++)
778  {
779  vector<cv::KeyPoint> &vNodeKeys = lit->vKeys;
780  cv::KeyPoint* pKP = &vNodeKeys[0];
781  float maxResponse = pKP->response;
782 
783  for(size_t k=1;k<vNodeKeys.size();k++)
784  {
785  if(vNodeKeys[k].response>maxResponse)
786  {
787  pKP = &vNodeKeys[k];
788  maxResponse = vNodeKeys[k].response;
789  }
790  }
791 
792  vResultKeys.push_back(*pKP);
793  }
794 
795  return vResultKeys;
796 }
