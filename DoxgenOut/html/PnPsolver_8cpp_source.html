57 // PnP求解：已知世界坐标系下的3D点与图像坐标系对应的2D点，求解相机的外参(R t)，即从世界坐标系到相机坐标系的变换。
58 // 而EPnP的思想是：
59 // 将世界坐标系所有的3D点用四个虚拟的控制点来表示，将图像上对应的特征点转化为相机坐标系下的四个控制点
60 // 根据世界坐标系下的四个控制点与相机坐标系下对应的四个控制点（与世界坐标系下四个控制点有相同尺度）即可恢复出(R t)

63 // |x|
64 // |u| |fx r u0||r11 r12 r13 t1||y|
65 // s |v| = |0 fy v0||r21 r22 r23 t2||z|
66 // |1| |0 0 1 ||r32 r32 r33 t3||1|
67 
68 // step1:用四个控制点来表达所有的3D点
69 // p_w = sigma(alphas_j * pctrl_w_j), j从0到4
70 // p_c = sigma(alphas_j * pctrl_c_j), j从0到4
71 // sigma(alphas_j) = 1, j从0到4
72 
73 // step2:根据针孔投影模型
74 // s * u = K * sigma(alphas_j * pctrl_c_j), j从0到4
75 
76 // step3:将step2的式子展开, 消去s
77 // sigma(alphas_j * fx * Xctrl_c_j) + alphas_j * (u0-u)*Zctrl_c_j = 0
78 // sigma(alphas_j * fy * Xctrl_c_j) + alphas_j * (v0-u)*Zctrl_c_j = 0
79 
80 // step4:将step3中的12未知参数（4个控制点*3维参考点坐标）提成列向量
81 // Mx = 0,计算得到初始的解x后可以用Gauss-Newton来提纯得到四个相机坐标系的控制点
82 
83 // step5:根据得到的p_w和对应的p_c，最小化重投影误差即可求解出R t

159 void PnPsolver::SetRansacParameters(double probability, int minInliers, int maxIterations, int minSet, float epsilon, float th2)
160 {
161  mRansacProb = probability;
162  mRansacMinInliers = minInliers;
163  mRansacMaxIts = maxIterations;
164  mRansacEpsilon = epsilon;
165  mRansacMinSet = minSet;
166 
167  N = mvP2D.size(); // number of correspondences, 所有二维特征点个数
168 
169  mvbInliersi.resize(N);// inlier index, mvbInliersi记录每次迭代inlier的点
170 
171  // Adjust Parameters according to number of correspondences
172  int nMinInliers = N*mRansacEpsilon;// RANSAC的残差
173  if(nMinInliers<mRansacMinInliers)
174  nMinInliers=mRansacMinInliers;
175  if(nMinInliers<minSet)
176  nMinInliers=minSet;
177  mRansacMinInliers = nMinInliers;
178 
179  if(mRansacEpsilon<(float)mRansacMinInliers/N)
180  mRansacEpsilon=(float)mRansacMinInliers/N;
181 
182  // Set RANSAC iterations according to probability, epsilon, and max iterations
183  int nIterations;
184 
185  if(mRansacMinInliers==N)//根据期望的残差大小来计算RANSAC需要迭代的次数
186  nIterations=1;
187  else
188  nIterations = ceil(log(1-mRansacProb)/log(1-pow(mRansacEpsilon,3)));
189 
190  mRansacMaxIts = max(1,min(nIterations,mRansacMaxIts));
191 
192  mvMaxError.resize(mvSigma2.size());// 图像提取特征的时候尺度层数
193  for(size_t i=0; i<mvSigma2.size(); i++)// 不同的尺度，设置不同的最大偏差
194  mvMaxError[i] = mvSigma2[i]*th2;
195 }

203 cv::Mat PnPsolver::iterate(int nIterations, bool &bNoMore, vector<bool> &vbInliers, int &nInliers)
204 {
205  bNoMore = false;
206  vbInliers.clear();
207  nInliers=0;
208 
209  // mRansacMinSet为每次RANSAC需要的特征点数，默认为4组3D-2D对应点
210  set_maximum_number_of_correspondences(mRansacMinSet);
211 
212  // N为所有2D点的个数, mRansacMinInliers为RANSAC迭代过程中最少的inlier数
213  if(N<mRansacMinInliers)
214  {
215  bNoMore = true;
216  return cv::Mat();
217  }
218 
219  // mvAllIndices为所有参与PnP的2D点的索引
220  // vAvailableIndices为每次从mvAllIndices中随机挑选mRansacMinSet组3D-2D对应点进行一次RANSAC
221  vector<size_t> vAvailableIndices;
222 
223  int nCurrentIterations = 0;
224  while(mnIterations<mRansacMaxIts || nCurrentIterations<nIterations)
225  {
226  nCurrentIterations++;
227  mnIterations++;
228  reset_correspondences();
229 
230  vAvailableIndices = mvAllIndices;
231 
232  // Get min set of points
233  for(short i = 0; i < mRansacMinSet; ++i)
234  {
235  int randi = DUtils::Random::RandomInt(0, vAvailableIndices.size()-1);
236 
237  int idx = vAvailableIndices[randi];
238 
239  // 将对应的3D-2D压入到pws和us
240  add_correspondence(mvP3Dw[idx].x,mvP3Dw[idx].y,mvP3Dw[idx].z,mvP2D[idx].x,mvP2D[idx].y);
241 
242  vAvailableIndices[randi] = vAvailableIndices.back();
243  vAvailableIndices.pop_back();
244  }
245 
246  // Compute camera pose
247  compute_pose(mRi, mti);
248 
249  // Check inliers
250  CheckInliers();
251 
252  if(mnInliersi>=mRansacMinInliers)
253  {
254  // If it is the best solution so far, save it
255  if(mnInliersi>mnBestInliers)
256  {
257  mvbBestInliers = mvbInliersi;
258  mnBestInliers = mnInliersi;
259 
260  cv::Mat Rcw(3,3,CV_64F,mRi);
261  cv::Mat tcw(3,1,CV_64F,mti);
262  Rcw.convertTo(Rcw,CV_32F);
263  tcw.convertTo(tcw,CV_32F);
264  mBestTcw = cv::Mat::eye(4,4,CV_32F);
265  Rcw.copyTo(mBestTcw.rowRange(0,3).colRange(0,3));
266  tcw.copyTo(mBestTcw.rowRange(0,3).col(3));
267  }
268 
269  if(Refine())
270  {
271  nInliers = mnRefinedInliers;
272  vbInliers = vector<bool>(mvpMapPointMatches.size(),false);
273  for(int i=0; i<N; i++)
274  {
275  if(mvbRefinedInliers[i])
276  vbInliers[mvKeyPointIndices[i]] = true;
277  }
278  return mRefinedTcw.clone();
279  }
280 
281  }
282  }
283 
284  if(mnIterations>=mRansacMaxIts)
285  {
286  bNoMore=true;
287  if(mnBestInliers>=mRansacMinInliers)
288  {
289  nInliers=mnBestInliers;
290  vbInliers = vector<bool>(mvpMapPointMatches.size(),false);
291  for(int i=0; i<N; i++)
292  {
293  if(mvbBestInliers[i])
294  vbInliers[mvKeyPointIndices[i]] = true;
295  }
296  return mBestTcw.clone();
297  }
298  }
299 
300  return cv::Mat();
301 }

351 // 通过之前求解的(R t)检查哪些3D-2D点对属于inliers
352 void PnPsolver::CheckInliers()
353 {
354  mnInliersi=0;
355 
356  for(int i=0; i<N; i++)
357  {
358  cv::Point3f P3Dw = mvP3Dw[i];
359  cv::Point2f P2D = mvP2D[i];
360 
361  // 将3D点由世界坐标系旋转到相机坐标系
362  float Xc = mRi[0][0]*P3Dw.x+mRi[0][1]*P3Dw.y+mRi[0][2]*P3Dw.z+mti[0];
363  float Yc = mRi[1][0]*P3Dw.x+mRi[1][1]*P3Dw.y+mRi[1][2]*P3Dw.z+mti[1];
364  float invZc = 1/(mRi[2][0]*P3Dw.x+mRi[2][1]*P3Dw.y+mRi[2][2]*P3Dw.z+mti[2]);
365 
366  // 将相机坐标系下的3D进行针孔投影
367  double ue = uc + fu * Xc * invZc;
368  double ve = vc + fv * Yc * invZc;
369 
370  // 计算残差大小
371  float distX = P2D.x-ue;
372  float distY = P2D.y-ve;
373 
374  float error2 = distX*distX+distY*distY;
375 
376  if(error2<mvMaxError[i])
377  {
378  mvbInliersi[i]=true;
379  mnInliersi++;
380  }
381  else
382  {
383  mvbInliersi[i]=false;
384  }
385  }
386 }

392 void PnPsolver::set_maximum_number_of_correspondences(int n)
393 {
394  if (maximum_number_of_correspondences < n) {
395  if (pws != 0) delete [] pws;
396  if (us != 0) delete [] us;
397  if (alphas != 0) delete [] alphas;
398  if (pcs != 0) delete [] pcs;

401  pws = new double[3 * maximum_number_of_correspondences];// 每个3D点有(X Y Z)三个值
402  us = new double[2 * maximum_number_of_correspondences];// 每个图像2D点有(u v)两个值
403  alphas = new double[4 * maximum_number_of_correspondences];// 每个3D点由四个控制点拟合，有四个系数
404  pcs = new double[3 * maximum_number_of_correspondences];// 每个3D点有(X Y Z)三个值

425 void PnPsolver::choose_control_points(void)
426 {
427  // Take C0 as the reference points centroid:
428  // 步骤1：第一个控制点：参与PnP计算的参考3D点的几何中心
429  cws[0][0] = cws[0][1] = cws[0][2] = 0;
430  for(int i = 0; i < number_of_correspondences; i++)
431  for(int j = 0; j < 3; j++)
432  cws[0][j] += pws[3 * i + j];
433 
434  for(int j = 0; j < 3; j++)
435  cws[0][j] /= number_of_correspondences;
436 
437 
438  // Take C1, C2, and C3 from PCA on the reference points:
439  // 步骤2：计算其它三个控制点，C1, C2, C3通过PCA分解得到
440  // 将所有的3D参考点写成矩阵，(number_of_correspondences *　３)的矩阵
441  CvMat * PW0 = cvCreateMat(number_of_correspondences, 3, CV_64F);
442 
443  double pw0tpw0[3 * 3], dc[3], uct[3 * 3];
444  CvMat PW0tPW0 = cvMat(3, 3, CV_64F, pw0tpw0);
445  CvMat DC = cvMat(3, 1, CV_64F, dc);
446  CvMat UCt = cvMat(3, 3, CV_64F, uct);
447 
448  // 步骤2.1：将存在pws中的参考3D点减去第一个控制点的坐标（相当于把第一个控制点作为原点）, 并存入PW0
449  for(int i = 0; i < number_of_correspondences; i++)
450  for(int j = 0; j < 3; j++)
451  PW0->data.db[3 * i + j] = pws[3 * i + j] - cws[0][j];
452 
453  // 步骤2.2：利用SVD分解P'P可以获得P的主分量
454  // 类似于齐次线性最小二乘求解的过程，
455  // PW0的转置乘以PW0
456  cvMulTransposed(PW0, &PW0tPW0, 1);
457  cvSVD(&PW0tPW0, &DC, &UCt, 0, CV_SVD_MODIFY_A | CV_SVD_U_T);
458 
459  cvReleaseMat(&PW0);
460 
461  // 步骤2.3：得到C1, C2, C3三个3D控制点，最后加上之前减掉的第一个控制点这个偏移量
462  for(int i = 1; i < 4; i++) {
463  double k = sqrt(dc[i - 1] / number_of_correspondences);
464  for(int j = 0; j < 3; j++)
465  cws[i][j] = cws[0][j] + k * uct[3 * (i - 1) + j];
466  }
467 }
468 
469 // 求解四个控制点的系数alphas
470 // (a2 a3 a4)' = inverse(cws2-cws1 cws3-cws1 cws4-cws1)*(pws-cws1)，a1 = 1-a2-a3-a4
471 // 每一个3D控制点，都有一组alphas与之对应
472 // cws1 cws2 cws3 cws4为四个控制点的坐标
473 // pws为3D参考点的坐标
474 void PnPsolver::compute_barycentric_coordinates(void)
475 {
476  double cc[3 * 3], cc_inv[3 * 3];
477  CvMat CC = cvMat(3, 3, CV_64F, cc);
478  CvMat CC_inv = cvMat(3, 3, CV_64F, cc_inv);
479 
480  // 第一个控制点在质心的位置，后面三个控制点减去第一个控制点的坐标（以第一个控制点为原点）
481  // 步骤1：减去质心后得到x y z轴
482  //
483  // cws的排列 |cws1_x cws1_y cws1_z| ---> |cws1|
484  // |cws2_x cws2_y cws2_z| |cws2|
485  // |cws3_x cws3_y cws3_z| |cws3|
486  // |cws4_x cws4_y cws4_z| |cws4|
487  //
488  // cc的排列 |cc2_x cc3_x cc4_x| --->|cc2 cc3 cc4|
489  // |cc2_y cc3_y cc4_y|
490  // |cc2_z cc3_z cc4_z|
491  for(int i = 0; i < 3; i++)
492  for(int j = 1; j < 4; j++)
493  cc[3 * i + j - 1] = cws[j][i] - cws[0][i];
494 
495  cvInvert(&CC, &CC_inv, CV_SVD);
496  double * ci = cc_inv;
497  for(int i = 0; i < number_of_correspondences; i++) {
498  double * pi = pws + 3 * i;// pi指向第i个3D点的首地址
499  double * a = alphas + 4 * i;// a指向第i个控制点系数alphas的首地址
500 
501  // pi[]-cws[0][]表示将pi和步骤1进行相同的平移
502  for(int j = 0; j < 3; j++)
503  a[1 + j] = ci[3 * j ] * (pi[0] - cws[0][0]) +
504  ci[3 * j + 1] * (pi[1] - cws[0][1]) +
505  ci[3 * j + 2] * (pi[2] - cws[0][2]);
506  a[0] = 1.0f - a[1] - a[2] - a[3];
507  }
508 }
509 
510 // 填充最小二乘的M矩阵
511 // 对每一个3D参考点：
512 // |ai1 0 -ai1*ui, ai2 0 -ai2*ui, ai3 0 -ai3*ui, ai4 0 -ai4*ui|
513 // |0 ai1 -ai1*vi, 0 ai2 -ai2*vi, 0 ai3 -ai3*vi, 0 ai4 -ai4*vi|
514 // 其中i从0到4
515 void PnPsolver::fill_M(CvMat * M,
516  const int row, const double * as, const double u, const double v)
517 {
518  double * M1 = M->data.db + row * 12;
519  double * M2 = M1 + 12;
520 
521  for(int i = 0; i < 4; i++) {
522  M1[3 * i ] = as[i] * fu;
523  M1[3 * i + 1] = 0.0;
524  M1[3 * i + 2] = as[i] * (uc - u);
525 
526  M2[3 * i ] = 0.0;
527  M2[3 * i + 1] = as[i] * fv;
528  M2[3 * i + 2] = as[i] * (vc - v);
529  }

642 double PnPsolver::reprojection_error(const double R[3][3], const double t[3])
643 {
644  double sum2 = 0.0;
645 
646  for(int i = 0; i < number_of_correspondences; i++) {
647  double * pw = pws + 3 * i;
648  double Xc = dot(R[0], pw) + t[0];
649  double Yc = dot(R[1], pw) + t[1];
650  double inv_Zc = 1.0 / (dot(R[2], pw) + t[2]);
651  double ue = uc + fu * Xc * inv_Zc;
652  double ve = vc + fv * Yc * inv_Zc;
653  double u = us[2 * i], v = us[2 * i + 1];
654 
655  sum2 += sqrt( (u - ue) * (u - ue) + (v - ve) * (v - ve) );
656  }
657 
658  return sum2 / number_of_correspondences;
659 }

662 void PnPsolver::estimate_R_and_t(double R[3][3], double t[3])
663 {
664  double pc0[3], pw0[3];
665 
666  pc0[0] = pc0[1] = pc0[2] = 0.0;
667  pw0[0] = pw0[1] = pw0[2] = 0.0;
668 
669  for(int i = 0; i < number_of_correspondences; i++) {
670  const double * pc = pcs + 3 * i;
671  const double * pw = pws + 3 * i;
672 
673  for(int j = 0; j < 3; j++) {
674  pc0[j] += pc[j];
675  pw0[j] += pw[j];
676  }
677  }
678  for(int j = 0; j < 3; j++) {
679  pc0[j] /= number_of_correspondences;
680  pw0[j] /= number_of_correspondences;
681  }
682 
683  double abt[3 * 3], abt_d[3], abt_u[3 * 3], abt_v[3 * 3];
684  CvMat ABt = cvMat(3, 3, CV_64F, abt);
685  CvMat ABt_D = cvMat(3, 1, CV_64F, abt_d);
686  CvMat ABt_U = cvMat(3, 3, CV_64F, abt_u);
687  CvMat ABt_V = cvMat(3, 3, CV_64F, abt_v);
688 
689  cvSetZero(&ABt);
690  for(int i = 0; i < number_of_correspondences; i++) {
691  double * pc = pcs + 3 * i;
692  double * pw = pws + 3 * i;
693 
694  for(int j = 0; j < 3; j++) {
695  abt[3 * j ] += (pc[j] - pc0[j]) * (pw[0] - pw0[0]);
696  abt[3 * j + 1] += (pc[j] - pc0[j]) * (pw[1] - pw0[1]);
697  abt[3 * j + 2] += (pc[j] - pc0[j]) * (pw[2] - pw0[2]);
698  }
699  }
700 
701  cvSVD(&ABt, &ABt_D, &ABt_U, &ABt_V, CV_SVD_MODIFY_A);
702 
703  for(int i = 0; i < 3; i++)
704  for(int j = 0; j < 3; j++)
705  R[i][j] = dot(abt_u + 3 * i, abt_v + 3 * j);
706 
707  const double det =
708  R[0][0] * R[1][1] * R[2][2] + R[0][1] * R[1][2] * R[2][0] + R[0][2] * R[1][0] * R[2][1] -
709  R[0][2] * R[1][1] * R[2][0] - R[0][1] * R[1][0] * R[2][2] - R[0][0] * R[1][2] * R[2][1];
710 
711  if (det < 0) {
712  R[2][0] = -R[2][0];
713  R[2][1] = -R[2][1];
714  R[2][2] = -R[2][2];
715  }
716 
717  t[0] = pc0[0] - dot(R[0], pw0);
718  t[1] = pc0[1] - dot(R[1], pw0);
719  t[2] = pc0[2] - dot(R[2], pw0);
720 }

955 void PnPsolver::qr_solve(CvMat * A, CvMat * b, CvMat * X)
956 {
957  static int max_nr = 0;
958  static double * A1, * A2;
959 
960  const int nr = A->rows;
961  const int nc = A->cols;
962 
963  if (max_nr != 0 && max_nr < nr) {
964  delete [] A1;
965  delete [] A2;
966  }
967  if (max_nr < nr) {
968  max_nr = nr;
969  A1 = new double[nr];
970  A2 = new double[nr];
971  }
972 
973  double * pA = A->data.db, * ppAkk = pA;
974  for(int k = 0; k < nc; k++) {
975  double * ppAik = ppAkk, eta = fabs(*ppAik);
976  for(int i = k + 1; i < nr; i++) {
977  double elt = fabs(*ppAik);
978  if (eta < elt) eta = elt;
979  ppAik += nc;
980  }
981 
982  if (eta == 0) {
983  A1[k] = A2[k] = 0.0;
984  cerr << "God damnit, A is singular, this shouldn't happen." << endl;
985  return;
986  } else {
987  double * ppAik = ppAkk, sum = 0.0, inv_eta = 1. / eta;
988  for(int i = k; i < nr; i++) {
989  *ppAik *= inv_eta;
990  sum += *ppAik * *ppAik;
991  ppAik += nc;
992  }
993  double sigma = sqrt(sum);
994  if (*ppAkk < 0)
995  sigma = -sigma;
996  *ppAkk += sigma;
997  A1[k] = sigma * *ppAkk;
998  A2[k] = -eta * sigma;
999  for(int j = k + 1; j < nc; j++) {
1000  double * ppAik = ppAkk, sum = 0;
1001  for(int i = k; i < nr; i++) {
1002  sum += *ppAik * ppAik[j - k];
1003  ppAik += nc;
1004  }
1005  double tau = sum / A1[k];
1006  ppAik = ppAkk;
