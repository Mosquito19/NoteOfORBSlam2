202  vector<double> vTimestamps;

206  LoadImages(strFile, vstrImageFilenames, vTimestamps);

210  // Create SLAM system. It initializes all system threads and gets ready to process frames.
211  // ORB_SLAM2::System SLAM(argv[1],argv[2],ORB_SLAM2::System::MONOCULAR,true);
212  ORB_SLAM2::System SLAM("/home/zhao/ORB_SLAM2-master/Vocabulary/ORBvoc.txt","/home/zhao/ORB_SLAM2-master/Examples/Monocular/TUM1.yaml",ORB_SLAM2::System::MONOCULAR,true);
213 
214  // Vector for tracking time statistics
215  vector<float> vTimesTrack;
216  vTimesTrack.resize(nImages);
217 
218  cout << endl << "-------" << endl;
219  cout << "Start processing sequence ..." << endl;
220  cout << "Images in the sequence: " << nImages << endl << endl;
221 
222  // Main loop
223  cv::Mat im;
224  for(int ni=0; ni<nImages; ni++)
225  {
226  // Read image from file这个数据集我放在ORB_slam2文件夹里面，我经常删除这个文件夹/。。。。
228  im = cv::imread(string("/home/zhao/ORB_SLAM2-master/Examples/Monocular/rgbd_dataset_freiburg1_desk")+"/"+vstrImageFilenames[ni],CV_LOAD_IMAGE_UNCHANGED);
229  double tframe = vTimestamps[ni];
230 
231  if(im.empty())
232  {
233  cerr << endl << "Failed to load image at: "
234  << string(argv[3]) << "/" << vstrImageFilenames[ni] << endl;
235  return 1;

238 #ifdef COMPILEDWITHC11
239  std::chrono::steady_clock::time_point t1 = std::chrono::steady_clock::now();
240 #else
241  std::chrono::monotonic_clock::time_point t1 = std::chrono::monotonic_clock::now();
242 #endif
243 
244  // Pass the image to the SLAM system
245  SLAM.TrackMonocular(im,tframe);
246 
247 #ifdef COMPILEDWITHC11
248  std::chrono::steady_clock::time_point t2 = std::chrono::steady_clock::now();
249 #else
250  std::chrono::monotonic_clock::time_point t2 = std::chrono::monotonic_clock::now();
251 #endif
252 
253  double ttrack= std::chrono::duration_cast<std::chrono::duration<double> >(t2 - t1).count();
254 
255  vTimesTrack[ni]=ttrack;
256 
257  // Wait to load the next frame
258  double T=0;
259  if(ni<nImages-1)
260  T = vTimestamps[ni+1]-tframe;
261  else if(ni>0)
262  T = tframe-vTimestamps[ni-1];
263 
264  if(ttrack<T)
265  usleep((T-ttrack)*1e6);
266  }
267 
268  // Stop all threads
269  SLAM.Shutdown();
270 
271  // Tracking time statistics
272  sort(vTimesTrack.begin(),vTimesTrack.end());
273  float totaltime = 0;
274  for(int ni=0; ni<nImages; ni++)
275  {
276  totaltime+=vTimesTrack[ni];
277  }
278  cout << "-------" << endl << endl;
279  cout << "median tracking time: " << vTimesTrack[nImages/2] << endl;
280  cout << "mean tracking time: " << totaltime/nImages << endl;
281 
282  // Save camera trajectory
283  SLAM.SaveKeyFrameTrajectoryTUM("KeyFrameTrajectory.txt");
284 
285  return 0;
