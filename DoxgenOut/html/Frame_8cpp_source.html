286 void Frame::UpdatePoseMatrices()
287 {
288  // [x_camera 1] = [R|t]*[x_world 1]，坐标为齐次形式
289  // x_camera = R*x_world + t
290  mRcw = mTcw.rowRange(0,3).colRange(0,3);
291  mRwc = mRcw.t();
292  mtcw = mTcw.rowRange(0,3).col(3);
293  // mtcw, 即相机坐标系下相机坐标系到世界坐标系间的向量, 向量方向由相机坐标系指向世界坐标系
294  // mOw, 即世界坐标系下世界坐标系到相机坐标系间的向量, 向量方向由世界坐标系指向相机坐标系
295  mOw = -mRcw.t()*mtcw;

307 bool Frame::isInFrustum(MapPoint *pMP, float viewingCosLimit)
308 {
309  pMP->mbTrackInView = false;
310 
311  // 3D in absolute coordinates
312  cv::Mat P = pMP->GetWorldPos();
313 
314  // 3D in camera coordinates
315  // 3D点P在相机坐标系下的坐标
316  const cv::Mat Pc = mRcw*P+mtcw; // 这里的Rt是经过初步的优化后的
317  const float &PcX = Pc.at<float>(0);
318  const float &PcY = Pc.at<float>(1);
319  const float &PcZ = Pc.at<float>(2);
320 
321  // Check positive depth
322  if(PcZ<0.0f)
323  return false;
324 
325  // Project in image and check it is not outside
326  // V-D 1) 将MapPoint投影到当前帧, 并判断是否在图像内
327  const float invz = 1.0f/PcZ;
328  const float u=fx*PcX*invz+cx;
329  const float v=fy*PcY*invz+cy;
330 
331  if(u<mnMinX || u>mnMaxX)
332  return false;
333  if(v<mnMinY || v>mnMaxY)
334  return false;
335 
336  // Check distance is in the scale invariance region of the MapPoint
337  // V-D 3) 计算MapPoint到相机中心的距离, 并判断是否在尺度变化的距离内
338  const float maxDistance = pMP->GetMaxDistanceInvariance();
339  const float minDistance = pMP->GetMinDistanceInvariance();
340  // 世界坐标系下，相机到3D点P的向量, 向量方向由相机指向3D点P
341  const cv::Mat PO = P-mOw;
342  const float dist = cv::norm(PO);
343 
344  if(dist<minDistance || dist>maxDistance)
345  return false;
346 
347  // Check viewing angle
348  // V-D 2) 计算当前视角和平均视角夹角的余弦值, 若小于cos(60), 即夹角大于60度则返回
349  cv::Mat Pn = pMP->GetNormal();
350 
351  const float viewCos = PO.dot(Pn)/dist;
352 
353  if(viewCos<viewingCosLimit)
354  return false;
355 
356  // Predict scale in the image
357  // V-D 4) 根据深度预测尺度（对应特征点在一层）
358  const int nPredictedLevel = pMP->PredictScale(dist,this);
359 
360  // Data used by the tracking
361  // 标记该点将来要被投影
362  pMP->mbTrackInView = true;
363  pMP->mTrackProjX = u;
364  pMP->mTrackProjXR = u - mbf*invz; //该3D点投影到双目右侧相机上的横坐标
365  pMP->mTrackProjY = v;
366  pMP->mnTrackScaleLevel = nPredictedLevel;
367  pMP->mTrackViewCos = viewCos;
368 
369  return true;
370 }

381 vector<size_t> Frame::GetFeaturesInArea(const float &x, const float &y, const float &r, const int minLevel, const int maxLevel) const
383 {
384  vector<size_t> vIndices;
385  vIndices.reserve(N);
386 
387  const int nMinCellX = max(0,(int)floor((x-mnMinX-r)*mfGridElementWidthInv));
388  if(nMinCellX>=FRAME_GRID_COLS)
389  return vIndices;
390 
391  const int nMaxCellX = min((int)FRAME_GRID_COLS-1,(int)ceil((x-mnMinX+r)*mfGridElementWidthInv));
392  if(nMaxCellX<0)
393  return vIndices;
394 
395  const int nMinCellY = max(0,(int)floor((y-mnMinY-r)*mfGridElementHeightInv));
396  if(nMinCellY>=FRAME_GRID_ROWS)
397  return vIndices;
398 
399  const int nMaxCellY = min((int)FRAME_GRID_ROWS-1,(int)ceil((y-mnMinY+r)*mfGridElementHeightInv));
400  if(nMaxCellY<0)
401  return vIndices;
402 
403  const bool bCheckLevels = (minLevel>0) || (maxLevel>=0);
404 
405  for(int ix = nMinCellX; ix<=nMaxCellX; ix++)
406  {
407  for(int iy = nMinCellY; iy<=nMaxCellY; iy++)
408  {
409  const vector<size_t> vCell = mGrid[ix][iy];//这个二维数组 网格里的 放的特征的序号 组成的向量
410  if(vCell.empty())
411  continue;
412 
413  for(size_t j=0, jend=vCell.size(); j<jend; j++)//网格里的 这个向量里的特征序号 遍历
414  {
415  const cv::KeyPoint &kpUn = mvKeysUn[vCell[j]];
416  if(bCheckLevels)
417  {
418  if(kpUn.octave<minLevel)
419  continue;
420  if(maxLevel>=0)
421  if(kpUn.octave>maxLevel)//如果大于maxlevel层，也放弃
422  continue;
423  }
424 
425  const float distx = kpUn.pt.x-x;
426  const float disty = kpUn.pt.y-y;
427 
428  if(fabs(distx)<r && fabs(disty)<r)
429  vIndices.push_back(vCell[j]);
430  }
431  }
432  }
433 
434  return vIndices;
435 }

465 void Frame::UndistortKeyPoints()
466 {
467  // 如果没有图像是矫正过的，没有失真
468  if(mDistCoef.at<float>(0)==0.0)
469  {
470  mvKeysUn=mvKeys;
471  return;
472  }
473 
474  // Fill matrix with points
475  // N为提取的特征点数量，将N个特征点保存在N*2的mat中
476  cv::Mat mat(N,2,CV_32F);
477  for(int i=0; i<N; i++)
478  {
479  mat.at<float>(i,0)=mvKeys[i].pt.x;
480  mat.at<float>(i,1)=mvKeys[i].pt.y;
481  }
482 
483  // Undistort points
484  // 调整mat的通道为2，矩阵的行列形状不变
485  mat=mat.reshape(2);
486  cv::undistortPoints(mat,mat,mK,mDistCoef,cv::Mat(),mK); // 用cv的函数进行失真校正
487  mat=mat.reshape(1);
488 
489  // Fill undistorted keypoint vector
490  // 存储校正后的特征点
491  mvKeysUn.resize(N);
492  for(int i=0; i<N; i++)
493  {
494  cv::KeyPoint kp = mvKeys[i];
495  kp.pt.x=mat.at<float>(i,0);
496  kp.pt.y=mat.at<float>(i,1);
497  mvKeysUn[i]=kp;
498  }
499 }

501 void Frame::ComputeImageBounds(const cv::Mat &imLeft)//就是无畸变图象的边界。。畸变处理后，肯定边界会变化的
502 {

506  cv::Mat mat(4,2,CV_32F);
507  mat.at<float>(0,0)=0.0; //左上
508  mat.at<float>(0,1)=0.0;
509  mat.at<float>(1,0)=imLeft.cols; //右上

526  else
527  {

543 void Frame::ComputeStereoMatches()
544 {
545  mvuRight = vector<float>(N,-1.0f);
546  mvDepth = vector<float>(N,-1.0f);
547 
548  const int thOrbDist = (ORBmatcher::TH_HIGH+ORBmatcher::TH_LOW)/2;
549 
550  const int nRows = mpORBextractorLeft->mvImagePyramid[0].rows;
551 
552  //Assign keypoints to row table
553  // 步骤1：建立特征点搜索范围对应表，一个特征点在一个带状区域内搜索匹配特征点
554  // 匹配搜索的时候，不仅仅是在一条横线上搜索，而是在一条横向搜索带上搜索,简而言之，原本每个特征点的纵坐标为1，这里把特征点体积放大，纵坐标占好几行
555  // 例如左目图像某个特征点的纵坐标为20，那么在右侧图像上搜索时是在纵坐标为18到22这条带上搜索，搜索带宽度为正负2，搜索带的宽度和特征点所在金字塔层数有关
556  // 简单来说，如果纵坐标是20，特征点在图像第20行，那么认为18 19 20 21 22行都有这个特征点
557  // vRowIndices[18]、vRowIndices[19]、vRowIndices[20]、vRowIndices[21]、vRowIndices[22]都有这个特征点编号
558  vector<vector<size_t> > vRowIndices(nRows,vector<size_t>());
559 
560  for(int i=0; i<nRows; i++)
561  vRowIndices[i].reserve(200);
562 
563  const int Nr = mvKeysRight.size();
564 
565  for(int iR=0; iR<Nr; iR++)
566  {
567  // !!在这个函数中没有对双目进行校正，双目校正是在外层程序中实现的
568  const cv::KeyPoint &kp = mvKeysRight[iR];
569  const float &kpY = kp.pt.y;
570  // 计算匹配搜索的纵向宽度，尺度越大（层数越高，距离越近），搜索范围越大
571  // 如果特征点在金字塔第一层，则搜索范围为:正负2
572  // 尺度越大其位置不确定性越高，所以其搜索半径越大
573  const float r = 2.0f*mvScaleFactors[mvKeysRight[iR].octave];
574  const int maxr = ceil(kpY+r);
575  const int minr = floor(kpY-r);
576 
577  for(int yi=minr;yi<=maxr;yi++)
578  vRowIndices[yi].push_back(iR);
579  }
580 
581  // Set limits for search
582  const float minZ = mb; // NOTE bug mb没有初始化，mb的赋值在构造函数中放在ComputeStereoMatches函数的后面
583  const float minD = 0; // 最小视差, 设置为0即可
584  const float maxD = mbf/minZ; // 最大视差, 对应最小深度 mbf/minZ = mbf/mb = mbf/(mbf/fx) = fx (wubo???)
585 
586  // For each left keypoint search a match in the right image
587  vector<pair<int, int> > vDistIdx;
588  vDistIdx.reserve(N);
589 
590  // 步骤2：对左目相机每个特征点，通过描述子在右目带状搜索区域找到匹配点, 再通过SAD做亚像素匹配
591  // 注意：这里是校正前的mvKeys，而不是校正后的mvKeysUn
592  // KeyFrame::UnprojectStereo和Frame::UnprojectStereo函数中不一致
593  // 这里是不是应该对校正后特征点求深度呢？(wubo???)
594  for(int iL=0; iL<N; iL++)
595  {
596  const cv::KeyPoint &kpL = mvKeys[iL];
597  const int &levelL = kpL.octave;
598  const float &vL = kpL.pt.y;
599  const float &uL = kpL.pt.x;
600 
601  // 可能的匹配点
602  const vector<size_t> &vCandidates = vRowIndices[vL];
603 
604  if(vCandidates.empty())
605  continue;
606 
607  const float minU = uL-maxD; // 最小匹配范围
608  const float maxU = uL-minD; // 最大匹配范围
609 
610  if(maxU<0)
611  continue;
612 
613  int bestDist = ORBmatcher::TH_HIGH;
614  size_t bestIdxR = 0;
615 
616  // 每个特征点描述子占一行，建立一个指针指向iL特征点对应的描述子
617  const cv::Mat &dL = mDescriptors.row(iL);
618 
619  // Compare descriptor to right keypoints
620  // 步骤2.1：遍历右目所有可能的匹配点，找出最佳匹配点（描述子距离最小）
621  for(size_t iC=0; iC<vCandidates.size(); iC++)
622  {
623  const size_t iR = vCandidates[iC];
624  const cv::KeyPoint &kpR = mvKeysRight[iR];
625 
626  // 仅对近邻尺度的特征点进行匹配
627  if(kpR.octave<levelL-1 || kpR.octave>levelL+1)
628  continue;
629 
630  const float &uR = kpR.pt.x;
631 
632  if(uR>=minU && uR<=maxU)
633  {
634  const cv::Mat &dR = mDescriptorsRight.row(iR);
635  const int dist = ORBmatcher::DescriptorDistance(dL,dR);
636 
637  if(dist<bestDist)
638  {
639  bestDist = dist;
640  bestIdxR = iR;
641  }
642  }
643  }
644  // 最好的匹配的匹配误差存在bestDist，匹配点位置存在bestIdxR中
645 
646  // Subpixel match by correlation
647  // 步骤2.2：通过SAD匹配提高像素匹配修正量bestincR
648  if(bestDist<thOrbDist)
649  {
650  // coordinates in image pyramid at keypoint scale
651  // kpL.pt.x对应金字塔最底层坐标，将最佳匹配的特征点对尺度变换到尺度对应层 (scaleduL, scaledvL) (scaleduR0, )
652  const float uR0 = mvKeysRight[bestIdxR].pt.x;
653  const float scaleFactor = mvInvScaleFactors[kpL.octave];
654  const float scaleduL = round(kpL.pt.x*scaleFactor);
655  const float scaledvL = round(kpL.pt.y*scaleFactor);
656  const float scaleduR0 = round(uR0*scaleFactor);
657 
658  // sliding window search
659  const int w = 5; // 滑动窗口的大小11*11 注意该窗口取自resize后的图像
660  cv::Mat IL = mpORBextractorLeft->mvImagePyramid[kpL.octave].rowRange(scaledvL-w,scaledvL+w+1).colRange(scaleduL-w,scaleduL+w+1);
661  IL.convertTo(IL,CV_32F);
662  IL = IL - IL.at<float>(w,w) * cv::Mat::ones(IL.rows,IL.cols,CV_32F);//窗口中的每个元素减去正中心的那个元素，简单归一化，减小光照强度影响
663 
664  int bestDist = INT_MAX;
665  int bestincR = 0;
666  const int L = 5;
667  vector<float> vDists;
668  vDists.resize(2*L+1); // 11
669 
670  // 滑动窗口的滑动范围为（-L, L）,提前判断滑动窗口滑动过程中是否会越界
671  const float iniu = scaleduR0+L-w; //这个地方是否应该是scaleduR0-L-w (wubo???)
672  const float endu = scaleduR0+L+w+1;
673  if(iniu<0 || endu >= mpORBextractorRight->mvImagePyramid[kpL.octave].cols)
674  continue;
675 
676  for(int incR=-L; incR<=+L; incR++)
677  {
678  // 横向滑动窗口
679  cv::Mat IR = mpORBextractorRight->mvImagePyramid[kpL.octave].rowRange(scaledvL-w,scaledvL+w+1).colRange(scaleduR0+incR-w,scaleduR0+incR+w+1);
680  IR.convertTo(IR,CV_32F);
681  IR = IR - IR.at<float>(w,w) * cv::Mat::ones(IR.rows,IR.cols,CV_32F);//窗口中的每个元素减去正中心的那个元素，简单归一化，减小光照强度影响
682 
683  float dist = cv::norm(IL,IR,cv::NORM_L1); // 一范数，计算差的绝对值
684  if(dist<bestDist)
685  {
686  bestDist = dist;// SAD匹配目前最小匹配偏差
687  bestincR = incR; // SAD匹配目前最佳的修正量
688  }
689 
690  vDists[L+incR] = dist; // 正常情况下，这里面的数据应该以抛物线形式变化
691  }
692 
693  if(bestincR==-L || bestincR==L) // 整个滑动窗口过程中，SAD最小值不是以抛物线形式出现，SAD匹配失败，同时放弃求该特征点的深度
694  continue;
695 
696  // Sub-pixel match (Parabola fitting)
697  // 步骤2.3：做抛物线拟合找谷底得到亚像素匹配deltaR
698  // (bestincR,dist) (bestincR-1,dist) (bestincR+1,dist)三个点拟合出抛物线
699  // bestincR+deltaR就是抛物线谷底的位置，相对SAD匹配出的最小值bestincR的修正量为deltaR
700  const float dist1 = vDists[L+bestincR-1];
701  const float dist2 = vDists[L+bestincR];
702  const float dist3 = vDists[L+bestincR+1];
703 
704  const float deltaR = (dist1-dist3)/(2.0f*(dist1+dist3-2.0f*dist2));
705 
706  // 抛物线拟合得到的修正量不能超过一个像素，否则放弃求该特征点的深度
707  if(deltaR<-1 || deltaR>1)
708  continue;
709 
710  // Re-scaled coordinate
711  // 通过描述子匹配得到匹配点位置为scaleduR0
712  // 通过SAD匹配找到修正量bestincR
713  // 通过抛物线拟合找到亚像素修正量deltaR
714  float bestuR = mvScaleFactors[kpL.octave]*((float)scaleduR0+(float)bestincR+deltaR);
715 
716  // 这里是disparity，根据它算出depth
717  float disparity = (uL-bestuR);
718 
719  if(disparity>=minD && disparity<maxD) // 最后判断视差是否在范围内
720  {
721  if(disparity<=0)
722  {
723  disparity=0.01;
724  bestuR = uL-0.01;
725  }
726  // depth 是在这里计算的
727  // depth=baseline*fx/disparity
728  mvDepth[iL]=mbf/disparity; // 深度
729  mvuRight[iL] = bestuR; // 匹配对在右图的横坐标
730  vDistIdx.push_back(pair<int,int>(bestDist,iL)); // 该特征点SAD匹配最小匹配偏差
731  }
732  }
733  }
734 
735  // 步骤3：剔除SAD匹配偏差较大的匹配特征点
736  // 前面SAD匹配只判断滑动窗口中是否有局部最小值，这里通过对比剔除SAD匹配偏差比较大的特征点的深度
737  sort(vDistIdx.begin(),vDistIdx.end()); // 根据所有匹配对的SAD偏差进行排序, 距离由小到大
738  const float median = vDistIdx[vDistIdx.size()/2].first;
739  const float thDist = 1.5f*1.4f*median; // 计算自适应距离, 大于此距离的匹配对将剔除
740 
741  for(int i=vDistIdx.size()-1;i>=0;i--)
742  {
743  if(vDistIdx[i].first<thDist)
744  break;
745  else
746  {
747  mvuRight[vDistIdx[i].second]=-1;
748  mvDepth[vDistIdx[i].second]=-1;
749  }
750  }
751 }
