283 bool LoopClosing::ComputeSim3()
284 {
285  // For each consistent loop candidate we try to compute a Sim3
286 
287  const int nInitialCandidates = mvpEnoughConsistentCandidates.size();
288 
289  // We compute first ORB matches for each candidate
290  // If enough matches are found, we setup a Sim3Solver
291  ORBmatcher matcher(0.75,true);
292 
293  vector<Sim3Solver*> vpSim3Solvers;
294  vpSim3Solvers.resize(nInitialCandidates);// 每个候选帧都有一个Sim3Solver
295 
296  vector<vector<MapPoint*> > vvpMapPointMatches;
297  vvpMapPointMatches.resize(nInitialCandidates);
298 
299  vector<bool> vbDiscarded;
300  vbDiscarded.resize(nInitialCandidates);
301 
302  int nCandidates=0; //candidates with enough matches
303 
304  for(int i=0; i<nInitialCandidates; i++)
305  {
306  // 步骤1：从筛选的闭环候选帧中取出一帧关键帧pKF
307  KeyFrame* pKF = mvpEnoughConsistentCandidates[i];
308 
309  // avoid that local mapping erase it while it is being processed in this thread
310  // 防止在LocalMapping中KeyFrameCulling函数将此关键帧作为冗余帧剔除
311  pKF->SetNotErase();
312 
313  if(pKF->isBad())
314  {
315  vbDiscarded[i] = true;// 直接将该候选帧舍弃
316  continue;
317  }
318 
319  // 步骤2：将当前帧mpCurrentKF与闭环候选关键帧pKF匹配
320  // 通过bow加速得到mpCurrentKF与pKF之间的匹配特征点，vvpMapPointMatches是匹配特征点对应的MapPoints
321  int nmatches = matcher.SearchByBoW(mpCurrentKF,pKF,vvpMapPointMatches[i]);
322 
323  // 匹配的特征点数太少，该候选帧剔除
324  if(nmatches<20)
325  {
326  vbDiscarded[i] = true;
327  continue;
328  }
329  else
330  {
331  // if bFixScale is true, 6DoF optimization (stereo,rgbd), 7DoF otherwise (mono)
332  // 构造Sim3求解器
333  // 如果mbFixScale为true，则是6DoFf优化（双目 RGBD），如果是false，则是7DoF优化（单目）
334  Sim3Solver* pSolver = new Sim3Solver(mpCurrentKF,pKF,vvpMapPointMatches[i],mbFixScale);
335  pSolver->SetRansacParameters(0.99,20,300);// 至少20个inliers 300次迭代
336  vpSim3Solvers[i] = pSolver;
337  }
338 
339  // 参与Sim3计算的候选关键帧数加1
340  nCandidates++;
341  }
342 
343  bool bMatch = false;// 用于标记是否有一个候选帧通过Sim3的求解与优化
344 
345  // Perform alternatively RANSAC iterations for each candidate
346  // until one is succesful or all fail
347  // 一直循环所有的候选帧，每个候选帧迭代5次，如果5次迭代后得不到结果，就换下一个候选帧
348  // 直到有一个候选帧首次迭代成功bMatch为true，或者某个候选帧总的迭代次数超过限制，直接将它剔除
349  while(nCandidates>0 && !bMatch)
350  {
351  for(int i=0; i<nInitialCandidates; i++)
352  {
353  if(vbDiscarded[i])
354  continue;
355 
356  KeyFrame* pKF = mvpEnoughConsistentCandidates[i];
357 
358  // Perform 5 Ransac Iterations
359  vector<bool> vbInliers;
360  int nInliers;
361  bool bNoMore;// 这是局部变量，在pSolver->iterate(...)内进行初始化
362 
363  // 步骤3：对步骤2中有较好的匹配的关键帧求取Sim3变换
364  Sim3Solver* pSolver = vpSim3Solvers[i];
365  // 最多迭代5次，返回的Scm是候选帧pKF到当前帧mpCurrentKF的Sim3变换（T12）
366  cv::Mat Scm = pSolver->iterate(5,bNoMore,vbInliers,nInliers);
367 
368  // If Ransac reachs max. iterations discard keyframe
369  // 经过n次循环，每次迭代5次，总共迭代 n*5 次
370  // 总迭代次数达到最大限制还没有求出合格的Sim3变换，该候选帧剔除
371  if(bNoMore)
372  {
373  vbDiscarded[i]=true;
374  nCandidates--;
375  }
376 
377  // If RANSAC returns a Sim3, perform a guided matching and optimize with all correspondences
378  if(!Scm.empty())
379  {
380  vector<MapPoint*> vpMapPointMatches(vvpMapPointMatches[i].size(), static_cast<MapPoint*>(NULL));
381  for(size_t j=0, jend=vbInliers.size(); j<jend; j++)
382  {
383  // 保存inlier的MapPoint
384  if(vbInliers[j])
385  vpMapPointMatches[j]=vvpMapPointMatches[i][j];
386  }
387 
388  // 步骤4：通过步骤3求取的Sim3变换引导关键帧匹配弥补步骤2中的漏匹配
389  // [sR t;0 1]
390  cv::Mat R = pSolver->GetEstimatedRotation();// 候选帧pKF到当前帧mpCurrentKF的R（R12）
391  cv::Mat t = pSolver->GetEstimatedTranslation();// 候选帧pKF到当前帧mpCurrentKF的t（t12），当前帧坐标系下，方向由pKF指向当前帧
392  const float s = pSolver->GetEstimatedScale();// 候选帧pKF到当前帧mpCurrentKF的变换尺度s（s12）
393  // 查找更多的匹配（成功的闭环匹配需要满足足够多的匹配特征点数，之前使用SearchByBoW进行特征点匹配时会有漏匹配）
394  // 通过Sim3变换，确定pKF1的特征点在pKF2中的大致区域，同理，确定pKF2的特征点在pKF1中的大致区域
395  // 在该区域内通过描述子进行匹配捕获pKF1和pKF2之前漏匹配的特征点，更新匹配vpMapPointMatches
396  matcher.SearchBySim3(mpCurrentKF,pKF,vpMapPointMatches,s,R,t,7.5);
397 
398  // 步骤5：Sim3优化，只要有一个候选帧通过Sim3的求解与优化，就跳出停止对其它候选帧的判断
399  // OpenCV的Mat矩阵转成Eigen的Matrix类型
400  g2o::Sim3 gScm(Converter::toMatrix3d(R),Converter::toVector3d(t),s);
401  // 如果mbFixScale为true，则是6DoFf优化（双目 RGBD），如果是false，则是7DoF优化（单目）
402  // 优化mpCurrentKF与pKF对应的MapPoints间的Sim3，得到优化后的量gScm
403  const int nInliers = Optimizer::OptimizeSim3(mpCurrentKF, pKF, vpMapPointMatches, gScm, 10, mbFixScale);// 卡方chi2检验阈值
404 
405  // If optimization is succesful stop ransacs and continue
406  if(nInliers>=20)
407  {
408  bMatch = true;
409  // mpMatchedKF就是最终闭环检测出来与当前帧形成闭环的关键帧
410  mpMatchedKF = pKF;
411  // 得到从世界坐标系到该候选帧的Sim3变换，Scale=1
412  g2o::Sim3 gSmw(Converter::toMatrix3d(pKF->GetRotation()),Converter::toVector3d(pKF->GetTranslation()),1.0);
413  // 得到g2o优化后从世界坐标系到当前帧的Sim3变换
414  mg2oScw = gScm*gSmw;
415  mScw = Converter::toCvMat(mg2oScw);
416 
417  mvpCurrentMatchedPoints = vpMapPointMatches;
418  break;// 只要有一个候选帧通过Sim3的求解与优化，就跳出停止对其它候选帧的判断
419  }
420  }
421  }
422  }
423 
424  // 没有一个闭环匹配候选帧通过Sim3的求解与优化
425  if(!bMatch)
426  {
427  // 清空mvpEnoughConsistentCandidates
428  for(int i=0; i<nInitialCandidates; i++)
429  mvpEnoughConsistentCandidates[i]->SetErase();
430  mpCurrentKF->SetErase();
431  return false;
432  }
433 
434  // Retrieve MapPoints seen in Loop Keyframe and neighbors
435  // 步骤6：取出闭环匹配上关键帧的相连关键帧，得到它们的MapPoints放入mvpLoopMapPoints
436  // 注意是匹配上的那个关键帧：mpMatchedKF
437  // 将mpMatchedKF相连的关键帧全部取出来放入vpLoopConnectedKFs
438  // 将vpLoopConnectedKFs的MapPoints取出来放入mvpLoopMapPoints
439  vector<KeyFrame*> vpLoopConnectedKFs = mpMatchedKF->GetVectorCovisibleKeyFrames();
440  // 包含闭环匹配关键帧本身
441  vpLoopConnectedKFs.push_back(mpMatchedKF);
442  mvpLoopMapPoints.clear();
443  for(vector<KeyFrame*>::iterator vit=vpLoopConnectedKFs.begin(); vit!=vpLoopConnectedKFs.end(); vit++)
444  {
445  KeyFrame* pKF = *vit;
446  vector<MapPoint*> vpMapPoints = pKF->GetMapPointMatches();
447  for(size_t i=0, iend=vpMapPoints.size(); i<iend; i++)
448  {
449  MapPoint* pMP = vpMapPoints[i];
450  if(pMP)
451  {
452  if(!pMP->isBad() && pMP->mnLoopPointForKF!=mpCurrentKF->mnId)
453  {
454  mvpLoopMapPoints.push_back(pMP);
455  // 标记该MapPoint被mpCurrentKF闭环时观测到并添加，避免重复添加
456  pMP->mnLoopPointForKF=mpCurrentKF->mnId;
457  }
458  }
459  }
460  }
461 
462  // Find more matches projecting with the computed Sim3
463  // 步骤7：将闭环匹配上关键帧以及相连关键帧的MapPoints投影到当前关键帧进行投影匹配
464  // 根据投影查找更多的匹配（成功的闭环匹配需要满足足够多的匹配特征点数）
465  // 根据Sim3变换，将每个mvpLoopMapPoints投影到mpCurrentKF上，并根据尺度确定一个搜索区域，
466  // 根据该MapPoint的描述子与该区域内的特征点进行匹配，如果匹配误差小于TH_LOW即匹配成功，更新mvpCurrentMatchedPoints
467  // mvpCurrentMatchedPoints将用于SearchAndFuse中检测当前帧MapPoints与匹配的MapPoints是否存在冲突
468  matcher.SearchByProjection(mpCurrentKF, mScw, mvpLoopMapPoints, mvpCurrentMatchedPoints,10);// 搜索范围系数为10
469 
470  // If enough matches accept Loop
471  // 步骤8：判断当前帧与检测出的所有闭环关键帧是否有足够多的MapPoints匹配
472  int nTotalMatches = 0;
473  for(size_t i=0; i<mvpCurrentMatchedPoints.size(); i++)
474  {
475  if(mvpCurrentMatchedPoints[i])
476  nTotalMatches++;
477  }
478 
479  // 步骤9：清空mvpEnoughConsistentCandidates
480  if(nTotalMatches>=40)
481  {
482  for(int i=0; i<nInitialCandidates; i++)
483  if(mvpEnoughConsistentCandidates[i]!=mpMatchedKF)
484  mvpEnoughConsistentCandidates[i]->SetErase();
485  return true;
486  }
487  else
488  {
489  for(int i=0; i<nInitialCandidates; i++)
490  mvpEnoughConsistentCandidates[i]->SetErase();
491  mpCurrentKF->SetErase();
492  return false;
493  }

506 {
507  cout << "Loop detected!" << endl;
508 
509  // Send a stop signal to Local Mapping
510  // Avoid new keyframes are inserted while correcting the loop
511  // 步骤0：请求局部地图停止，防止局部地图线程中InsertKeyFrame函数插入新的关键帧
512  mpLocalMapper->RequestStop();
513

515  if(isRunningGBA())

542  // 步骤2：通过位姿传播，得到Sim3优化后，与当前帧相连的关键帧的位姿，以及它们的MapPoints
543  // 当前帧与世界坐标系之间的Sim变换在ComputeSim3函数中已经确定并优化，
544  // 通过相对位姿关系，可以确定这些相连的关键帧与世界坐标系之间的Sim3变换

552  CorrectedSim3[mpCurrentKF]=mg2oScw;

564

568  if(pKFi!=mpCurrentKF)

578  // 得到闭环g2o优化后各个关键帧的位姿
579  CorrectedSim3[pKFi]=g2oCorrectedSiw;
580  }
581

583  cv::Mat tiw = Tiw.rowRange(0,3).col(3);

586  // 当前帧相连关键帧，没有进行闭环g2o优化的位姿

588  }

591  // 步骤2.2：步骤2.1得到调整相连帧位姿后，修正这些关键帧的MapPoints

602  {
603  MapPoint* pMPi = vpMPsi[iMP];
604  if(!pMPi)
605  continue;

607  continue;

609  continue;

612  // 将该未校正的eigP3Dw先从世界坐标系映射到未校正的pKFi相机坐标系，然后再反映射到校正后的世界坐标系下

618  pMPi->SetWorldPos(cvCorrectedP3Dw);
619  pMPi->mnCorrectedByKF = mpCurrentKF->mnId;
620  pMPi->mnCorrectedReference = pKFi->mnId;
621  pMPi->UpdateNormalAndDepth();
622  }

625  // 步骤2.3：将Sim3转换为SE3，根据更新的Sim3，更新关键帧的位姿
626  Eigen::Matrix3d eigR = g2oCorrectedSiw.rotation().toRotationMatrix();
627  Eigen::Vector3d eigt = g2oCorrectedSiw.translation();
628  double s = g2oCorrectedSiw.scale();
629 
630  eigt *=(1./s); //[R t/s;0 1]
631 
632  cv::Mat correctedTiw = Converter::toCvSE3(eigR,eigt);
633 
634  pKFi->SetPose(correctedTiw);
635 
636  // Make sure connections are updated
637  // 步骤2.4：根据共视关系更新当前帧与其它关键帧之间的连接
638  pKFi->UpdateConnections();
639  }
640

642  // Update matched map points and replace if duplicated
643  // 步骤3：检查当前帧的MapPoints与闭环匹配帧的MapPoints是否存在冲突，对冲突的MapPoints进行替换或填补
644  for(size_t i=0; i<mvpCurrentMatchedPoints.size(); i++)
645  {
646  if(mvpCurrentMatchedPoints[i])
647  {
648  MapPoint* pLoopMP = mvpCurrentMatchedPoints[i];
649  MapPoint* pCurMP = mpCurrentKF->GetMapPoint(i);
650  if(pCurMP)// 如果有重复的MapPoint（当前帧和匹配帧各有一个），则用匹配帧的代替现有的
651  pCurMP->Replace(pLoopMP);
652  else// 如果当前帧没有该MapPoint，则直接添加
653  {
654  mpCurrentKF->AddMapPoint(pLoopMP,i);
655  pLoopMP->AddObservation(mpCurrentKF,i);
656  pLoopMP->ComputeDistinctiveDescriptors();
657  }
658  }
659  }
660 
661  }
662 
663  // Project MapPoints observed in the neighborhood of the loop keyframe
664  // into the current keyframe and neighbors using corrected poses.
665  // Fuse duplications.
666  // 步骤4：通过将闭环时相连关键帧的mvpLoopMapPoints投影到这些关键帧中，进行MapPoints检查与替换
667  SearchAndFuse(CorrectedSim3);
668

671  // 步骤5：更新当前关键帧之间的共视相连关系，得到因闭环时MapPoints融合而新得到的连接关系
672  map<KeyFrame*, set<KeyFrame*> > LoopConnections;
673

675  for(vector<KeyFrame*>::iterator vit=mvpCurrentConnectedKFs.begin(), vend=mvpCurrentConnectedKFs.end(); vit!=vend; vit++)
676  {

679  vector<KeyFrame*> vpPreviousNeighbors = pKFi->GetVectorCovisibleKeyFrames();
680

682  // 步骤5.3：更新一级相连关键帧的连接关系
683  pKFi->UpdateConnections();
684  // 步骤5.4：取出该帧更新后的连接关系
685  LoopConnections[pKFi]=pKFi->GetConnectedKeyFrames();
686  // 步骤5.5：从连接关系中去除闭环之前的二级连接关系，剩下的连接就是由闭环得到的连接关系
687  for(vector<KeyFrame*>::iterator vit_prev=vpPreviousNeighbors.begin(), vend_prev=vpPreviousNeighbors.end(); vit_prev!=vend_prev; vit_prev++)
688  {
689  LoopConnections[pKFi].erase(*vit_prev);
690  }
691  // 步骤5.6：从连接关系中去除闭环之前的一级连接关系，剩下的连接就是由闭环得到的连接关系
692  for(vector<KeyFrame*>::iterator vit2=mvpCurrentConnectedKFs.begin(), vend2=mvpCurrentConnectedKFs.end(); vit2!=vend2; vit2++)
693  {
694  LoopConnections[pKFi].erase(*vit2);
695  }
696  }
697 
698  // Optimize graph
699  // 步骤6：进行EssentialGraph优化，LoopConnections是形成闭环后新生成的连接关系，不包括步骤7中当前帧与闭环匹配帧之间的连接关系
700  Optimizer::OptimizeEssentialGraph(mpMap, mpMatchedKF, mpCurrentKF, NonCorrectedSim3, CorrectedSim3, LoopConnections, mbFixScale);
701 
702  // Add loop edge
703  // 步骤7：添加当前帧与闭环匹配帧之间的边（这个连接关系不优化）
704  // 这两句话应该放在OptimizeEssentialGraph之前，因为OptimizeEssentialGraph的步骤4.2中有优化，（wubo???）
705  mpMatchedKF->AddLoopEdge(mpCurrentKF);
706  mpCurrentKF->AddLoopEdge(mpMatchedKF);
707 
708  // Launch a new thread to perform Global Bundle Adjustment
709  // 步骤8：新建一个线程用于全局BA优化
710  // OptimizeEssentialGraph只是优化了一些主要关键帧的位姿，这里进行全局BA可以全局优化所有位姿和MapPoints
711  mbRunningGBA = true;
712  mbFinishedGBA = false;
713  mbStopGBA = false;
714  mpThreadGBA = new thread(&LoopClosing::RunGlobalBundleAdjustment,this,mpCurrentKF->mnId);
715 
716  // Loop closed. Release Local Mapping.
717  mpLocalMapper->Release();
718 
719  cout << "Loop Closed!" << endl;
720 
721  mLastLoopKFid = mpCurrentKF->mnId;
722 }

787 void LoopClosing::RunGlobalBundleAdjustment(unsigned long nLoopKF)
788 {
789  cout << "Starting Global Bundle Adjustment" << endl;
790 
791  int idx = mnFullBAIdx;
792  Optimizer::GlobalBundleAdjustemnt(mpMap,10,&mbStopGBA,nLoopKF,false);
793 
794  // Update all MapPoints and KeyFrames
795  // Local Mapping was active during BA, that means that there might be new keyframes
796  // not included in the Global BA and they are not consistent with the updated map.
797  // We need to propagate the correction through the spanning tree
798  {
799  unique_lock<mutex> lock(mMutexGBA);
800  if(idx!=mnFullBAIdx)
801  return;
802 
803  if(!mbStopGBA)
804  {
805  cout << "Global Bundle Adjustment finished" << endl;
806  cout << "Updating map ..." << endl;
807  mpLocalMapper->RequestStop();
808  // Wait until Local Mapping has effectively stopped
809 
810  while(!mpLocalMapper->isStopped() && !mpLocalMapper->isFinished())
811  {
812  //usleep(1000);
813  std::this_thread::sleep_for(std::chrono::milliseconds(1));
814  }
815 
816  // Get Map Mutex
817  unique_lock<mutex> lock(mpMap->mMutexMapUpdate);
818 
819  // Correct keyframes starting at map first keyframe
820  list<KeyFrame*> lpKFtoCheck(mpMap->mvpKeyFrameOrigins.begin(),mpMap->mvpKeyFrameOrigins.end());
821 
822  while(!lpKFtoCheck.empty())
823  {
824  KeyFrame* pKF = lpKFtoCheck.front();
825  const set<KeyFrame*> sChilds = pKF->GetChilds();
826  cv::Mat Twc = pKF->GetPoseInverse();
827  for(set<KeyFrame*>::const_iterator sit=sChilds.begin();sit!=sChilds.end();sit++)
828  {
829  KeyFrame* pChild = *sit;
830  if(pChild->mnBAGlobalForKF!=nLoopKF)
831  {
832  cv::Mat Tchildc = pChild->GetPose()*Twc;
833  pChild->mTcwGBA = Tchildc*pKF->mTcwGBA;//*Tcorc*pKF->mTcwGBA;
834  pChild->mnBAGlobalForKF=nLoopKF;
835 
836  }
837  lpKFtoCheck.push_back(pChild);
838  }
839 
840  pKF->mTcwBefGBA = pKF->GetPose();
841  pKF->SetPose(pKF->mTcwGBA);
842  lpKFtoCheck.pop_front();
843  }
844 
845  // Correct MapPoints
846  const vector<MapPoint*> vpMPs = mpMap->GetAllMapPoints();
847 
848  for(size_t i=0; i<vpMPs.size(); i++)
849  {
850  MapPoint* pMP = vpMPs[i];
851 
852  if(pMP->isBad())
853  continue;
854 
855  if(pMP->mnBAGlobalForKF==nLoopKF)
856  {
857  // If optimized by Global BA, just update
858  pMP->SetWorldPos(pMP->mPosGBA);
859  }
860  else
861  {
862  // Update according to the correction of its reference keyframe
863  KeyFrame* pRefKF = pMP->GetReferenceKeyFrame();
864 
865  if(pRefKF->mnBAGlobalForKF!=nLoopKF)
866  continue;
867 
868  // Map to non-corrected camera
869  cv::Mat Rcw = pRefKF->mTcwBefGBA.rowRange(0,3).colRange(0,3);
870  cv::Mat tcw = pRefKF->mTcwBefGBA.rowRange(0,3).col(3);
871  cv::Mat Xc = Rcw*pMP->GetWorldPos()+tcw;
872 
873  // Backproject using corrected camera
874  cv::Mat Twc = pRefKF->GetPoseInverse();
875  cv::Mat Rwc = Twc.rowRange(0,3).colRange(0,3);
876  cv::Mat twc = Twc.rowRange(0,3).col(3);
877 
878  pMP->SetWorldPos(Rwc*Xc+twc);
879  }
880  }
881 
882  mpLocalMapper->Release();
883 
884  cout << "Map updated!" << endl;
885  }
886 
887  mbFinishedGBA = true;
888  mbRunningGBA = false;
889  }
